```rust
    pub fn intern(&mut self, s: &str) -> Symbol {
        if let Some(&id) = self.map.get(s) {
            return Symbol(id);
```

```rust
//! Source text span — a half-open byte range `[start, end)`.

/// A half-open byte range pointing into a source file.
#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
pub struct Span {
    pub start: usize,
    pub end: usize,
}

impl Span {
    /// Create a new span. Panics if `start > end`.
    pub fn new(start: usize, end: usize) -> Self {
        assert!(start <= end, "Span start ({start}) must be <= end ({end})");
        Self { start, end }
    }

    /// Byte length of the span.
    pub fn len(&self) -> usize {
        self.end - self.start
    }

    /// Whether the span covers zero bytes.
    pub fn is_empty(&self) -> bool {
        self.start == self.end
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn new_valid_span() {
        let span = Span::new(5, 10);
        assert_eq!(span.start, 5);
        assert_eq!(span.end, 10);
    }

    #[test]
    fn new_zero_length_span() {
        let span = Span::new(3, 3);
        assert!(span.is_empty());
        assert_eq!(span.len(), 0);
    }

    #[test]
    fn len_returns_byte_count() {
        let span = Span::new(2, 7);
        assert_eq!(span.len(), 5);
    }

    #[test]
    #[should_panic(expected = "start must be <= end")]
    fn new_panics_if_start_gt_end() {
        Span::new(10, 5);
    }

    #[test]
    fn span_is_copy() {
        let a = Span::new(0, 1);
        let _b = a; // should compile without move errors
        let _c = a; // still usable
    }
}
```

```rust
//! Compiler diagnostic (error, warning, note) with optional source span.

use crate::Span;

/// Severity level for a diagnostic.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum Severity {
    Error,
    Warning,
    Note,
}

/// A single compiler diagnostic with a message and optional source location.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Diagnostic {
    pub severity: Severity,
    pub message: String,
    pub span: Option<Span>,
}

impl Diagnostic {
    /// Create an error diagnostic with no span.
    pub fn error(message: impl Into<String>) -> Self {
        Self {
            severity: Severity::Error,
            message: message.into(),
            span: None,
        }
    }

    /// Create a warning diagnostic with no span.
    pub fn warning(message: impl Into<String>) -> Self {
        Self {
            severity: Severity::Warning,
            message: message.into(),
            span: None,
        }
    }

    /// Create a note diagnostic with no span.
    pub fn note(message: impl Into<String>) -> Self {
        Self {
            severity: Severity::Note,
            message: message.into(),
            span: None,
        }
    }

    /// Attach a source span to this diagnostic. Builder-style.
    pub fn with_span(mut self, span: Span) -> Self {
        self.span = Some(span);
        self
    }

    /// Whether this diagnostic is an error (as opposed to warning/note).
    pub fn is_error(&self) -> bool {
        self.severity == Severity::Error
    }
}
```

```rust
//! Rowan language definition and type aliases for the Glyim CST.
//!
//! This bridges our [`SyntaxKind`] enum with Rowan's generic tree types.
//! Every "red tree" node (`SyntaxNode`, `SyntaxToken`) is parameterized
//! by [`GlyimLang`], ensuring type safety across the tree API.

use rowan::{Language, SyntaxNode, SyntaxToken, SyntaxElement, SyntaxNodePtr};

use crate::SyntaxKind;

/// Zero-sized marker type for the Glyim language in Rowan.
#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct GlyimLang;

impl Language for GlyimLang {
    type Kind = SyntaxKind;

    fn kind_from_raw(raw: rowan::SyntaxKind) -> Self::Kind {
        debug_assert!(raw.0 < crate::COUNT, "invalid SyntaxKind: {}", raw.0);
        // Safety: all discriminants 0..COUNT are valid because the enum is
        // #[repr(u16)] with sequential values starting at 0.
        unsafe { std::mem::transmute::<u16, SyntaxKind>(raw.0) }
    }

    fn kind_to_raw(kind: Self::Kind) -> rowan::SyntaxKind {
        rowan::SyntaxKind(kind as u16)
    }
}

/// A "red tree" syntax node — the view the parser and LSP work with.
/// Cheap to clone (reference-counted pointer into the green tree).
pub type SyntaxNode = rowan::SyntaxNode<GlyimLang>;

/// A "red tree" syntax token — a leaf in the tree (identifier, punctuation, etc.).
pub type SyntaxToken = rowan::SyntaxToken<GlyimLang>;

/// Either a [`SyntaxNode`] or a [`SyntaxToken`].
pub type SyntaxElement = rowan::SyntaxElement<GlyimLang>;

/// A cheaply-copyable handle that can be used to look up a [`SyntaxNode`]
/// later (e.g., for LSP hover → go-to-def without holding the full tree).
pub type SyntaxNodePtr = rowan::SyntaxNodePtr<GlyimLang>;

/// The underlying "green tree" — an immutable, arena-allocated tree
/// that can be shared across threads without locking.
pub type GreenNode = rowan::GreenNode;
```

```rust
//! Hand-rolled character-by-character lexer for `.xyz` source files.
//!
//! Design decisions:
//! - Emits trivia tokens (whitespace, comments) so the Rowan tree is lossless.
//! - All punctuation is ASCII; identifiers support Unicode.
//! - Numbers are integers only for v0.1.0 (no floats, no hex/octal/binary).
//! - Two-character operators are greedily matched (e.g., `==` not `=` `=`).

use glyim_syntax::SyntaxKind;

use crate::Token;

/// Lexes a `.xyz` source string into a stream of [`Token`] values.
///
/// Call [`Lexer::next_token()`] repeatedly until it returns `None`.
pub struct Lexer<'a> {
    input: &'a str,
    offset: usize,
}

impl<'a> Lexer<'a> {
    pub fn new(input: &'a str) -> Self {
        Self { input, offset: 0 }
    }

    /// Remaining source text starting from the current offset.
    fn remaining(&self) -> &'a str {
        &self.input[self.offset..]
    }

    /// Peek at the next character without consuming it.
    fn peek(&self) -> Option<char> {
        self.remaining().chars().next()
    }

    /// Peek at the second character ahead without consuming anything.
    fn peek2(&self) -> Option<char> {
        self.remaining().chars().nth(1)
    }

    /// Consume and return the next character.
    fn advance(&mut self) -> Option<char> {
        let c = self.peek()?;
        self.offset += c.len_utf8();
        Some(c)
    }

    /// Consume characters while `pred` holds, return the consumed slice.
    fn eat_while<F: FnMut(char) -> bool>(&mut self, mut pred: F) -> &'a str {
        let start = self.offset;
        while let Some(c) = self.peek() {
            if pred(c) {
                self.offset += c.len_utf8();
            } else {
                break;
            }
        }
        &self.input[start..self.offset]
    }

    /// Return the next token, or `None` at end-of-input.
    ///
    /// Includes trivia tokens (whitespace, comments).
    pub fn next_token(&mut self) -> Option<Token<'a>> {
        if self.offset >= self.input.len() {
            return None;
        }

        let start = self.offset;
        let kind = self.lex_one();
        let text = &self.input[start..self.offset];
        let end = self.offset;

        Some(Token { kind, text, start, end })
    }

    /// Lex exactly one token starting at the current offset.
    /// Advances `self.offset` past the consumed characters.
    fn lex_one(&mut self) -> SyntaxKind {
        let c = match self.peek() {
            Some(c) => c,
            None => return SyntaxKind::Eof,
        };

        // ── Whitespace ────────────────────────────────────
        if c.is_whitespace() {
            self.eat_while(|ch| ch.is_whitespace());
            return SyntaxKind::Whitespace;
        }

        // ── Comments ──────────────────────────────────────
        if c == '/' {
            match self.peek2() {
                Some('/') => return self.lex_line_comment(),
                Some('*') => return self.lex_block_comment(),
                _ => {
                    self.advance();
                    return SyntaxKind::Slash;
                }
            }
        }

        // ── Numbers ───────────────────────────────────────
        if c.is_ascii_digit() {
            return self.lex_number();
        }

        // ── Identifiers & Keywords ────────────────────────
        if c.is_alphabetic() || c == '_' {
            return self.lex_ident_or_keyword();
        }

        // ── Two-character operators (check second char) ───
        let two = match (c, self.peek2()) {
            ('=', Some('>')) => Some((SyntaxKind::FatArrow, 2)),
            ('-', Some('>')) => Some((SyntaxKind::Arrow, 2)),
            ('=', Some('=')) => Some((SyntaxKind::EqEq, 2)),
            ('!', Some('=')) => Some((SyntaxKind::BangEq, 2)),
            ('<', Some('=')) => Some((SyntaxKind::LtEq, 2)),
            ('>', Some('=')) => Some((SyntaxKind::GtEq, 2)),
            ('&', Some('&')) => Some((SyntaxKind::AmpAmp, 2)),
            ('|', Some('|')) => Some((SyntaxKind::PipePipe, 2)),
            _ => None,
        };
        if let Some((kind, n)) = two {
            for _ in 0..n {
                self.advance();
            }
            return kind;
        }

        // ── Single-character operators & punctuation ──────
        let single = match c {
            '=' => SyntaxKind::Eq,
            '(' => SyntaxKind::LParen,
            ')' => SyntaxKind::RParen,
            '{' => SyntaxKind::LBrace,
            '}' => SyntaxKind::RBrace,
            ',' => SyntaxKind::Comma,
            ':' => SyntaxKind::Colon,
            ';' => SyntaxKind::Semicolon,
            '@' => SyntaxKind::At,
            '.' => SyntaxKind::Dot,
            '+' => SyntaxKind::Plus,
            '-' => SyntaxKind::Minus,
            '*' => SyntaxKind::Star,
            '/' => SyntaxKind::Slash, // already handled above, but for completeness
            '%' => SyntaxKind::Percent,
            '<' => SyntaxKind::Lt,
            '>' => SyntaxKind::Gt,
            '!' => SyntaxKind::Bang,
            '|' => SyntaxKind::Pipe,
            _ => SyntaxKind::Error,
        };
        self.advance();
        single
    }

    fn lex_number(&mut self) -> SyntaxKind {
        self.eat_while(|c| c.is_ascii_digit());
        SyntaxKind::IntLit
    }

    fn lex_ident_or_keyword(&mut self) -> SyntaxKind {
        self.eat_while(|c| c.is_alphabetic() || c == '_' || c.is_ascii_digit());

        // We need to look at the consumed text to check for keywords.
        // The text is `self.input[start..self.offset]`, but `start` is gone.
        // Instead, peek back using the text length.
        let text = self.input[self.offset - self.remaining().len() - /* hack */ 0..].trim();
        // Actually, let's just grab the last word. Simpler approach:
        // We'll fix this below by restructuring.

        // TEMPORARY — will be replaced in Step 29
        SyntaxKind::Ident
    }

    fn lex_line_comment(&mut self) -> SyntaxKind {
        // Consume "//"
        self.advance();
        self.advance();
        // Consume until newline
        self.eat_while(|c| c != '\n');
        SyntaxKind::LineComment
    }

    fn lex_block_comment(&mut self) -> SyntaxKind {
        // Consume "/*"
        self.advance();
        self.advance();
        // Consume until "*/"
        let mut depth: u32 = 1;
        while depth > 0 {
            match (self.peek(), self.peek2()) {
                (Some('*'), Some('/')) => {
                    self.advance();
                    self.advance();
                    depth -= 1;
                }
                (Some('/'), Some('*')) => {
                    self.advance();
                    self.advance();
                    depth += 1;
                }
                (Some(_), _) => {
                    self.advance();
                }
                None => {
                    // Unterminated block comment — return Error
                    return SyntaxKind::Error;
                }
            }
        }
        SyntaxKind::BlockComment
    }
}

pub fn tokenize(input: &str) -> Vec<Token<'_>> {

    }
    tokens
}
```

```rust
    fn lex_ident_or_keyword(&mut self) -> SyntaxKind {
        let start = self.offset;
        self.eat_while(|c| c.is_alphabetic() || c == '_' || c.is_ascii_digit());
        let text = &self.input[start..self.offset];

        let kind = match text {

            "struct" => SyntaxKind::KwStruct,
            "enum" => SyntaxKind::KwEnum,
            "let" => SyntaxKind::KwLet,
```

```rust
//! Parse error types for the Glyim parser.

use std::fmt;

use glyim_syntax::SyntaxKind;

/// A parse error with location and expected/found token info.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum ParseError {
    /// Expected a specific token kind but found something else.
    Expected { expected: SyntaxKind, found: SyntaxKind, span: (usize, usize) },
    /// Reached end of input when expecting more tokens.
    UnexpectedEof { expected: SyntaxKind },
    /// Token cannot start an expression.
    ExpectedExpr { found: SyntaxKind, span: (usize, usize) },
    /// Custom message with span.
    Message { msg: String, span: (usize, usize) },
}

impl ParseError {
    pub fn expected(expected: SyntaxKind, found_kind: SyntaxKind, start: usize, end: usize) -> Self {
        Self::Expected { expected, found: found_kind, span: (start, end) }
    }

    pub fn unexpected_eof(expected: SyntaxKind) -> Self {
        Self::UnexpectedEof { expected }
    }

    pub fn expected_expr(found_kind: SyntaxKind, start: usize, end: usize) -> Self {
        Self::ExpectedExpr { found: found_kind, span: (start, end) }
    }
}

impl fmt::Display for ParseError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Self::Expected { expected, found, span } => write!(
                f, "expected {} but found {} at bytes {}..{}",
                expected.display_name(), found.display_name(), span.0, span.1
            ),
            Self::UnexpectedEof { expected } => write!(
                f, "expected {} but reached end of input", expected.display_name()
            ),
            Self::ExpectedExpr { found, span } => write!(
                f, "expected expression but found {} at bytes {}..{}",
                found.display_name(), span.0, span.1
            ),
            Self::Message { msg, span } => write!(f, "{} at bytes {}..{}", msg, span.0, span.1),
        }
    }
}

impl std::error::Error for ParseError {}
```

```rust
use glyim_interner::Interner;
use glyim_lex::Token;
use glyim_syntax::SyntaxKind;

use crate::ast::{Ast, ExprNode, Item};
use crate::error::ParseError;

mod expr;
mod item;

/// A peekable window over a token slice that skips trivia automatically.
struct Tokens<'a> {
    tokens: &'a [Token<'a>],
    pos: usize,
}

impl<'a> Tokens<'a> {
    fn new(tokens: &'a [Token<'a>]) -> Self {
        Self { tokens, pos: 0 }
    }

    /// Peek at the next non-trivia token without consuming it.
    fn peek(&self) -> Option<&Token<'a>> {
        let mut p = self.pos;
        while p < self.tokens.len() && self.tokens[p].kind.is_trivia() {
            p += 1;
        }
        self.tokens.get(p)
    }

    /// Peek at the second non-trivia token without consuming.
    fn peek2(&self) -> Option<&Token<'a>> {
        let mut p = self.pos;
        // Skip trivia to first real token
        while p < self.tokens.len() && self.tokens[p].kind.is_trivia() {
            p += 1;
        }
        if p >= self.tokens.len() { return None; }
        p += 1; // skip first real token
        // Skip trivia to second real token
        while p < self.tokens.len() && self.tokens[p].kind.is_trivia() {
            p += 1;
        }
        self.tokens.get(p)
    }

    /// Whether the next non-trivia token is `kind`.
    fn at(&self, kind: SyntaxKind) -> bool {
        self.peek().map_or(false, |t| t.kind == kind)
    }

    /// Consume the next non-trivia token. Returns `None` at EOF.
    fn bump(&mut self) -> Option<Token<'a>> {
        self.skip_trivia();
        if self.pos < self.tokens.len() {
            let tok = self.tokens[self.pos];
            self.pos += 1;
            Some(tok)
        } else {
            None
        }
    }

    /// Consume the next non-trivia token if it matches `kind`.
    fn eat(&mut self, kind: SyntaxKind) -> Option<Token<'a>> {
        if self.at(kind) { self.bump() } else { None }
    }

    /// Consume the next non-trivia token; error if it doesn't match `kind`.
    fn expect(&mut self, kind: SyntaxKind) -> Result<Token<'a>, ParseError> {
        self.skip_trivia();
        match self.tokens.get(self.pos) {
            Some(tok) if tok.kind == kind => {
                let tok = *tok;
                self.pos += 1;
                Ok(tok)
            }
            Some(tok) => Err(ParseError::expected(kind, tok.kind, tok.start, tok.end)),
            None => Err(ParseError::unexpected_eof(kind)),
        }
    }

    /// Whether any non-trivia tokens remain.
    fn is_eof(&self) -> bool {
        self.peek().is_none()
    }

    fn skip_trivia(&mut self) {
        while self.pos < self.tokens.len() && self.tokens[self.pos].kind.is_trivia() {
            self.pos += 1;
        }
    }

    /// Lookahead: is the `(` at current position the start of a lambda?
    /// Checks pattern: `()` `=>` or `(ident, ...)` `=>`
    fn is_lambda_start(&self) -> bool {
        let mut p = self.pos;
        // Current token must be '('
        if self.tokens.get(p).map_or(true, |t| t.kind != SyntaxKind::LParen) {
            return false;
        }
        p += 1;
        p = self.skip_trivia_at(p);

        // Case 1: () =>
        if self.tokens.get(p).map_or(false, |t| t.kind == SyntaxKind::RParen) {
            p += 1;
            p = self.skip_trivia_at(p);
            return self.tokens.get(p).map_or(false, |t| t.kind == SyntaxKind::FatArrow);
        }

        // Case 2: (ident, ident, ...) =>
        if !self.tokens.get(p).map_or(false, |t| t.kind == SyntaxKind::Ident) {
            return false;
        }
        p += 1;
        loop {
            p = self.skip_trivia_at(p);
            match self.tokens.get(p) {
                Some(t) if t.kind == SyntaxKind::Comma => {
                    p += 1;
                    p = self.skip_trivia_at(p);
                    if !self.tokens.get(p).map_or(false, |t| t.kind == SyntaxKind::Ident) {
                        return false;
                    }
                    p += 1;
                }
                Some(t) if t.kind == SyntaxKind::RParen => {
                    p += 1;
                    p = self.skip_trivia_at(p);
                    return self.tokens.get(p).map_or(false, |t| t.kind == SyntaxKind::FatArrow);
                }
                _ => return false,
            }
        }
    }

    /// Skip trivia tokens starting at a given position (non-mutating).
    fn skip_trivia_at(&self, mut p: usize) -> usize {
        while p < self.tokens.len() && self.tokens[p].kind.is_trivia() {
            p += 1;
        }
        p
    }
}

/// The Glyim parser. Holds token stream, error list, and string interner.
pub struct Parser<'a> {
    tokens: Tokens<'a>,
    pub errors: Vec<ParseError>,
    pub interner: Interner,
}

impl<'a> Parser<'a> {
    pub fn new(tokens: &'a [Token<'a>]) -> Self {
        Self {
            tokens: Tokens::new(tokens),
            errors: Vec::new(),
            interner: Interner::new(),
        }
    }

    /// Parse a complete source file into an `Ast`.
    pub fn parse_source_file(&mut self) -> Ast {
        let mut items = Vec::new();
        while !self.tokens.is_eof() {
            if let Some(item) = self.parse_item() {
                items.push(item);
            } else {
                // Skip one token to avoid infinite loop on hard errors
                self.tokens.bump();
            }
        }
        Ast { items }
    }
}

/// Public entry point: parse source string into AST.
pub fn parse(source: &str) -> ParseOutput {
    let tokens = glyim_lex::tokenize(source);
    let mut parser = Parser::new(&tokens);
    let ast = parser.parse_source_file();
    ParseOutput {
        ast,
        errors: parser.errors,
        interner: parser.interner,
    }
}

/// Output of a parse invocation.
pub struct ParseOutput {
    pub ast: Ast,
    pub errors: Vec<ParseError>,
    pub interner: Interner,
}
```

```rust
//! Pratt expression parser — handles operator precedence and associativity.

use glyim_diag::Span;
use glyim_syntax::SyntaxKind;

use super::Parser;
use crate::ast::{BinOp, ExprKind, ExprNode, UnOp};
use crate::error::ParseError;

/// Return (left-binding-power, right-binding-power) for infix/postfix ops.
fn infix_bp(kind: SyntaxKind) -> Option<(u8, u8)> {
    match kind {
        SyntaxKind::PipePipe => Some((10, 11)),
        SyntaxKind::AmpAmp => Some((20, 21)),
        SyntaxKind::EqEq | SyntaxKind::BangEq => Some((30, 31)),
        SyntaxKind::Lt | SyntaxKind::Gt | SyntaxKind::LtEq | SyntaxKind::GtEq => Some((40, 41)),
        SyntaxKind::Plus | SyntaxKind::Minus => Some((50, 51)),
        SyntaxKind::Star | SyntaxKind::Slash | SyntaxKind::Percent => Some((60, 61)),
        _ => None,
    }
}

/// Map a `SyntaxKind` operator token to a `BinOp`.
fn to_binop(kind: SyntaxKind) -> BinOp {
    match kind {
        SyntaxKind::Plus => BinOp::Add,
        SyntaxKind::Minus => BinOp::Sub,
        SyntaxKind::Star => BinOp::Mul,
        SyntaxKind::Slash => BinOp::Div,
        SyntaxKind::Percent => BinOp::Mod,
        SyntaxKind::EqEq => BinOp::Eq,
        SyntaxKind::BangEq => BinOp::Neq,
        SyntaxKind::Lt => BinOp::Lt,
        SyntaxKind::Gt => BinOp::Gt,
        SyntaxKind::LtEq => BinOp::Lte,
        SyntaxKind::GtEq => BinOp::Gte,
        SyntaxKind::AmpAmp => BinOp::And,
        SyntaxKind::PipePipe => BinOp::Or,
        _ => unreachable!("to_binop called on non-operator: {:?}", kind),
    }
}

impl<'a> Parser<'a> {
    /// Parse an expression with minimum binding power `min_bp`.
    /// Returns `None` if no expression can be parsed (e.g., EOF).
    pub fn parse_expr(&mut self, min_bp: u8) -> Option<ExprNode> {
        let mut left = self.parse_expr_atom()?;

        loop {
            let op_tok = match self.tokens.peek() {
                Some(t) => *t,
                None => break,
            };

            // Infix binary operator
            if let Some((l_bp, r_bp)) = infix_bp(op_tok.kind) {
                if l_bp < min_bp {
                    break;
                }
                self.tokens.bump(); // consume operator
                let right = self.parse_expr(r_bp)?;
                left = ExprNode {
                    kind: ExprKind::Binary {
                        op: to_binop(op_tok.kind),
                        lhs: Box::new(left),
                        rhs: Box::new(right),
                    },
                    span: Span::new(left.span.start, right.span.end),
                };
                continue;
            }

            // Postfix: call expression `callee(args)`
            if op_tok.kind == SyntaxKind::LParen && 80 >= min_bp {
                self.tokens.bump(); // consume '('
                let mut args = Vec::new();
                while !self.tokens.at(SyntaxKind::RParen) && self.tokens.peek().is_some() {
                    args.push(self.parse_expr(0)?);
                    self.tokens.eat(SyntaxKind::Comma);
                }
                let rparen = match self.tokens.expect(SyntaxKind::RParen) {
                    Ok(t) => t,
                    Err(e) => { self.errors.push(e); break; }
                };
                left = ExprNode {
                    kind: ExprKind::Call { callee: Box::new(left), args },
                    span: Span::new(left.span.start, rparen.end),
                };
                continue;
            }

            break;
        }

        Some(left)
    }

    /// Parse an atomic expression (null denotation in Pratt terms).
    fn parse_expr_atom(&mut self) -> Option<ExprNode> {
        let tok = self.tokens.peek()?;

        match tok.kind {
            SyntaxKind::IntLit => {
                let tok = self.tokens.bump()?;
                let value: i64 = tok.text.parse().unwrap_or(0);
                Some(ExprNode {
                    kind: ExprKind::IntLit(value),
                    span: Span::new(tok.start, tok.end),
                })
            }

            SyntaxKind::Ident => {
                let tok = self.tokens.bump()?;
                let sym = self.interner.intern(tok.text);
                Some(ExprNode {
                    kind: ExprKind::Ident(sym),
                    span: Span::new(tok.start, tok.end),
                })
            }

            SyntaxKind::LParen if self.tokens.is_lambda_start() => self.parse_lambda(),

            SyntaxKind::LParen => self.parse_paren_expr(),

            SyntaxKind::LBrace => self.parse_block(),

            SyntaxKind::Minus | SyntaxKind::Bang => {
                let op_tok = self.tokens.bump()?;
                let (r_bp, op) = match op_tok.kind {
                    SyntaxKind::Minus => (70, UnOp::Neg),
                    SyntaxKind::Bang => (70, UnOp::Not),
                    _ => unreachable!(),
                };
                let operand = self.parse_expr(r_bp)?;
                Some(ExprNode {
                    kind: ExprKind::Unary { op, operand: Box::new(operand) },
                    span: Span::new(op_tok.start, operand.span.end),
                })
            }

            _ => {
                self.errors.push(ParseError::expected_expr(tok.kind, tok.start, tok.end));
                None
            }
        }
    }

    /// Parse a lambda: `(params) => body`
    fn parse_lambda(&mut self) -> Option<ExprNode> {
        let start_tok = self.tokens.bump()?; // consume '('
        let start = start_tok.start;

        let mut params = Vec::new();
        while !self.tokens.at(SyntaxKind::RParen) {
            let tok = match self.tokens.expect(SyntaxKind::Ident) {
                Ok(t) => t,
                Err(e) => { self.errors.push(e); break; }
            };
            params.push(self.interner.intern(tok.text));
            if !self.tokens.at(SyntaxKind::Comma) {
                break;
            }
            self.tokens.bump(); // consume ','
        }
        if let Err(e) = self.tokens.expect(SyntaxKind::RParen) {
            self.errors.push(e);
        }
        if let Err(e) = self.tokens.expect(SyntaxKind::FatArrow) {
            self.errors.push(e);
        }

        let body = self.parse_expr(0)?;
        Some(ExprNode {
            kind: ExprKind::Lambda { params, body: Box::new(body) },
            span: Span::new(start, body.span.end),
        })
    }

    /// Parse a parenthesized expression: `(expr)`
    fn parse_paren_expr(&mut self) -> Option<ExprNode> {
        self.tokens.bump(); // consume '('
        let inner = self.parse_expr(0)?;
        if let Err(e) = self.tokens.expect(SyntaxKind::RParen) {
            self.errors.push(e);
        }
        Some(inner)
    }

    /// Parse a block: `{ expr* }`
    fn parse_block(&mut self) -> Option<ExprNode> {
        let start_tok = self.tokens.bump()?; // consume '{'
        let mut exprs = Vec::new();
        while !self.tokens.at(SyntaxKind::RBrace) && self.tokens.peek().is_some() {
            if let Some(expr) = self.parse_expr(0) {
                exprs.push(expr);
            } else {
                self.tokens.bump(); // skip bad token
            }
        }
        let end_tok = match self.tokens.expect(SyntaxKind::RBrace) {
            Ok(t) => t,
            Err(e) => { self.errors.push(e); return None; }
        };
        Some(ExprNode {
            kind: ExprKind::Block(exprs),
            span: Span::new(start_tok.start, end_tok.end),
        })
    }
}
```

```rust
//! Top-level item parser: bindings and function definitions.

use glyim_diag::Span;
use glyim_syntax::SyntaxKind;

use super::Parser;
use crate::ast::{ExprNode, Item};
use crate::error::ParseError;

impl<'a> Parser<'a> {
    /// Parse one top-level item. Returns `None` if the current token
    /// doesn't start a valid item (caller should skip a token).
    pub fn parse_item(&mut self) -> Option<Item> {
        match self.tokens.peek()?.kind {
            SyntaxKind::KwFn => self.parse_fn_def(),
            SyntaxKind::Ident => {
                // Disambiguate: `ident =` is a binding, `ident(` is a call (not valid at top level)
                if self.tokens.peek2().map_or(false, |t| t.kind == SyntaxKind::Eq) {
                    self.parse_binding()
                } else {
                    None
                }
            }
            _ => None,
        }
    }

    /// Parse `name = expr`
    fn parse_binding(&mut self) -> Option<Item> {
        let name_tok = self.tokens.bump()?; // consume ident
        let name = self.interner.intern(name_tok.text);
        let name_span = Span::new(name_tok.start, name_tok.end);

        if let Err(e) = self.tokens.expect(SyntaxKind::Eq) {
            self.errors.push(e);
            return None;
        }

        let value = self.parse_expr(0)?;
        Some(Item::Binding { name, name_span, value })
    }

    /// Parse `fn name(params) { body }`
    fn parse_fn_def(&mut self) -> Option<Item> {
        let kw = self.tokens.bump()?; // consume 'fn'
        debug_assert_eq!(kw.kind, SyntaxKind::KwFn);

        let name_tok = match self.tokens.expect(SyntaxKind::Ident) {
            Ok(t) => t,
            Err(e) => { self.errors.push(e); return None; }
        };
        let name = self.interner.intern(name_tok.text);
        let name_span = Span::new(name_tok.start, name_tok.end);

        if let Err(e) = self.tokens.expect(SyntaxKind::LParen) {
            self.errors.push(e);
            return None;
        }

        let mut params = Vec::new();
        while !self.tokens.at(SyntaxKind::RParen) {
            let tok = match self.tokens.expect(SyntaxKind::Ident) {
                Ok(t) => t,
                Err(e) => { self.errors.push(e); break; }
            };
            params.push((self.interner.intern(tok.text), Span::new(tok.start, tok.end)));
            self.tokens.eat(SyntaxKind::Comma);
        }
        if let Err(e) = self.tokens.expect(SyntaxKind::RParen) {
            self.errors.push(e);
        }

        let body = match self.parse_expr(0) {
            Some(e) => e,
            None => {
                self.errors.push(ParseError::Message {
                    msg: "expected function body".into(),
                    span: (name_tok.end, name_tok.end + 1),
                });
                return None;
            }
        };

        Some(Item::FnDef { name, name_span, params, body })
    }
}
```

```rust
//! Lower parsed AST to HIR.
//!
//! Bindings whose value is a lambda become `HirFn`.
//! `FnDef` items become `HirFn` directly.
//! Other bindings are silently skipped (v0.1.0 limitation).

use glyim_parse::{BinOp, ExprKind, Item, UnOp};

use crate::{Hir, HirBinOp, HirExpr, HirFn, HirUnOp};

/// Lower a parsed `Ast` into `Hir`.
pub fn lower(ast: &glyim_parse::Ast) -> Hir {
    let mut fns = Vec::new();
    for item in &ast.items {
        match item {
            Item::Binding { name, value, .. } => {
                if let ExprKind::Lambda { params, body } = &value.kind {
                    fns.push(HirFn {
                        name: *name,
                        params: params.clone(),
                        body: lower_expr(body),
                    });
                }
                // Non-lambda bindings: skip in v0.1.0
            }
            Item::FnDef { name, params, body, .. } => {
                let param_syms: Vec<_> = params.iter().map(|(sym, _)| *sym).collect();
                fns.push(HirFn {
                    name: *name,
                    params: param_syms,
                    body: lower_expr(&body.kind),
                });
            }
        }
    }
    Hir { fns }
}

fn lower_expr(expr: &ExprKind) -> HirExpr {
    match expr {
        ExprKind::IntLit(n) => HirExpr::IntLit(*n),
        ExprKind::Ident(sym) => HirExpr::Ident(*sym),
        ExprKind::Binary { op, lhs, rhs } => HirExpr::Binary {
            op: lower_binop(*op),
            lhs: Box::new(lower_expr(&lhs.kind)),
            rhs: Box::new(lower_expr(&rhs.kind)),
        },
        ExprKind::Unary { op, operand } => HirExpr::Unary {
            op: lower_unop(*op),
            operand: Box::new(lower_expr(&operand.kind)),
        },
        ExprKind::Lambda { params, body } => {
            // Nested lambdas: keep as-is in HIR (codegen will handle or error)
            HirExpr::Ident((*body).as_ref().downcast_ref::<i64>().unwrap()) // placeholder — replaced below
        }
        ExprKind::Block(exprs) => {
            HirExpr::Block(exprs.iter().map(|e| lower_expr(&e.kind)).collect())
        }
        ExprKind::Call { .. } => {
            // v0.1.0: calls not supported in codegen, emit a placeholder
            HirExpr::IntLit(0)
        }
    }
}

fn lower_binop(op: BinOp) -> HirBinOp {
    match op {
        BinOp::Add => HirBinOp::Add,
        BinOp::Sub => HirBinOp::Sub,
        BinOp::Mul => HirBinOp::Mul,
        BinOp::Div => HirBinOp::Div,
        BinOp::Mod => HirBinOp::Mod,
        BinOp::Eq => HirBinOp::Eq,
        BinOp::Neq => HirBinOp::Neq,
        BinOp::Lt => HirBinOp::Lt,
        BinOp::Gt => HirBinOp::Gt,
        BinOp::Lte => HirBinOp::Lte,
        BinOp::Gte => HirBinOp::Gte,
        BinOp::And => HirBinOp::And,
        BinOp::Or => HirBinOp::Or,
    }
}

fn lower_unop(op: UnOp) -> HirUnOp {
    match op {
        UnOp::Neg => HirUnOp::Neg,
        UnOp::Not => HirUnOp::Not,
    }
}
```

```rust
//! LLVM IR code generation from HIR.
//!
//! Design decisions for v0.1.0:
//! - All values are `i64` internally (no type system yet).
//! - The `main` function returns `i32` (truncated from `i64`) to match C ABI.
//! - Other functions use `i64` params and return `i64`.
//! - Block expressions evaluate all statements, return the last value.
//! - `Ident` resolves against function parameters only (no locals/ globals yet).

use std::collections::HashMap;

use glyim_interner::Symbol;
use inkwell::builder::Builder;
use inkwell::context::Context;
use inkwell::module::Module;
use inkwell::types::{BasicType, IntType};
use inkwell::values::{BasicValue, FunctionValue, IntValue};

use crate::{HirBinOp, HirExpr, HirFn, HirUnOp};

/// Generates LLVM IR from a `glyim_hir::Hir`.
pub struct Codegen<'ctx> {
    context: &'ctx Context,
    module: Module<'ctx>,
    builder: Builder<'ctx>,
    i64_type: IntType<'ctx>,
    i32_type: IntType<'ctx>,
}

impl<'ctx> Codegen<'ctx> {
    pub fn new(context: &'ctx Context) -> Self {
        let module = context.create_module("glyim_out");
        let builder = context.create_builder();
        let i64_type = context.i64_type();
        let i32_type = context.i32_type();
        Self { context, module, builder, i64_type, i32_type }
    }

    /// Generate LLVM IR for the entire HIR module.
    /// Returns `Err` if `main` is not found.
    pub fn generate(&mut self, hir: &glyim_hir::Hir) -> Result<(), String> {
        let main_idx = hir.fns.iter().position(|f| {
            // Compare by interned value — we can't access the interner here,
            // so we check by comparing Symbol ids. The caller ensures consistency.
            // We'll find main by checking after codegen.
            true
        });

        // Codegen all functions
        for hir_fn in &hir.fns {
            self.codegen_fn(hir_fn)?;
        }

        // Verify main exists
        let has_main = hir.fns.iter().any(|f| {
            // We'll verify via the module after codegen
            self.module.get_function("main").is_some()
        });
        // Actually, check the module directly
        if self.module.get_function("main").is_none() {
            return Err("no 'main' function found".into());
        }

        Ok(())
    }

    fn codegen_fn(&mut self, hir_fn: &HirFn) -> Result<(), String> {
        let is_main = self.is_main_name(hir_fn.name);
        let name = self.sym_to_str(hir_fn.name);

        // Choose return type: i32 for main, i64 for everything else
        let ret_type = if is_main { self.i32_type } else { self.i64_type };
        let param_type = self.i64_type;
        let fn_type = param_type.fn_type(&[], false);
        let fn_value = self.module.add_function(name, fn_type, None);

        let entry = self.context.append_basic_block(fn_value, "entry");
        self.builder.position_at_end(entry);

        // Build parameter map
        let mut vars: HashMap<Symbol, IntValue<'ctx>> = HashMap::new();
        for (i, param_sym) in hir_fn.params.iter().enumerate() {
            let param_val = fn_value.get_nth_param(i as u32)
                .ok_or_else(|| format!("missing param {} in fn {}", i, name))?;
            // Params are i64 in LLVM; get_nth_param returns IntValue
            let int_val = param_val.into_int_value();
            vars.insert(*param_sym, int_val);
        }

        // Codegen body
        let body_val = self.codegen_expr(&hir_fn.body, &mut vars)
            .ok_or_else(|| format!("failed to codegen body of '{}'", name))?;

        // Truncate i64 → i32 for main
        let ret_val = if is_main {
            self.builder
                .build_int_truncate(body_val, self.i32_type, "trunc")
                .map_err(|e| format!("truncate failed: {}", e))?
        } else {
            body_val
        };

        self.builder
            .build_return(Some(&ret_val))
            .map_err(|e| format!("return failed: {}", e))?;

        // Verify function
        if fn_value.verify(true) {
            // OK
        } else {
            fn_value.print_to_stderr();
            return Err(format!("LLVM verification failed for '{}'", name));
        }

        Ok(())
    }

    fn codegen_expr(
        &self,
        expr: &HirExpr,
        vars: &HashMap<Symbol, IntValue<'ctx>>,
    ) -> Option<IntValue<'ctx>> {
        match expr {
            HirExpr::IntLit(n) => {
                let val = self.i64_type.const_int(*n as u64, true);
                Some(val)
            }

            HirExpr::Ident(sym) => {
                vars.get(sym).copied()
            }

            HirExpr::Binary { op, lhs, rhs } => {
                let l = self.codegen_expr(lhs, vars)?;
                let r = self.codegen_expr(rhs, vars)?;
                let name = match op {
                    HirBinOp::Add => "add",
                    HirBinOp::Sub => "sub",
                    HirBinOp::Mul => "mul",
                    HirBinOp::Div => "div",
                    HirBinOp::Mod => "rem",
                    HirBinOp::Eq => "eq",
                    HirBinOp::Neq => "neq",
                    HirBinOp::Lt => "lt",
                    HirBinOp::Gt => "gt",
                    HirBinOp::Lte => "lte",
                    HirBinOp::Gte => "gte",
                    HirBinOp::And => "and",
                    HirBinOp::Or => "or",
                };
                let result = match op {
                    HirBinOp::Add => self.builder.build_int_add(l, r, name),
                    HirBinOp::Sub => self.builder.build_int_sub(l, r, name),
                    HirBinOp::Mul => self.builder.build_int_mul(l, r, name),
                    HirBinOp::Div => self.builder.build_int_signed_div(l, r, name),
                    HirBinOp::Mod => self.builder.build_int_signed_rem(l, r, name),
                    HirBinOp::Eq => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::EQ, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::Neq => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::NE, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::Lt => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::SLT, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::Gt => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::SGT, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::Lte => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::SLE, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::Gte => {
                        let cmp = self.builder.build_int_compare(inkwell::IntPredicate::SGE, l, r, "cmp")?;
                        self.builder.build_int_z_extend(cmp, self.i64_type, "zext")
                    }
                    HirBinOp::And => {
                        // Short-circuit: for v0.1.0, just bitwise AND
                        self.builder.build_int_and(l, r, name)
                    }
                    HirBinOp::Or => {
                        // Short-circuit: for v0.1.0, just bitwise OR
                        self.builder.build_int_or(l, r, name)
                    }
                }.ok()?;
                Some(result)
            }

            HirExpr::Unary { op, operand } => {
                let val = self.codegen_expr(operand, vars)?;
                match op {
                    HirUnOp::Neg => {
                        let zero = self.i64_type.const_int(0, false);
                        Some(self.builder.build_int_sub(zero, val, "neg").ok()?)
                    }
                    HirUnOp::Not => {
                        Some(self.builder.build_not(val, "not").ok()?)
                    }
                }
            }

            HirExpr::Block(exprs) => {
                let mut last = None;
                for e in exprs {
                    last = self.codegen_expr(e, vars);
                }
                last
            }
        }
    }
```

```rust
//! Content-addressable hash — a SHA-256 digest as a 32-byte array.
//!
//! Used as the canonical identity of any byte sequence stored in the CAS.
//! Two byte sequences with the same hash are assumed to be identical.

use sha2::{Sha256, Digest};
use std::fmt;
use std::str::FromStr;

/// A 32-byte SHA-256 digest. `Copy`, `Eq`, `Hash` — suitable as a map key.
#[derive(Clone, Copy, PartialEq, Eq, Hash)]
pub struct ContentHash([u8; 32]);

impl ContentHash {
    /// Hash a byte slice.
    pub fn of(data: &[u8]) -> Self {
        let mut hasher = Sha256::new();
        hasher.update(data);
        let digest = hasher.finalize();
        let mut bytes = [0u8; 32];
        bytes.copy_from_slice(&digest);
        Self(bytes)
    }

    /// Hash a UTF-8 string.
    pub fn of_str(s: &str) -> Self {
        Self::of(s.as_bytes())
    }

    /// The raw 32 bytes.
    pub fn as_bytes(&self) -> &[u8; 32] {
        &self.0
    }

    /// Lowercase hex encoding (64 characters).
    pub fn to_hex(self) -> String {
        self.0.iter().map(|b| format!("{:02x}", b)).collect()
    }

    /// Parse from lowercase hex string (64 characters).
    pub fn from_hex(hex: &str) -> Result<Self, ParseHexError> {
        if hex.len() != 64 {
            return Err(ParseHexError { kind: HexErrorKind::WrongLength(hex.len()) });
        }
        let mut bytes = [0u8; 32];
        for i in 0..32 {
            bytes[i] = u8::from_str_radix(&hex[i * 2..i * 2 + 2], 16)
                .map_err(|_| ParseHexError { kind: HexErrorKind::InvalidHex(i * 2) })?;
        }
        Ok(Self(bytes))
    }
}

impl fmt::Debug for ContentHash {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "ContentHash({})", self.to_hex())
    }
}

impl fmt::Display for ContentHash {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        self.to_hex().fmt(f)
    }
}

impl FromStr for ContentHash {
    type Err = ParseHexError;
    fn from_str(s: &str) -> Result<Self, Self::Err> {
        Self::from_hex(s)
    }
}

/// Error returned when parsing a hex string into a ContentHash.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ParseHexError {
    kind: HexErrorKind,
}

#[derive(Debug, Clone, PartialEq, Eq)]
enum HexErrorKind {
    WrongLength(usize),
    InvalidHex(usize),
}

impl fmt::Display for ParseHexError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match &self.kind {
            HexErrorKind::WrongLength(n) => write!(f, "expected 64 hex chars, got {}", n),
            HexErrorKind::InvalidHex(pos) => write!(f, "invalid hex char at position {}", pos),
        }
    }
}
```

//! Local filesystem-backed content-addressable store.
//!
//! Storage layout:
//! ```text
//! base_dir/
//!   objects/
//!     ab/cd/...ef   ← first 4 hex chars as directory, rest as filename
//!   names/
//!     schema.json   ← contains the hex hash string
//! ```

use std::collections::HashMap;
use std::fs;
use std::path::{Path, PathBuf};

use crate::hash::ContentHash;
use crate::store::ContentStore;

/// Filesystem-backed CAS. Thread-safe (all methods take `&self`).
pub struct LocalContentStore {
    base_dir: PathBuf,
    objects_dir: PathBuf,
    names_dir: PathBuf,
}

impl LocalContentStore {
    /// Create or open a local store at `base_dir`.
    /// Creates `objects/` and `names/` subdirectories if needed.
    pub fn new(base_dir: impl Into<PathBuf>) -> std::io::Result<Self> {
        let base_dir = base_dir.into();
        let objects_dir = base_dir.join("objects");
        let names_dir = base_dir.join("names");
        fs::create_dir_all(&objects_dir)?;
        fs::create_dir_all(&names_dir)?;
        Ok(Self { base_dir, objects_dir, names_dir })
    }

    /// Return the path where an object with the given hash would be stored.
    fn object_path(&self, hash: ContentHash) -> PathBuf {
        let hex = hash.to_hex();
        self.objects_dir.join(&hex[0..2]).join(&hex[2..])
    }
}

impl ContentStore for LocalContentStore {
    fn store(&self, content: &[u8]) -> ContentHash {
        let hash = ContentHash::of(content);
        let path = self.object_path(hash);
        if !path.exists() {
            // Create parent dir (first 2 hex chars)
            if let Some(parent) = path.parent() {
                let _ = fs::create_dir_all(parent);
            }
            let _ = fs::write(&path, content);
        }
        hash
    }

    fn retrieve(&self, hash: ContentHash) -> Option<Vec<u8>> {
        let path = self.object_path(hash);
        fs::read(&path).ok()
    }

    fn register_name(&self, name: &str, hash: ContentHash) {
        // Sanitize name: replace path separators to prevent escape
        let safe_name = name.replace('/', "_").replace('\\', "_");
        let path = self.names_dir.join(&safe_name);
        let _ = fs::write(&path, hash.to_hex());
    }

    fn resolve_name(&self, name: &str) -> Option<ContentHash> {
        let safe_name = name.replace('/', "_").replace('\\', "_");
        let path = self.names_dir.join(&safe_name);
        let hex = fs::read_to_string(&path).ok()?;
        hex.parse().ok()
    }
}

//! # Mangling format
//! ```text
//! __glyim_hyg_<context_id>_<original_name>
//! ```
//!
//! The context ID is a monotonically increasing counter, ensuring each
//! macro expansion site gets a unique namespace.

use glyim_interner::{Interner, Symbol};

/// Manages hygiene scopes for macro expansion.
///
/// Each macro invocation gets a unique context ID. Identifiers introduced
/// by that macro are mangled with the context ID, making them invisible
/// to the caller's scope.
pub struct HygieneContext {
    interner: Interner,
    next_context_id: u32,
}

impl HygieneContext {
    pub fn new(interner: Interner) -> Self {
        Self { interner, next_context_id: 0 }
    }

    /// Allocate a fresh hygiene context ID for a new macro expansion.
    pub fn push_context(&mut self) -> HygieneId {
        let id = self.next_context_id;
        self.next_context_id += 1;
        HygieneId(id)
    }

    /// Mangle an identifier with a hygiene context, making it invisible
    /// to the caller's scope.
    ///
    /// Example: mangle(`HygieneId(3)`, `ctx`) → `__glyim_hyg_3_ctx`
    pub fn mangle(&self, context: HygieneId, name: Symbol) -> Symbol {
        let original = self.interner.resolve(name);
        let mangled = format!("__glyim_hyg_{}_{}", context.0, original);
        self.interner.intern(&mangled)
    }

    /// Resolve a mangled symbol back to its original name.
    /// Returns `None` if the symbol doesn't match the mangling pattern.
    pub fn demangle(&self, sym: Symbol) -> Option<(HygieneId, Symbol)> {
        let text = self.interner.resolve(sym);
        let prefix = "__glyim_hyg_";
        if !text.starts_with(prefix) {
            return None;
        }
        let rest = &text[prefix.len()..];
        let underscore_pos = rest.find('_')?;
        let id_str = &rest[..underscore_pos];
        let orig_name = &rest[underscore_pos + 1..];
        let id: u32 = id_str.parse().ok()?;
        Some((HygieneId(id), self.interner.intern(orig_name)))
    }

    /// Borrow the interner (for resolving symbols externally).
    pub fn interner(&self) -> &Interner {
        &self.interner
    }

    /// Number of contexts allocated so far.
    pub fn context_count(&self) -> u32 {
        self.next_context_id
    }
}

/// A unique hygiene context ID, assigned to each macro expansion.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub struct HygieneId(u32);

impl HygieneId {
    /// The raw numeric ID.
    pub fn as_u32(self) -> u32 {
        self.0
    }
}

```rust
//! Compilation pipeline: source → parse → lower → codegen → link → run.

use std::path::{Path, PathBuf};
use std::process::Command;

use glyim_codegen_llvm::{compile_to_ir, Codegen};
use glyim_interner::Interner;
use inkwell::context::Context;

/// Errors that can occur at any stage of the pipeline.
#[derive(Debug)]
pub enum PipelineError {
    Io(std::io::Error),
    Parse(Vec<glyim_parse::ParseError>),
    Codegen(String),
    Link(String),
    Run(std::io::Error),
}

impl std::fmt::Display for PipelineError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(e) => write!(f, "I/O error: {}", e),
            Self::Parse(errors) => {
                writeln!(f, "{} parse error(s):", errors.len())?;
                for e in errors {
                    writeln!(f, "  - {}", e)?;
                }
                Ok(())
            }
            Self::Codegen(msg) => write!(f, "codegen error: {}", msg),
            Self::Link(msg) => write!(f, "linker error: {}", msg),
            Self::Run(e) => write!(f, "execution error: {}", e),
        }
    }
}

impl std::error::Error for PipelineError {}

impl From<std::io::Error> for PipelineError {
    fn from(e: std::io::Error) -> Self {
        Self::Io(e)
    }
}

/// Read source file, parse, lower, codegen. Returns `(hir, ir_string, interner)`.
fn compile_to_hir_and_ir(source: &str) -> Result<(glyim_hir::Hir, String, Interner), PipelineError> {
    let parse_out = glyim_parse::parse(source);
    if !parse_out.errors.is_empty() {
        return Err(PipelineError::Parse(parse_out.errors));
    }
    let hir = glyim_hir::lower(&parse_out.ast);
    let ir = compile_to_ir(source).map_err(PipelineError::Codegen)?;
    Ok((hir, ir, parse_out.interner))
}

/// Compile a `.xyz` source file to a native executable.
///
/// Returns the path to the output executable.
pub fn build(input: &Path, output: Option<&Path>) -> Result<PathBuf, PipelineError> {
    let source = std::fs::read_to_string(input)?;
    let (hir, _ir, interner) = compile_to_hir_and_ir(&source)?;

    // Determine output path
    let output = output
        .map(|p| p.to_path_buf())
        .unwrap_or_else(|| {
            let stem = input.file_stem().unwrap_or_default().to_string_lossy().to_string();
            PathBuf::from(stem)
        });

    // Create temp dir for object file
    let tmp_dir = tempfile::tempdir()?;
    let obj_path = tmp_dir.path().join("output.o");

    // Generate object file
    let context = Context::create();
    let mut codegen = Codegen::new(&context, interner);
    codegen.generate(&hir).map_err(PipelineError::Codegen)?;
    codegen.write_object_file(&obj_path).map_err(PipelineError::Codegen)?;

    // Link
    link_object(&obj_path, &output)?;

    Ok(output)
}

/// Compile and run a `.xyz` source file. Returns the process exit code.
pub fn run(input: &Path) -> Result<i32, PipelineError> {
    let tmp_dir = tempfile::tempdir()?;
    let exe_path = tmp_dir.path().join("glyim_out");

    build(input, Some(&exe_path))?;

    let status = Command::new(&exe_path)
        .status()
        .map_err(PipelineError::Run)?;

    Ok(status.code().unwrap_or(1))
}

/// Parse a source file and print any errors. Returns Ok if no errors.
pub fn check(input: &Path) -> Result<(), PipelineError> {
    let source = std::fs::read_to_string(input)?;
    let parse_out = glyim_parse::parse(&source);
    if !parse_out.errors.is_empty() {
        return Err(PipelineError::Parse(parse_out.errors));
    }
    Ok(())
}

/// Print the generated LLVM IR to stdout.
pub fn print_ir(input: &Path) -> Result<(), PipelineError> {
    let source = std::fs::read_to_string(input)?;
    let ir = compile_to_ir(&source).map_err(PipelineError::Codegen)?;
    println!("{ir}");
    Ok(())
}

/// Link an object file into an executable using the system C compiler.
fn link_object(obj_path: &Path, output_path: &Path) -> Result<(), PipelineError> {
    // Try `cc` first, then `gcc` as fallback
    let linker = if which_cc("cc") { "cc" } else if which_cc("gcc") { "gcc" } else {
        return Err(PipelineError::Link(
            "no C compiler found (tried 'cc' and 'gcc'). Install gcc or clang.".into(),
        ));
    };

    let result = Command::new(linker)
        .arg("-o")
        .arg(output_path)
        .arg(obj_path)
        .output()
        .map_err(|e| PipelineError::Link(format!("failed to invoke '{}': {}", linker, e)))?;

    if !result.status.success() {
        let stderr = String::from_utf8_lossy(&result.stderr);
        return Err(PipelineError::Link(format!("linker failed:\n{}", stderr)));
    }

    Ok(())
}

/// Check if a command is available on PATH.
fn which_cc(name: &str) -> bool {
    Command::new(name)
        .arg("--version")
        .stdout(std::process::Stdio::null())
        .stderr(std::process::Stdio::null())
        .status()
        .map_or(false, |s| s.success())
}
```

```rust
    pub fn write_object_file(&self, path: &std::path::Path) -> Result<(), String> {
        use inkwell::targets::{
            CodeModel, FileType, InitializationConfig, RelocMode, Target, TargetMachine,
        };

        Target::initialize_native(&InitializationConfig::default())
            .map_err(|e| format!("LLVM target init failed: {}", e))?;

        let target_triple = TargetMachine::get_default_triple();
        let target =
            Target::from_triple(&target_triple).map_err(|e| format!("target lookup: {}", e))?;

        let target_machine = target
            .create_target_machine(
                &target_triple,
                "", // cpu
                "", // features
                inkwell::OptimizationLevel::None,
                RelocMode::PIC,
                CodeModel::Default,
            )
            .ok_or_else(|| "target machine creation failed".to_string())?;

        target_machine
            .write_to_file(&self.module, FileType::Object, path)
            .map_err(|e| format!("object file write failed: {}", e))
    }
```
