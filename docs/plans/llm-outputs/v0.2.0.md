```rust
    fn lex_string(&mut self) -> SyntaxKind {
        let start = self.offset;
        debug_assert_eq!(self.input.as_bytes()[start], b'"');
        self.advance(); // consume opening "

        loop {
            match self.peek() {
                None => {
                    // Unterminated string — consumed everything
                    return SyntaxKind::Error;
                }
                Some('"') => {
                    self.advance(); // consume closing "
                    return SyntaxKind::StringLit;
                }
                Some('\\') => {
                    self.advance(); // consume backslash
                    match self.peek() {
                        Some('"') | Some('\\') | Some('n') | Some('t') | Some('r') | Some('0') => {
                            self.advance(); // consume escaped char
                        }
                        _ => {
                            // Invalid escape — keep going (report as error later)
                            // For v0.2.0, just consume the next char
                            if self.peek().is_some() {
                                self.advance();
                            }
                        }
                    }
                }
                Some('\n') => {
                    // Literal newline in string is an error
                    return SyntaxKind::Error;
                }
                Some(_) => {
                    self.advance();
                }
            }
        }
    }
```

```rust
//! High-level Intermediate Representation — the codegen's input.
//!
//! HIR is a simplified, span-free representation of the AST.
//! Spans are stripped because codegen doesn't need them.
//! Lambdas are lowered to top-level functions.
//! Blocks contain statements (Let, Assign, Expr).

use glyim_interner::Symbol;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum HirBinOp {
    Add, Sub, Mul, Div, Mod,
    Eq, Neq, Lt, Gt, Lte, Gte,
    And, Or,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum HirUnOp {
    Neg,
    Not,
}

/// A statement — the unit of a block body.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum HirStmt {
    /// `let name = value` or `let mut name = value`
    Let { name: Symbol, mutable: bool, value: HirExpr },
    /// `target = value` (reassignment)
    Assign { target: Symbol, value: HirExpr },
    /// An expression used for its side effects or as the block's return value.
    Expr(HirExpr),
}

/// An expression in the HIR. All values are `i64` for v0.1.0; strings are `&str` fat pointers from v0.2.0.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum HirExpr {
    IntLit(i64),
    StrLit(String),
    Ident(Symbol),
    Binary { op: HirBinOp, lhs: Box<HirExpr>, rhs: Box<HirExpr> },
    Unary { op: HirUnOp, operand: Box<HirExpr> },
    Block(Vec<HirStmt>),
    /// `if cond { then } else { else }` — always produces a value (0 if no else).
    If {
        condition: Box<HirExpr>,
        then_branch: Box<HirExpr>,
        else_branch: Option<Box<HirExpr>>,
    },
}

/// A function in the HIR. Lambdas from the AST become HirFn here.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct HirFn {
    pub name: Symbol,
    pub params: Vec<Symbol>,
    pub body: HirExpr,
}

/// The complete HIR module — a collection of functions.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Hir {
    pub fns: Vec<HirFn>,
}
```

```rust
//! Lower parsed AST to HIR.

use glyim_parse::{BinOp, ExprKind, Item, UnOp};

use crate::{Hir, HirBinOp, HirExpr, HirFn, HirStmt, HirUnOp};

/// Lower a parsed `Ast` into `Hir`.
pub fn lower(ast: &glyim_parse::Ast) -> Hir {
    let mut fns = Vec::new();
    for item in &ast.items {
        match item {
            Item::Binding { name, value, .. } => {
                if let ExprKind::Lambda { params, body } = &value.kind {
                    fns.push(HirFn {
                        name: *name,
                        params: params.clone(),
                        body: lower_expr(body),
                    });
                }
            }
            Item::FnDef { name, params, body, .. } => {
                let param_syms: Vec<_> = params.iter().map(|(sym, _)| *sym).collect();
                fns.push(HirFn {
                    name: *name,
                    params: param_syms,
                    body: lower_expr(&body.kind),
                });
            }
        }
    }
    Hir { fns }
}

fn lower_expr(expr: &ExprKind) -> HirExpr {
    match expr {
        ExprKind::IntLit(n) => HirExpr::IntLit(*n),
        ExprKind::Ident(sym) => HirExpr::Ident(*sym),
        ExprKind::Binary { op, lhs, rhs } => HirExpr::Binary {
            op: lower_binop(*op),
            lhs: Box::new(lower_expr(&lhs.kind)),
            rhs: Box::new(lower_expr(&rhs.kind)),
        },
        ExprKind::Unary { op, operand } => HirExpr::Unary {
            op: lower_unop(*op),
            operand: Box::new(lower_expr(&operand.kind)),
        },
        ExprKind::Lambda { params, body } => lower_expr(&body.kind),
        ExprKind::Block(exprs) => {
            let stmts: Vec<HirStmt> = exprs
                .iter()
                .map(|e| HirStmt::Expr(lower_expr(&e.kind)))
                .collect();
            HirExpr::Block(stmts)
        }
        ExprKind::Call { .. } => HirExpr::IntLit(0),
    }
}

fn lower_binop(op: BinOp) -> HirBinOp {
    match op {
        BinOp::Add => HirBinOp::Add,
        BinOp::Sub => HirBinOp::Sub,
        BinOp::Mul => HirBinOp::Mul,
        BinOp::Div => HirBinOp::Div,
        BinOp::Mod => HirBinOp::Mod,
        BinOp::Eq => HirBinOp::Eq,
        BinOp::Neq => HirBinOp::Neq,
        BinOp::Lt => HirBinOp::Lt,
        BinOp::Gt => HirBinOp::Gt,
        BinOp::Lte => HirBinOp::Lte,
        BinOp::Gte => HirBinOp::Gte,
        BinOp::And => HirBinOp::And,
        BinOp::Or => HirBinOp::Or,
    }
}

fn lower_unop(op: UnOp) -> HirUnOp {
    match op {
        UnOp::Neg => HirUnOp::Neg,
        UnOp::Not => HirUnOp::Not,
    }
}
```

//! Render diagnostics using `ariadne` for beautiful source snippets.
//!
//! Produces output like:
//! ```text
//! error: expected `=` after binding name
//!  --> src/main.xyz:2:8
//!   |
//! 2 |   let x 42
//!   |        ^^ expected `=`
//! ```

use ariadne::{Label, Report, ReportKind, Source};
use std::collections::HashMap;

use crate::{Diagnostic, Severity, Span};

/// Render a list of diagnostics for a source file into a human-readable string.
pub fn render_diagnostics(
    source: &str,
    file_path: &str,
    diagnostics: &[Diagnostic],
) -> String {
    let mut out = Vec::new();
    let cache = {
        let mut m = HashMap::new();
        m.insert(file_path.to_string(), Source::from(source));
        m
    };

    for diag in diagnostics {
        let kind = match diag.severity {
            Severity::Error => ReportKind::Error,
            Severity::Warning => ReportKind::Warning,
            Severity::Note => ReportKind::Advice,
        };

        let byte_offset = diag.span.map_or(0, |s| s.start);

        let mut report = Report::build(kind, file_path, byte_offset)
            .with_message(&diag.message);

        if let Some(span) = diag.span {
            if span.start != span.end {
                report = report.with_label(
                    Label::new((file_path, span.start..span.end))
                        .with_message(&diag.message),
                );
            }
        }

        report
            .finish()
            .write((&cache[file_path], &cache), &mut out)
            .unwrap();
    }

    String::from_utf8(out).unwrap_or_default()
}

```rust
//! Wrapper around Rowan's `GreenNodeBuilder` for Glyim CST construction.
//!
//! Provides a typed API: `start_node(SyntaxKind)`, `token(SyntaxKind, &str)`,
//! `finish_node()`. Tracks errors separately from the tree.

use glyim_syntax::{GreenNode, SyntaxKind, SyntaxNode};

use crate::ParseError;

/// Builds a Rowan GreenNode tree from a stream of syntax events.
pub struct CstBuilder {
    builder: rowan::GreenNodeBuilder<'static>,
    errors: Vec<ParseError>,
}

impl CstBuilder {
    pub fn new() -> Self {
        Self {
            builder: rowan::GreenNodeBuilder::new(),
            errors: Vec::new(),
        }
    }

    /// Start a composite node (e.g., `SourceFile`, `BinaryExpr`).
    pub fn start_node(&mut self, kind: SyntaxKind) {
        self.builder.start_node(glyim_syntax::GlyimLang::kind_to_raw(kind));
    }

    /// Emit a leaf token.
    pub fn token(&mut self, kind: SyntaxKind, text: &str) {
        self.builder.token(glyim_syntax::GlyimLang::kind_to_raw(kind), text);
    }

    /// Finish the current composite node.
    pub fn finish_node(&mut self) {
        self.builder.finish_node();
    }

    /// Record a parse error (does not affect the tree structure).
    pub fn error(&mut self, err: ParseError) {
        self.errors.push(err);
    }

    /// Emit an error node containing the given tokens.
    /// Wraps them in an `Error` composite node.
    pub fn error_node(&mut self, tokens: &[(SyntaxKind, &str)], err: ParseError) {
        self.start_node(SyntaxKind::Error);
        for (kind, text) in tokens {
            self.token(*kind, text);
        }
        self.finish_node();
        self.errors.push(err);
    }

    /// Complete the tree and return the root GreenNode + collected errors.
    pub fn finish(mut self) -> (GreenNode, Vec<ParseError>) {
        let green = self.builder.finish();
        (green, self.errors)
    }
}

impl Default for CstBuilder {
    fn default() -> Self {
        Self::new()
    }
}

/// Convert a GreenNode into a red-tree SyntaxNode for traversal.
pub fn green_to_syntax(green: GreenNode) -> SyntaxNode {
    SyntaxNode::new_root(green)
}
```

```rust
//! Lower a Rowan SyntaxNode (CST) to a typed Ast (AST enum).
//!
//! This pass skips error nodes and trivia, producing a clean typed tree.
//! It is the bridge between the lossless CST (for LSP) and the
//! compiler pipeline (which needs typed AST enums).

use glyim_diag::Span;
use glyim_syntax::SyntaxKind;
use glyim_syntax::SyntaxNode;

use crate::ast::{Ast, BinOp, ExprKind, ExprNode, Item, UnOp};
use crate::error::ParseError;

/// Lower a CST root node to an AST. Returns any errors found in the tree.
pub fn lower_cst_to_ast(root: &SyntaxNode) -> (Ast, Vec<ParseError>) {
    let mut errors = Vec::new();
    let mut items = Vec::new();

    for child in root.children() {
        match child.kind() {
            SyntaxKind::FnDef => {
                if let Some(item) = lower_fn_def(&child, &mut errors) {
                    items.push(item);
                }
            }
            SyntaxKind::Error => {
                errors.push(ParseError::Message {
                    msg: "invalid top-level construct".into(),
                    span: text_range_to_span(&child),
                });
            }
            _ => {
                // Try to lower as a binding (ident = expr)
                if let Some(item) = lower_binding_at_top(&child, &mut errors) {
                    items.push(item);
                }
            }
        }
    }

    (Ast { items }, errors)
}

fn text_range_to_span(node: &SyntaxNode) -> (usize, usize) {
    let range = node.text_range();
    (range.start().into(), range.end().into())
}

fn token_text_range_to_span(node: &SyntaxNode) -> Span {
    let range = node.text_range();
    Span::new(range.start().into(), range.end().into())
}

/// Try to lower a top-level node as `ident = expr` binding.
fn lower_binding_at_top(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<Item> {
    let mut children = node.children_with_tokens().filter(|n| !n.kind().is_trivia());

    let name_tok = children.next()?;
    if name_tok.kind() != SyntaxKind::Ident {
        return None;
    }
    let name_text = name_tok.as_token()?.text();
    let name_span = Span::new(
        name_tok.text_range().start().into(),
        name_tok.text_range().end().into(),
    );

    let eq_tok = children.next()?;
    if eq_tok.kind() != SyntaxKind::Eq {
        return None;
    }

    // The rest should be a single expression subtree
    let value_node = node.children().find(|n| {
        !matches!(n.kind(), SyntaxKind::Ident | SyntaxKind::Eq)
    })?;

    let value = lower_expr(&value_node, errors)?;
    let name = /* need interner — see note below */ 0.into();

    Some(Item::Binding { name, name_span, value })
}

fn lower_fn_def(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<Item> {
    let mut children = node.children_with_tokens().filter(|n| !n.kind().is_trivia());

    // Skip 'fn' keyword
    let kw = children.next()?;
    debug_assert_eq!(kw.kind(), SyntaxKind::KwFn);

    let name_tok = children.next()?.as_token()?;
    if name_tok.kind() != SyntaxKind::Ident {
        errors.push(ParseError::expected(SyntaxKind::Ident, name_tok.kind(),
            name_tok.text_range().start().into(), name_tok.text_range().end().into()));
        return None;
    }

    // For Phase 2, we return a placeholder — the real implementation
    // needs the interner, which we'll wire in the next step.
    None
}

/// Lower a CST expression node to an AST ExprNode.
fn lower_expr(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    match node.kind() {
        SyntaxKind::IntLit => {
            let text = node.first_token()?.text();
            let value: i64 = text.parse().unwrap_or(0);
            Some(ExprNode {
                kind: ExprKind::IntLit(value),
                span: token_text_range_to_span(node),
            })
        }
        SyntaxKind::Ident => {
            let text = node.first_token()?.text();
            let sym = /* interner needed */ 0.into();
            Some(ExprNode {
                kind: ExprKind::Ident(sym),
                span: token_text_range_to_span(node),
            })
        }
        SyntaxKind::BinaryExpr => lower_binary(node, errors),
        SyntaxKind::UnaryExpr => lower_unary(node, errors),
        SyntaxKind::LambdaExpr => lower_lambda(node, errors),
        SyntaxKind::BlockExpr => lower_block(node, errors),
        SyntaxKind::CallExpr => lower_call(node, errors),
        SyntaxKind::LitExpr => {
            // Dive into the literal node
            let child = node.children().next()?;
            lower_expr(&child, errors)
        }
        SyntaxKind::ParenExpr => {
            // Dive into the inner expression
            let child = node.children().find(|n| n.kind() != SyntaxKind::LParen && n.kind() != SyntaxKind::RParen)?;
            lower_expr(&child, errors)
        }
        SyntaxKind::Error => {
            errors.push(ParseError::Message {
                msg: format!("error node in expression: {:?}", node.text()),
                span: text_range_to_span(node),
            });
            None
        }
        other => {
            errors.push(ParseError::Message {
                msg: format!("unexpected node kind in expression: {:?}", other),
                span: text_range_to_span(node),
            });
            None
        }
    }
}

fn lower_binary(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    let mut children: Vec<_> = node.children().collect();
    // BinaryExpr: [lhs, op_token, rhs]
    if children.len() < 3 {
        errors.push(ParseError::Message {
            msg: "malformed binary expression".into(),
            span: text_range_to_span(node),
        });
        return None;
    }
    let lhs = lower_expr(&children[0], errors)?;
    let op_tok = children[1].first_token()?;
    let op = match op_tok.kind() {
        SyntaxKind::Plus => BinOp::Add,
        SyntaxKind::Minus => BinOp::Sub,
        SyntaxKind::Star => BinOp::Mul,
        SyntaxKind::Slash => BinOp::Div,
        SyntaxKind::Percent => BinOp::Mod,
        SyntaxKind::EqEq => BinOp::Eq,
        SyntaxKind::BangEq => BinOp::Neq,
        SyntaxKind::Lt => BinOp::Lt,
        SyntaxKind::Gt => BinOp::Gt,
        SyntaxKind::LtEq => BinOp::Lte,
        SyntaxKind::GtEq => BinOp::Gte,
        SyntaxKind::AmpAmp => BinOp::And,
        SyntaxKind::PipePipe => BinOp::Or,
        other => {
            errors.push(ParseError::Message {
                msg: format!("not a binary operator: {:?}", other),
                span: (op_tok.text_range().start().into(), op_tok.text_range().end().into()),
            });
            return None;
        }
    };
    let rhs = lower_expr(&children[2], errors)?;
    Some(ExprNode {
        kind: ExprKind::Binary { op, lhs: Box::new(lhs), rhs: Box::new(rhs) },
        span: Span::new(lhs.span.start, rhs.span.end),
    })
}

fn lower_unary(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    let mut children: Vec<_> = node.children().collect();
    let op_tok = children.first()?.first_token()?;
    let op = match op_tok.kind() {
        SyntaxKind::Minus => UnOp::Neg,
        SyntaxKind::Bang => UnOp::Not,
        other => {
            errors.push(ParseError::Message {
                msg: format!("not a unary operator: {:?}", other),
                span: text_range_to_span(node),
            });
            return None;
        }
    };
    let operand = lower_expr(&children[1], errors)?;
    Some(ExprNode {
        kind: ExprKind::Unary { op, operand: Box::new(operand) },
        span: Span::new(op_tok.text_range().start().into(), operand.span.end),
    })
}

fn lower_lambda(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    // LambdaExpr: '(' params ')' '=>' body
    let mut children: Vec<_> = node.children_with_tokens()
        .filter(|n| !n.kind().is_trivia())
        .collect();

    // Skip '('
    children.remove(0);

    // Collect params until ')'
    let mut params = Vec::new();
    while let Some(child) = children.first() {
        match child.kind() {
            SyntaxKind::Ident => {
                let sym = /* interner */ 0.into();
                params.push(sym);
                children.remove(0);
                // Skip comma if present
                if children.first().map_or(false, |c| c.kind() == SyntaxKind::Comma) {
                    children.remove(0);
                }
            }
            SyntaxKind::RParen => {
                children.remove(0);
                break;
            }
            _ => break,
        }
    }

    // Skip '=>'
    children.remove(0);

    // Remaining is the body expression
    let body_node = children.first()?.as_node()?;
    let body = lower_expr(body_node, errors)?;

    Some(ExprNode {
        kind: ExprKind::Lambda { params, body: Box::new(body) },
        span: token_text_range_to_span(node),
    })
}

fn lower_block(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    let exprs: Vec<_> = node.children()
        .filter(|n| n.kind() != SyntaxKind::LBrace && n.kind() != SyntaxKind::RBrace)
        .filter_map(|n| lower_expr(&n, errors))
        .collect();
    Some(ExprNode {
        kind: ExprKind::Block(exprs),
        span: token_text_range_to_span(node),
    })
}

fn lower_call(node: &SyntaxNode, errors: &mut Vec<ParseError>) -> Option<ExprNode> {
    let mut children: Vec<_> = node.children().collect();
    let callee = lower_expr(&children[0], errors)?;
    // children[1] is the param list node; collect its children as args
    let param_list = &children[1];
    let args: Vec<_> = param_list.children()
        .filter(|n| n.kind() != SyntaxKind::LParen && n.kind() != SyntaxKind::RParen && n.kind() != SyntaxKind::Comma)
        .filter_map(|n| lower_expr(&n, errors))
        .collect();
    Some(ExprNode {
        kind: ExprKind::Call { callee: Box::new(callee), args },
        span: Span::new(callee.span.start, token_text_range_to_span(node).end),
    })
}
```

```rust
use glyim_syntax::SyntaxKind;

use crate::cst_builder::CstBuilder;
use crate::parser::Tokens;
use crate::recovery;

/// Binding power for infix operators. Returns (left_bp, right_bp).
fn infix_bp(kind: SyntaxKind) -> Option<(u8, u8)> {
    match kind {
        SyntaxKind::PipePipe => Some((10, 11)),
        SyntaxKind::AmpAmp => Some((20, 21)),
        SyntaxKind::EqEq | SyntaxKind::BangEq => Some((30, 31)),
        SyntaxKind::Lt | SyntaxKind::Gt | SyntaxKind::LtEq | SyntaxKind::GtEq => Some((40, 41)),
        SyntaxKind::Plus | SyntaxKind::Minus => Some((50, 51)),
        SyntaxKind::Star | SyntaxKind::Slash | SyntaxKind::Percent => Some((60, 61)),
        _ => None,
    }
}

impl<'a> Tokens<'a> {
    /// Parse an expression with minimum binding power, building Rowan nodes.
    pub fn parse_expr_cst(&self, builder: &mut CstBuilder, min_bp: u8) -> bool {
        if !self.parse_expr_atom_cst(builder) {
            return false;
        }

        loop {
            let op_kind = match self.peek() {
                Some(t) if !t.kind.is_trivia() => t.kind,
                _ => break,
            };

            if let Some((l_bp, r_bp)) = infix_bp(op_kind) {
                if l_bp < min_bp {
                    break;
                }
                let tok = self.bump().unwrap();
                builder.token(op_kind, tok.text);

                if !self.parse_expr_cst(builder, r_bp) {
                    return false; // missing RHS

                //
                // SOLUTION: Use a two-pass approach: parse into a Vec of events,
                // then build the tree. OR, build the tree bottom-up.
                //

                // We do this by having the caller always start a node, and we

                // This approach doesn't work well with Rowan's push model.

        }

    /// Parse an atomic expression into the CST.
    fn parse_expr_atom_cst(&self, builder: &mut CstBuilder) -> bool {
        let tok = match self.peek() {
            Some(t) if !t.kind.is_trivia() => t,
            _ => return false,
        };

        match tok.kind {
            SyntaxKind::IntLit => {
                let tok = self.bump().unwrap();
                builder.start_node(SyntaxKind::LitExpr);
                builder.token(SyntaxKind::IntLit, tok.text);
                builder.finish_node();
                true
            }
            SyntaxKind::StringLit => {
                let tok = self.bump().unwrap();
                builder.start_node(SyntaxKind::LitExpr);
                builder.token(SyntaxKind::StringLit, tok.text);
                builder.finish_node();
                true
            }
            SyntaxKind::Ident => {
                let tok = self.bump().unwrap();
                builder.start_node(SyntaxKind::PathExpr);
                builder.token(SyntaxKind::Ident, tok.text);
                builder.finish_node();
                true
            }
            SyntaxKind::LParen if self.is_lambda_start() => self.parse_lambda_cst(builder),
            SyntaxKind::LParen => self.parse_paren_cst(builder),
            SyntaxKind::LBrace => self.parse_block_cst(builder),
            SyntaxKind::Minus | SyntaxKind::Bang => {
                let tok = self.bump().unwrap();
                builder.start_node(SyntaxKind::UnaryExpr);
                builder.token(tok.kind, tok.text);
                self.parse_expr_cst(builder, 70);
                builder.finish_node();
                true
            }
            _ => false,
        }
    }

    fn parse_lambda_cst(&self, builder: &mut CstBuilder) -> bool {
        builder.start_node(SyntaxKind::LambdaExpr);
        let tok = self.bump(); // '('
        if let Some(t) = tok { builder.token(t.kind, t.text); }

        while !self.at(SyntaxKind::RParen) && self.peek().is_some() {
            if self.at(SyntaxKind::Ident) {
                let t = self.bump().unwrap();
                builder.token(SyntaxKind::Ident, t.text);
            }
            if self.at(SyntaxKind::Comma) {
                let t = self.bump().unwrap();
                builder.token(SyntaxKind::Comma, t.text);
            }
        }
        if let Some(t) = self.bump_if(SyntaxKind::RParen) {
            builder.token(SyntaxKind::RParen, t.text);
        }
        if let Some(t) = self.bump_if(SyntaxKind::FatArrow) {
            builder.token(SyntaxKind::FatArrow, t.text);
        }

        self.parse_expr_cst(builder, 0);
        builder.finish_node();
        true
    }

    fn parse_paren_cst(&self, builder: &mut CstBuilder) -> bool {
        builder.start_node(SyntaxKind::ParenExpr);
        if let Some(t) = self.bump_if(SyntaxKind::LParen) {
            builder.token(SyntaxKind::LParen, t.text);
        }
        self.parse_expr_cst(builder, 0);
        if let Some(t) = self.bump_if(SyntaxKind::RParen) {
            builder.token(SyntaxKind::RParen, t.text);
        }
        builder.finish_node();
        true
    }

    fn parse_block_cst(&self, builder: &mut CstBuilder) -> bool {
        builder.start_node(SyntaxKind::BlockExpr);
        if let Some(t) = self.bump_if(SyntaxKind::LBrace) {
            builder.token(SyntaxKind::LBrace, t.text);
        }
        while !self.at(SyntaxKind::RBrace) && self.peek().is_some() {
            self.parse_expr_cst(builder, 0);
        }
        if let Some(t) = self.bump_if(SyntaxKind::RBrace) {
            builder.token(SyntaxKind::RBrace, t.text);
        }
        builder.finish_node();
        true
    }

    /// Helper: consume token if it matches kind.
    fn bump_if(&self, kind: SyntaxKind) -> Option<glyim_lex::Token<'a>> {
        if self.at(kind) { self.bump() } else { None }
    }
}
```

```rust
//! Convert a typed AST back into a Rowan CST.
//!
//! This is the inverse of `ast_lower`. It proves the Rowan infrastructure
//! works end-to-end. In v0.3.0, the parser will build Rowan trees directly;
//! for v0.2.0, this round-trip validates the tree structure.

use glyim_syntax::{GreenNode, SyntaxKind};
use glyim_syntax::SyntaxNode;

use crate::ast::{Ast, BinOp, ExprKind, ExprNode, Item, UnOp};
use crate::cst_builder::CstBuilder;

/// Convert an AST into a Rowan GreenNode.
pub fn ast_to_green(ast: &Ast) -> GreenNode {
    let mut builder = CstBuilder::new();
    builder.start_node(SyntaxKind::SourceFile);
    for item in &ast.items {
        ast_item_to_cst(&mut builder, item);
    }
    builder.finish_node();
    let (green, _) = builder.finish();
    green
}

/// Convert an AST into a red-tree SyntaxNode (for traversal).
pub fn ast_to_cst(ast: &Ast) -> SyntaxNode {
    let green = ast_to_green(ast);
    crate::cst_builder::green_to_syntax(green)
}

fn ast_item_to_cst(builder: &mut CstBuilder, item: &Item) {
    match item {
        Item::Binding { name, value, .. } => {
            // Emit ident token (we can't resolve the Symbol without interner,
            // so we emit a placeholder — the real parser would have the text)
            builder.start_node(SyntaxKind::PathExpr);
            builder.token(SyntaxKind::Ident, "<binding>");
            builder.finish_node();
            builder.token(SyntaxKind::Eq, "=");
            ast_expr_to_cst(builder, value);
        }
        Item::FnDef { name, params, body, .. } => {
            builder.start_node(SyntaxKind::FnDef);
            builder.token(SyntaxKind::KwFn, "fn");
            builder.start_node(SyntaxKind::PathExpr);
            builder.token(SyntaxKind::Ident, "<fn_name>");
            builder.finish_node();
            builder.token(SyntaxKind::LParen, "(");
            for (i, _) in params.iter().enumerate() {
                if i > 0 { builder.token(SyntaxKind::Comma, ","); }
                builder.token(SyntaxKind::Ident, "<param>");
            }
            builder.token(SyntaxKind::RParen, ")");
            ast_expr_to_cst(builder, body);
            builder.finish_node();
        }
    }
}

fn ast_expr_to_cst(builder: &mut CstBuilder, expr: &ExprNode) {
    match &expr.kind {
        ExprKind::IntLit(n) => {
            builder.start_node(SyntaxKind::LitExpr);
            builder.token(SyntaxKind::IntLit, &n.to_string());
            builder.finish_node();
        }
        ExprKind::Ident(_) => {
            builder.start_node(SyntaxKind::PathExpr);
            builder.token(SyntaxKind::Ident, "<ident>");
            builder.finish_node();
        }
        ExprKind::Binary { op, lhs, rhs } => {
            builder.start_node(SyntaxKind::BinaryExpr);
            ast_expr_to_cst(builder, lhs);
            builder.token(binop_kind(*op), binop_text(*op));
            ast_expr_to_cst(builder, rhs);
            builder.finish_node();
        }
        ExprKind::Unary { op, operand } => {
            builder.start_node(SyntaxKind::UnaryExpr);
            builder.token(unop_kind(*op), unop_text(*op));
            ast_expr_to_cst(builder, operand);
            builder.finish_node();
        }
        ExprKind::Lambda { params, body } => {
            builder.start_node(SyntaxKind::LambdaExpr);
            builder.token(SyntaxKind::LParen, "(");
            for (i, _) in params.iter().enumerate() {
                if i > 0 { builder.token(SyntaxKind::Comma, ","); }
                builder.token(SyntaxKind::Ident, "<param>");
            }
            builder.token(SyntaxKind::RParen, ")");
            builder.token(SyntaxKind::FatArrow, "=>");
            ast_expr_to_cst(builder, body);
            builder.finish_node();
        }
        ExprKind::Block(exprs) => {
            builder.start_node(SyntaxKind::BlockExpr);
            builder.token(SyntaxKind::LBrace, "{");
            for e in exprs {
                ast_expr_to_cst(builder, e);
            }
            builder.token(SyntaxKind::RBrace, "}");
            builder.finish_node();
        }
        ExprKind::Call { callee, args } => {
            builder.start_node(SyntaxKind::CallExpr);
            ast_expr_to_cst(builder, callee);
            builder.token(SyntaxKind::LParen, "(");
            for (i, a) in args.iter().enumerate() {
                if i > 0 { builder.token(SyntaxKind::Comma, ","); }
                ast_expr_to_cst(builder, a);
            }
            builder.token(SyntaxKind::RParen, ")");
            builder.finish_node();
        }
    }
}

fn binop_kind(op: BinOp) -> SyntaxKind {
    match op {
        BinOp::Add => SyntaxKind::Plus,
        BinOp::Sub => SyntaxKind::Minus,
        BinOp::Mul => SyntaxKind::Star,
        BinOp::Div => SyntaxKind::Slash,
        BinOp::Mod => SyntaxKind::Percent,
        BinOp::Eq => SyntaxKind::EqEq,
        BinOp::Neq => SyntaxKind::BangEq,
        BinOp::Lt => SyntaxKind::Lt,
        BinOp::Gt => SyntaxKind::Gt,
        BinOp::Lte => SyntaxKind::LtEq,
        BinOp::Gte => SyntaxKind::GtEq,
        BinOp::And => SyntaxKind::AmpAmp,
        BinOp::Or => SyntaxKind::PipePipe,
    }
}

fn binop_text(op: BinOp) -> &'static str {
    match op {
        BinOp::Add => "+", BinOp::Sub => "-", BinOp::Mul => "*",
        BinOp::Div => "/", BinOp::Mod => "%",
        BinOp::Eq => "==", BinOp::Neq => "!=",
        BinOp::Lt => "<", BinOp::Gt => ">",
        BinOp::Lte => "<=", BinOp::Gte => ">=",
        BinOp::And => "&&", BinOp::Or => "||",
    }
}

fn unop_kind(op: UnOp) -> SyntaxKind {
    match op { UnOp::Neg => SyntaxKind::Minus, UnOp::Not => SyntaxKind::Bang }
}

fn unop_text(op: UnOp) -> &'static str {
    match op { UnOp::Neg => "-", UnOp::Not => "!" }
}
```

```rust
    /// Parse a complete source file into an `Ast`.
    /// Uses error recovery: continues parsing after errors to report multiple issues.
    pub fn parse_source_file(&mut self) -> Ast {
        let mut items = Vec::new();
        while !self.tokens.is_eof() {
            if let Some(item) = self.parse_item() {
                items.push(item);
            } else {
                // Error recovery: skip to next sync point
                self.errors.push(ParseError::Message {
                    msg: "failed to parse item".into(),
                    span: self.current_span(),
                });
                self.recover();
            }
        }
        Ast { items }
    }

    /// Skip tokens until a sync point is found.
    fn recover(&mut self) {
        loop {
            match self.tokens.peek() {
                None => break,
                Some(tok) if recovery::is_sync_point(tok.kind) => break,
                Some(tok) if recovery::is_block_end(tok.kind) => {
                    self.tokens.bump(); // consume the '}'
                    break;
                }
                _ => {
                    self.tokens.bump();
                }
            }
        }
    }

    /// Get the span of the current token for error reporting.
    fn current_span(&self) -> (usize, usize) {
        match self.tokens.peek() {
            Some(tok) => (tok.start, tok.end),
            None => (self.tokens.offset, self.tokens.offset),
        }
    }
```

```rust
    /// Parse `if cond { then } [else { else }]`.
    fn parse_if_expr(&mut self) -> Option<ExprNode> {
        let start = self.tokens.bump()?; // consume 'if'
        debug_assert_eq!(start.kind, SyntaxKind::KwIf);

        let condition = self.parse_expr(0)?;

        if let Err(e) = self.tokens.expect(SyntaxKind::LBrace) {
            self.errors.push(e);
            return None;
        }

        let then_start = self.tokens.peek()?.start;
        let mut then_items = Vec::new();
        while !self.tokens.at(SyntaxKind::RBrace) && self.tokens.peek().is_some() {
            if self.tokens.at(SyntaxKind::KwLet) {
                if let Some(stmt) = self.parse_let_stmt() {
                    then_items.push(BlockItem::Stmt(stmt));
                    continue;
                }
            }
            if let Some(expr) = self.parse_expr(0) {
                then_items.push(BlockItem::Expr(expr));
            } else {
                self.tokens.bump();
            }
        }
        let then_end_tok = match self.tokens.expect(SyntaxKind::RBrace) {
            Ok(t) => t,
            Err(e) => { self.errors.push(e); return None; }
        };
        let then_branch = ExprNode {
            kind: ExprKind::Block(then_items),
            span: Span::new(then_start, then_end_tok.end),
        };

        let else_branch = if self.tokens.eat(SyntaxKind::KwElse).is_some() {
            if self.tokens.at(SyntaxKind::KwIf) {
                // `else if` — parse as a nested if
                self.parse_if_expr()
            } else if self.tokens.at(SyntaxKind::LBrace) {
                let else_start = self.tokens.peek()?.start;
                let mut else_items = Vec::new();
                self.tokens.bump(); // consume '{'
                while !self.tokens.at(SyntaxKind::RBrace) && self.tokens.peek().is_some() {
                    if self.tokens.at(SyntaxKind::KwLet) {
                        if let Some(stmt) = self.parse_let_stmt() {
                            else_items.push(BlockItem::Stmt(stmt));
                            continue;
                        }
                    }
                    if let Some(expr) = self.parse_expr(0) {
                        else_items.push(BlockItem::Expr(expr));
                    } else {
                        self.tokens.bump();
                    }
                }
                let else_end_tok = match self.tokens.expect(SyntaxKind::RBrace) {
                    Ok(t) => t,
                    Err(e) => { self.errors.push(e); return None; }
                };
                Some(ExprNode {
                    kind: ExprKind::Block(else_items),
                    span: Span::new(else_start, else_end_tok.end),
                })
            } else {
                self.errors.push(ParseError::expected(
                    SyntaxKind::LBrace,
                    self.tokens.peek().map_or(SyntaxKind::Eof, |t| t.kind),
                    self.tokens.peek().map_or(0, |t| t.start),
                    self.tokens.peek().map_or(0, |t| t.end),
                ));
                None
            }
        } else {
            None
        };

        let end = else_branch.as_ref().map_or(then_end_tok.end, |e| e.span.end);
        Some(ExprNode {
            kind: ExprKind::If {
                condition: Box::new(condition),
                then_branch: Box::new(then_branch),
                else_branch: else_branch.map(Box::new),
            },
            span: Span::new(start.start, end),
        })
    }
```

```rust
        let start = self.tokens.bump()?; // consume 'use'
        debug_assert_eq!(start.kind, SyntaxKind::KwUse);

        // Collect path segments: ident (. ident)*
        let mut path_parts = Vec::new();
        loop {
            let tok = match self.tokens.expect(SyntaxKind::Ident) {
                Ok(t) => t,
                Err(e) => { self.errors.push(e); break; }
            };
            path_parts.push(tok.text);
            if !self.tokens.at(SyntaxKind::Dot) {
                break;
            }

            // Approximate end — we'd need to track token positions better
            start.end + path_parts.join(".").len()
        });
        Some(UseItem {

        })
```

```rust
    fn is_builtin_call(&self) -> bool {
        let ident = match self.tokens.peek() {
            Some(t) if t.kind == SyntaxKind::Ident => t.text,
            _ => return false,
        };
        matches!(ident, "println" | "assert")
    }

    /// Parse `println(args...)` or `assert(cond[, msg])`.
    fn parse_builtin_call(&mut self) -> Option<ExprNode> {
        let name_tok = self.tokens.bump().unwrap();
        let name = name_tok.text;
        let start = name_tok.start;

        if !self.tokens.at(SyntaxKind::LParen) {
            // Not a call — just an identifier
            let sym = self.interner.intern(name);
            return Some(ExprNode {
                kind: ExprKind::Ident(sym),
                span: Span::new(start, name_tok.end),
            });
        }

        self.tokens.bump(); // consume '('
        let mut args = Vec::new();
        while !self.tokens.at(SyntaxKind::RParen) && self.tokens.peek().is_some() {
            args.push(self.parse_expr(0)?);
            self.tokens.eat(SyntaxKind::Comma);
        }
        let end_tok = match self.tokens.expect(SyntaxKind::RParen) {
            Ok(t) => t,
            Err(e) => { self.errors.push(e); return None; }
        };

        // For now, represent as a regular Call with the builtin name
        let sym = self.interner.intern(name);
        Some(ExprNode {
            kind: ExprKind::Call {
                callee: Box::new(ExprNode {
                    kind: ExprKind::Ident(sym),
                    span: Span::new(start, name_tok.end),
                }),
                args,
            },
            span: Span::new(start, end_tok.end),
        })
    }
```

# Phase 4: Codegen — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement LLVM codegen for `let`/`let mut` (alloca), assignment (load/store), `if`/`else` (br/phi), string literals (global constants + fat pointers), `println`/`assert` runtime shims, and JIT execution.

**Architecture:** Each `HirFn` codegen now maintains a `vars: HashMap<Symbol, PointerValue>` mapping variable names to their stack slots (alloca pointers). `let` allocates and stores. Assignment loads-modifies-stores. `if`/`else` creates basic blocks with phi nodes. Strings become global constants with `{i8*, i64}` fat pointers. Runtime shims are emitted as LLVM IR before user code. JIT uses Inkwell's `ExecutionEngine`.

**Tech Stack:** `inkwell` (LLVM), `glyim-codegen-llvm`, `glyim-cli`

```rust
        let name = self.interner.resolve(hir_fn.name);
        let is_main = name == "main";
        let ret_type = if is_main { self.i32_type } else { self.i64_type };
        let fn_type = self.i64_type.fn_type(&[], false);
        let fn_value = self.module.add_function(name, fn_type, None);

        let entry = self.context.append_basic_block(fn_value, "entry");
        self.builder.position_at_end(entry);

        // Variables: Symbol → alloca pointer (NOT direct value)
        let mut vars: HashMap<Symbol, inkwell::values::PointerValue<'ctx>> = HashMap::new();

        // Allocate params on stack
        for (i, param_sym) in hir_fn.params.iter().enumerate() {
            let param_val = fn_value.get_nth_param(i as u32)
                .ok_or_else(|| format!("missing param {}", i))?;
            let alloca = self.builder.build_alloca(self.i64_type, &self.sym_name(*param_sym))
                .map_err(|e| format!("alloca: {}", e))?;
            self.builder.build_store(alloca, param_val)
                .map_err(|e| format!("store: {}", e))?;
            vars.insert(*param_sym, alloca);
        }

        // Codegen body
        let body_val = self.codegen_block(&hir_fn.body, &mut vars, fn_value)
            .ok_or_else(|| format!("codegen failed for '{}'", name))?;

        let ret_val = if is_main {
            self.builder.build_int_truncate(body_val, self.i32_type, "trunc")
                .map_err(|e| format!("truncate: {}", e))?
        } else {
            body_val
        };
        self.builder.build_return(Some(&ret_val))
            .map_err(|e| format!("return: {}", e))?;

        if !fn_value.verify(true) {
            fn_value.print_to_stderr();
            return Err(format!("verification failed for '{}'", name));
        }
        Ok(())
    }

                for stmt in stmts {

                }
                last
            }
            other => self.codegen_expr(other, vars),

        fn_value: FunctionValue<'ctx>,

            HirStmt::Let { name, mutable: _, value } => {

                let alloca = self.builder.build_alloca(self.i64_type, &self.sym_name(*name)).ok()?;
                self.builder.build_store(alloca, val).ok()?;
                vars.insert(*name, alloca);
                Some(val)
            }
            HirStmt::Assign { target, value } => {
                let ptr = vars.get(target)?;
                let new_val = self.codegen_expr(value, vars)?;
                self.builder.build_store(*ptr, new_val).ok()?;

            }

        }
    }

                    HirUnOp::Not => self.builder.build_not(val, "not").ok(),
```

                ```rust
                    fn process_string_escapes(&self, s: &str) -> Vec<u8> {
                        let mut bytes = Vec::new();
                        let mut chars = s.chars().peekable();
                        // Skip opening and closing quotes
                        if chars.peek() == Some(&'"') { chars.next(); }
                        let closing = chars.next_back().map_or(false, |c| c == '"');
                        let iter: &mut dyn Iterator<Item = char> = if closing { &mut chars } else { &mut std::iter::once('"').chain(chars) };
                
                        while let Some(c) = iter.next() {
                            if c == '\\' {
                                match iter.next() {
                                    Some('n') => bytes.push(0x0A),
                                    Some('t') => bytes.push(0x09),
                                    Some('r') => bytes.push(0x0D),
                                    Some('0') => bytes.push(0x00),
                                    Some('"') => bytes.push(0x22),
                                    Some('\\') => bytes.push(0x5C),
                                    Some(other) => {
                                        // Unknown escape — keep backslash + char
                                        bytes.push(b'\\');
                                        bytes.extend_from_slice(other.to_string().as_bytes());
                                    }
                                    None => bytes.push(b'\\'),
                                }
                            } else {
                                bytes.extend_from_slice(c.to_string().as_bytes());
                            }
                        }
                        bytes
                    }
                ```

                ```rust
                //! UI tests: compile .xyz source, compare stderr against .stderr snapshot.
                
                use std::path::{Path, PathBuf};
                use std::fs;
                
                fn ui_dir() -> PathBuf {
                    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("tests").join("ui")
                }
                
                /// Compile a .xyz file and return the rendered stderr.
                fn compile_stderr(source: &str, file_path: &str) -> String {
                    let parse_out = glyim_parse::parse(source);
                    if parse_out.errors.is_empty() {
                        // Also try codegen to catch codegen errors
                        match glyim_codegen_llvm::compile_to_ir(source) {
                            Ok(_) => String::new(),
                            Err(e) => format!("error: {e}"),
                        }
                    } else {
                        let diags: Vec<_> = parse_out.errors.iter().map(|e| {
                            let span = match e {
                                glyim_parse::ParseError::Expected { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                                glyim_parse::ParseError::UnexpectedEof { .. } => None,
                                glyim_parse::ParseError::ExpectedExpr { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                                glyim_parse::ParseError::Message { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                                glyim_parse::ParseError::CannotAssignImmutable { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                            };
                            glyim_diag::Diagnostic::error(e.to_string()).with_span_opt(span)
                        }).collect();
                        glyim_diag::render_diagnostics(source, file_path, &diags)
                    }
                }
                
                /// Run a single UI test by name (reads tests/ui/{name}.xyz).
                fn run_ui_test(name: &str) {
                    let xyz_path = ui_dir().join(format!("{name}.xyz"));
                    let stderr_path = ui_dir().join(format!("{name}.xyz.stderr"));
                
                    let source = fs::read_to_string(&xyz_path)
                        .unwrap_or_else(|e| panic!("failed to read {}: {e}", xyz_path.display()));
                    let file_path = format!("tests/ui/{name}.xyz");
                
                    let actual = compile_stderr(&source, &file_path);
                
                    if stderr_path.exists() {
                        let expected = fs::read_to_string(&stderr_path)
                            .unwrap_or_else(|e| panic!("failed to read {}: {e}", stderr_path.display()));
                        assert_eq!(actual, expected, "stderr mismatch for {}", name);
                    } else {
                        // No snapshot yet — create one via insta
                        insta::assert_snapshot!(name, actual);
                    }
                }
                
                // ── Individual test functions ────────────────────────────────
                
                #[test]
                fn ui_let_missing_eq() { run_ui_test("let_missing_eq"); }
                
                #[test]
                fn ui_assign_immutable() { run_ui_test("assign_immutable"); }
                
                #[test]
                fn ui_missing_main() { run_ui_test("missing_main"); }
                
                #[test]
                fn ui_unterminated_string() { run_ui_test("unterminated_string"); }
                
                #[test]

                #[test]
                ```
