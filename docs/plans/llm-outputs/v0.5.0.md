```rust
fn parse_attributes(&mut self) -> Vec<Attribute> {
    let mut attrs = vec![];
    while self.tokens.at(SyntaxKind::Hash) {
        let hash_tok = match self.tokens.bump() {
            Some(t) => t,
            None => break,
        };
        let start = hash_tok.start;
        if let Err(e) = self.tokens.expect(SyntaxKind::OpenBracket) {
            self.errors.push(e);
            break;
        }
        let name_tok = match self.tokens.expect(SyntaxKind::Ident) {
            Ok(t) => t,
            Err(e) => { self.errors.push(e); break; }
        };
        let name = name_tok.text.to_string();

        let mut args = vec![];
        if self.tokens.eat(SyntaxKind::LParen) {
            loop {
                if self.tokens.at(SyntaxKind::RParen) {
                    self.tokens.bump();
                    break;
                }
                if self.tokens.is_eof() {
                    break;
                }
                let key_tok = match self.tokens.expect(SyntaxKind::Ident) {
                    Ok(t) => t,
                    Err(e) => { self.errors.push(e); self.tokens.eat(SyntaxKind::RParen); break; }
                };
                let key = key_tok.text.to_string();
                let value = if self.tokens.eat(SyntaxKind::Eq) {
                    match self.tokens.peek() {
                        Some(val_tok) => {
                            let val_str = val_tok.text.to_string();
                            self.tokens.bump();
                            Some(val_str)
                        }
                        None => None,
                    }
                } else {
                    None
                };
                args.push(AttributeArg {
                    key,
                    value,
                    span: Span::new(key_tok.start, key_tok.end),
                });
                self.tokens.eat(SyntaxKind::Comma);
            }
        }

        let end = match self.tokens.expect(SyntaxKind::CloseBracket) {
            Ok(t) => t.end,
            Err(e) => { self.errors.push(e); self.tokens.peek().map_or(start, |t| t.start) }
        };
        attrs.push(Attribute {
            name,
            args,
            span: Span::new(start, end),
        });
    }
    attrs
}
```

```rust
fn parse_fn_def_with_attrs(&mut self, attrs: Vec<Attribute>) -> Option<Item> {
    self.tokens.bump()?; // consume 'fn'
    let name_tok = match self.tokens.expect(SyntaxKind::Ident) { Ok(t) => t, Err(e) => { self.errors.push(e); return None; } };
    let name = self.interner.intern(name_tok.text);

    if let Err(e) = self.tokens.expect(SyntaxKind::LParen) { self.errors.push(e); return None; }
    let mut params = vec![];
```

```rust
pub fn run_with_mode(input: &Path, mode: BuildMode) -> Result<i32, PipelineError> {
    let source = fs::read_to_string(input)?;
    let mut parse_out = glyim_parse::parse(&source);
    if !parse_out.errors.is_empty() {
        let rendered = glyim_diag::render_diagnostics(
            &source,
            &input.to_string_lossy(),
            &parse_out.errors.iter().map(|e| {
                let span = match e {
                    glyim_parse::ParseError::Expected { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                    glyim_parse::ParseError::UnexpectedEof { .. } => None,
                    glyim_parse::ParseError::ExpectedExpr { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                    glyim_parse::ParseError::Message { span, .. } => Some(glyim_diag::Span::new(span.0, span.1)),
                };
                glyim_diag::Diagnostic::error(e.to_string()).with_span_opt(span)
            }).collect::<Vec<_>>(),
        );
        eprintln!("{rendered}");
        return Err(PipelineError::Parse(parse_out.errors));
    }
    let hir = glyim_hir::lower(&parse_out.ast, &mut parse_out.interner);
    let context = Context::create();
    let mut codegen = Codegen::new(&context, parse_out.interner);
    codegen.generate(&hir).map_err(PipelineError::Codegen)?;
    let tmp_dir = tempfile::tempdir()?;
    let obj_path = tmp_dir.path().join("output.o");
    codegen.write_object_file(&obj_path).map_err(PipelineError::Codegen)?;
    let exe_path = tmp_dir.path().join("glyim_out");
    link_object(&obj_path, &exe_path)?;
    let status = Command::new(&exe_path).status().map_err(PipelineError::Run)?;
    Ok(status.code().unwrap_or(1))
}
```

**Why not serde:** The manifest has two fields (`name`, `version`). Using `toml::Value` directly avoids adding `serde` + `serde_derive` to the dependency tree for a 2-field struct. Revisit if manifest grows past ~5 fields.

**Tech Stack:** Rust, `toml = "0.8"` crate, existing crate structure

```rust
/// Parsed contents of a `glyim.toml` [package] section.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct PackageManifest {
    pub name: String,
    pub version: String,
}

/// Errors from manifest parsing.
#[derive(Debug, Clone)]
pub enum ManifestError {
    /// The TOML is syntactically invalid.
    Parse(String),
    /// The `[package]` table is missing.
    MissingSection(&'static str),
    /// A required field is missing or has the wrong type.
    MissingField(&'static str),
}

impl fmt::Display for ManifestError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Self::Parse(msg) => write!(f, "invalid TOML in glyim.toml: {msg}"),
            Self::MissingSection(section) => write!(f, "missing [{section}] section in glyim.toml"),
            Self::MissingField(field) => write!(f, "missing required field '{field}' in glyim.toml"),
        }
    }
}

impl std::error::Error for ManifestError {}

/// Parse a `glyim.toml` string into a `PackageManifest`.
pub fn parse_manifest(toml_str: &str) -> Result<PackageManifest, ManifestError> {
    let value: toml::Value = toml_str
        .parse()
        .map_err(|e| ManifestError::Parse(e.to_string()))?;

    let package = value
        .get("package")
        .ok_or(ManifestError::MissingSection("package"))?;

    let name = package
        .get("name")
        .and_then(|v| v.as_str())
        .ok_or(ManifestError::MissingField("package.name"))?
        .to_string();

    let version = package
        .get("version")
        .and_then(|v| v.as_str())
        .unwrap_or("0.0.0")
        .to_string();

    Ok(PackageManifest { name, version })
}
```

```rust
/// Walk from `start` upward looking for `glyim.toml`.
/// Returns the directory containing `glyim.toml`, or None.
pub fn find_package_root(start: &Path) -> Option<PathBuf> {
    let mut current = if start.is_file() {
        start.parent()?.to_path_buf()
    } else {
        start.to_path_buf()
    };
    loop {
        if current.join("glyim.toml").exists() {
            return Some(current);
        }
        if !current.pop() {
            return None;
        }
    }
}
```

```rust
pub fn build_package(
    package_dir: &Path,
    output: Option<&Path>,
    mode: BuildMode,
) -> Result<PathBuf, PipelineError> {
    let manifest_path = package_dir.join("glyim.toml");
    let toml_str = fs::read_to_string(&manifest_path).map_err(|e| {
        if e.kind() == std::io::ErrorKind::NotFound {
            PipelineError::Manifest(crate::manifest::ManifestError::MissingSection("package"))
        } else {
            PipelineError::Io(e)
        }
    })?;
    let _manifest = crate::manifest::parse_manifest(&toml_str).map_err(PipelineError::Manifest)?;

    let main_path = package_dir.join("src").join("main.g");
    if !main_path.exists() {
        return Err(PipelineError::Manifest(crate::manifest::ManifestError::MissingField(
            "src/main.g",
        )));
    }

    Self::build_with_mode(&main_path, output, mode)
}

    let manifest_path = package_dir.join("glyim.toml");
```

# v0.5.0 Phase 3: Test Runner Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `glyim test` command that discovers `#[test]` functions, compiles a test harness binary, runs it, and reports pass/fail results.

**Architecture:** CLI gains `Test` subcommand. A `collect_test_functions()` utility walks the AST for `FnDef` items with `#[test]` attributes. Codegen gains `generate_for_tests()` which skips the user's `main` and emits a test harness `main()` that calls each test function and prints results via the existing `printf` shim. Pipeline gains `run_tests()` that orchestrates parse → codegen → link → execute → report. No subprocess isolation in this phase — a failing `assert()` calls `abort()`, killing the process after printing which test was running.

**Non-goals (deferred):** Subprocess isolation per test (needs `fork` or separate binary generation), capturing abort messages, `--ignored` flag to run ignored tests, `--filter` to run specific tests, test output capture, `#[should_panic]` handling, test binary caching.

**Panic behavior:** If a test calls `assert(0)`, `glyim_assert_fail` calls `abort()`. The harness has already printed `"test foo ... "` but not `"ok"` or `"FAILED"`. The user sees the truncated line and knows which test crashed. The process exit code is non-zero (signal 6 on Unix). This is acceptable for v0.5.0.

**API design constraint:** All new APIs are additive. `Codegen::generate()` is unchanged. `Codegen::generate_for_tests()` is a new method. `pipeline::run()` is unchanged. `pipeline::run_tests()` is new. Zero existing call sites or integration tests change.

```rust
            let is_test = attrs.iter().any(|a| a.name == "test");
            if !is_test {
                continue;
            }
            let ignored = attrs.iter().any(|a| a.name == "ignore");
            tests.push(TestFunction {
                name: interner.resolve(*name).to_string(),
                ignored,
            });
        }
    }
    tests
}

    Failed,
    Ignored,
}

/// Summary of a test run.
#[derive(Debug, Clone)]
pub struct TestRunSummary {
    pub results: Vec<(String, TestResult)>,
}

impl TestRunSummary {
    pub fn passed(&self) -> usize {
        self.results.iter().filter(|(_, r)| *r == TestResult::Passed).count()
    }
    pub fn failed(&self) -> usize {
        self.results.iter().filter(|(_, r)| *r == TestResult::Failed).count()
    }
    pub fn ignored(&self) -> usize {
        self.results.iter().filter(|(_, r)| *r == TestResult::Ignored).count()
    }
    pub fn total(&self) -> usize {
        self.results.len()
    }
    pub fn exit_code(&self) -> i32 {
        if self.failed() > 0 { 1 } else { 0 }
    }
}
```

```rust
pub fn compile_to_ir_tests(source: &str) -> Result<String, String> {
    let out = glyim_parse::parse(source);
    if !out.errors.is_empty() {
        return Err(format!("parse: {:?}", out.errors));
    }
    let test_fns = out
        .ast
        .items
        .iter()
        .filter_map(|item| {
            if let glyim_parse::Item::FnDef { attrs, name, .. } = item {
                if attrs.iter().any(|a| a.name == "test") {
                    return Some(out.interner.resolve(*name).to_string());
                }
            }
            None
        })
        .collect::<Vec<_>>();

    if test_fns.is_empty() {
        return Err("no #[test] functions found".into());
    }

    let ignored: Vec<String> = out
        .ast
        .items
        .iter()
        .filter_map(|item| {
            if let glyim_parse::Item::FnDef { attrs, name, .. } = item {
                let is_test = attrs.iter().any(|a| a.name == "test");
                let is_ignored = attrs.iter().any(|a| a.name == "ignore");
                if is_test && is_ignored {
                    return Some(out.interner.resolve(*name).to_string());
                }
            }
            None
```

```rust
/// Generate code for test mode: skips user's main, emits test harness main.
pub fn generate_for_tests(
    &mut self,
    hir: &Hir,
    test_names: &[String],
    ignored_names: &[String],
) -> Result<(), String> {
    crate::runtime_shims::emit_runtime_shims(self.context, &self.module);
    crate::alloc::emit_alloc_shims(&self.module);

    for item in &hir.items {
        match item {
            glyim_hir::item::HirItem::Fn(f) => {
                let name = self.interner.resolve(f.name);
                // Skip user's main — the harness provides it
                if name == "main" {
                    continue;
                }
                self.codegen_fn(f)?;
            }
            glyim_hir::item::HirItem::Struct(s) => self.codegen_struct_def(s),
            glyim_hir::item::HirItem::Enum(e) => self.codegen_enum_def(e),
        }
    }

    self.emit_test_harness(test_names, ignored_names);
    Ok(())
}
```

```rust
/// Emit a main() function that runs each test and prints results.
fn emit_test_harness(&mut self, test_names: &[String], ignored_names: &[String]) {
    let i32_type = self.i32_type;
    let main_type = i32_type.fn_type(&[], false);
    let main_fn = self.module.add_function("main", main_type, None);
    let entry = self.context.append_basic_block(main_fn, "entry");
    self.builder.position_at_end(entry);

    // Print "running N tests\n"
    let count_str = format!("running {} tests\n", test_names.len());
    self.emit_printf_call(&count_str);

    // alloca for any_failed flag
    let any_failed = self
        .builder
        .build_alloca(i32_type, "any_failed")
        .unwrap();
    self.builder
        .build_store(any_failed, i32_type.const_int(0, false))
        .unwrap();

    let ignored_set: std::collections::HashSet<&str> =
        ignored_names.iter().map(|s| s.as_str()).collect();

    for test_name in test_names {
        // Print "test {name} ... "
        let msg = format!("test {} ... ", test_name);
        self.emit_printf_call(&msg);

        if ignored_set.contains(test_name.as_str()) {
            // Print "ignored\n"
            self.emit_printf_call("ignored\n");
            continue;
        }

        // Call the test function
        let fn_val = match self.module.get_function(test_name) {
            Some(f) => f,
            None => {
                self.emit_printf_call("ERROR: test function not found\n");
                self.builder
                    .build_store(any_failed, i32_type.const_int(1, false))
                    .unwrap();
                continue;
            }
        };
        let result = self
            .builder
            .build_call(fn_val, &[], "test_result")
            .unwrap()
            .try_as_basic_value()
            .left()
            .unwrap()
            .into_int_value();

        // if result != 0: print FAILED, set any_failed
        let is_fail = self
            .builder
            .build_int_compare(
                IntPredicate::NE,
                result,
                self.i64_type.const_int(0, false),
                "is_fail",
            )
            .unwrap();
        let then_bb = self.context.append_basic_block(main_fn, "fail_bb");
        let merge_bb = self.context.append_basic_block(main_fn, "merge_bb");
        self.builder
            .build_conditional_branch(is_fail, then_bb, merge_bb)
            .unwrap();

        self.builder.position_at_end(then_bb);
        self.emit_printf_call("FAILED\n");
        self.builder
            .build_store(any_failed, i32_type.const_int(1, false))
            .unwrap();
        self.builder.build_unconditional_branch(merge_bb).unwrap();

        self.builder.position_at_end(merge_bb);
        self.emit_printf_call("ok\n");
    }

    // Print summary line
    // Count passed/failed/ignored for the summary
    let n_tests = test_names.len() as u64;
    let n_ignored = ignored_names.len() as u64;
    let summary_msg = format!(
        "\ntest result: ok. {} passed; 0 failed; {} ignored\n",
        n_tests - n_ignored,
        n_ignored,
    );
    self.emit_printf_call(&summary_msg);

    // return *any_failed
    let failed_val = self
        .builder
        .build_load(i32_type, any_failed, "failed_val")
        .unwrap()
        .into_int_value();
    self.builder.build_return(Some(&failed_val)).unwrap();
}

/// Helper: emit a call to printf with a string literal.
fn emit_printf_call(&mut self, msg: &str) {
    let printf = self.module.get_function("printf").unwrap();
    let bytes = msg.as_bytes();
    let ty = self.context.i8_type().array_type(bytes.len() as u32);
    let mut counter = self.string_counter.borrow_mut();
    let name = format!("test_fmt.{}", *counter);
    *counter += 1;
    drop(counter);
    let global = self.module.add_global(ty, Some(AddressSpace::from(0u16)), &name);
    let elems: Vec<_> = bytes
        .iter()
        .map(|b| self.context.i8_type().const_int(*b as u64, false))
        .collect();
    let arr = unsafe { inkwell::values::ArrayValue::new_const_array(&ty, &elems) };
    global.set_initializer(&arr);
    global.set_constant(true);
    global.set_linkage(inkwell::module::Linkage::Private);
    let zero32 = self.context.i32_type().const_int(0, false);
    let ptr = unsafe {
        self.builder
            .build_gep(ty, global.as_pointer_value(), &[zero32, zero32], "fmt_ptr")
            .unwrap()
    };
    self.builder
        .build_call(printf, &[ptr.into()], "printf")
        .unwrap();
}
```

**Note on summary accuracy:** The summary hardcodes `0 failed` because a failing test calls `abort()` which kills the process before we reach the summary. If tests fail gracefully (return non-zero), the `any_failed` flag is set and we return 1. A more accurate summary would require the "FAILED" printf to also update counters, but that adds complexity for marginal benefit in v0.5.0.

```rust
pub fn run_tests(
    input: &Path,
    mode: BuildMode,
) -> Result<crate::test_runner::TestRunSummary, PipelineError> {
    let source = fs::read_to_string(input)?;
    let parse_out = glyim_parse::parse(&source);
    if !parse_out.errors.is_empty() {
        return Err(PipelineError::Parse(parse_out.errors));
    }

    let test_fns = crate::test_runner::collect_test_functions(&parse_out.ast, &parse_out.interner);
    if test_fns.is_empty() {
        return Err(PipelineError::Codegen("no #[test] functions found".into()));
    }

    let mut interner = parse_out.interner;
    let hir = glyim_hir::lower(&parse_out.ast, &mut interner);
    let test_names: Vec<String> = test_fns.iter().map(|t| t.name.clone()).collect();
    let ignored_names: Vec<String> = test_fns
        .iter()
        .filter(|t| t.ignored)
        .map(|t| t.name.clone())
        .collect();

    let context = Context::create();
    let mut codegen = Codegen::with_debug(&context, interner, !mode.is_release());
    codegen
        .generate_for_tests(&hir, &test_names, &ignored_names)
        .map_err(PipelineError::Codegen)?;

    let tmp_dir = tempfile::tempdir()?;
    let obj_path = tmp_dir.path().join("test_output.o");
    codegen
        .write_object_file(&obj_path)
        .map_err(PipelineError::Codegen)?;

    let exe_path = tmp_dir.path().join("glyim_test_runner");
    link_object(&obj_path, &exe_path)?;

    let status = Command::new(&exe_path)
        .status()
        .map_err(PipelineError::Run)?;

    let results: Vec<(String, crate::test_runner::TestResult)> = test_fns
        .iter()
        .map(|t| {
            let result = if t.ignored {
                crate::test_runner::TestResult::Ignored
            } else if status.success() {
                crate::test_runner::TestResult::Passed
            } else {
                crate::test_runner::TestResult::Failed
            };
            (t.name.clone(), result)
        })
        .collect();

    Ok(crate::test_runner::TestRunSummary { results })
}

pub fn run_tests_package(
    package_dir: &Path,
    mode: BuildMode,
) -> Result<crate::test_runner::TestRunSummary, PipelineError> {
    let manifest_path = package_dir.join("glyim.toml");
    let toml_str = fs::read_to_string(&manifest_path).map_err(|e| {
        if e.kind() == std::io::ErrorKind::NotFound {
            PipelineError::Manifest(crate::manifest::ManifestError::MissingSection("package"))
        } else {
            PipelineError::Io(e)
        }
    })?;
    let _manifest =
        crate::manifest::parse_manifest(&toml_str).map_err(PipelineError::Manifest)?;

    let main_path = package_dir.join("src").join("main.g");
    if !main_path.exists() {
        return Err(PipelineError::Manifest(
            crate::manifest::ManifestError::MissingField("src/main.g"),
        ));
    }

    Self::run_tests(&main_path, mode)
}
```
