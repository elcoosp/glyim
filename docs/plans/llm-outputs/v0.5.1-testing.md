```rust
use glyim_hir::types::HirType;
use glyim_interner::Interner;

/// Parse → lower → typecheck. Returns the TypeChecker (with interner still accessible).
/// Panics if the source has parse errors (these are unit tests for the type checker, not the parser).
fn typecheck(source: &str) -> TypeChecker {
    let parse_out = glyim_parse::parse(source);
    if !parse_out.errors.is_empty() {
        panic!("parse errors in test source: {:?}", parse_out.errors);
    }
    let mut interner = parse_out.interner;
    let hir = glyim_hir::lower(&parse_out.ast, &mut interner);
    let mut tc = TypeChecker::new(interner);
    let _ = tc.check(&hir);
    tc
}
```

```rust
    let mut tc = TypeChecker::new(Interner::new());

    tc.insert_binding(x, HirType::Int);
    tc.push_scope();
    tc.insert_binding(x, HirType::Bool);
```

```rust
fn expr_ids_are_monotonic() {
    let (hir, _) = lower_source("main = () => { let x = 1; let y = 2; x + y }");
    let mut last_id: i32 = -1;
    fn collect_ids(expr: &HirExpr, ids: &mut Vec<u32>) {
        ids.push(expr.get_id().as_usize() as u32);
        match expr {
            HirExpr::Block { stmts, .. } => {
                for stmt in stmts {
                    match stmt {
                        HirStmt::Expr(e) => collect_ids(e, ids),
                        HirStmt::Let { value, .. } => collect_ids(value, ids),
                        HirStmt::Assign { value, .. } => collect_ids(value, ids),
                        _ => {}
                    }
                }
            }
            HirExpr::Binary { lhs, rhs, .. } => {
                collect_ids(lhs, ids);
                collect_ids(rhs, ids);
            }
            HirExpr::Unary { operand, .. } => collect_ids(operand, ids),
            HirExpr::If { condition, then_branch, else_branch, .. } => {
                collect_ids(condition, ids);
                collect_ids(then_branch, ids);
                if let Some(e) = else_branch { collect_ids(e, ids); }
            }
            _ => {}
        }
    }
    let mut ids = Vec::new();
    collect_ids(get_main_body(&hir), &mut ids);
    for &id in &ids {
        assert!(id as i32 > last_id, "ExprId not monotonic: {} after {}", id, last_id);
        last_id = id as i32;
    }
}
```
