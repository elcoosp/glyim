**Goal:** Replace the Glyim type checker with a demand-driven, query-cached metacompiler that supports CHR trait resolution, bidirectional elaboration, reflection via profunctor optics, and Wasm-sandboxed comptime evaluation.

```rust
pub struct TyArena {
    kinds: Vec<TyKind>,
    /// Maps Ty::Infer to the exact Span that created it.
    infer_spans: Vec<Span>,
}

impl TyArena {
    pub fn new() -> Self {
        Self {
            kinds: Vec::new(),
            infer_spans: Vec::new(),
        }
    }

    pub fn alloc(&mut self, kind: TyKind) -> Ty {
        let id = self.kinds.len();
        self.kinds.push(kind);
        Ty(id)
    }

    pub fn fresh_infer(&mut self, span: Span) -> Ty {
        let id = self.kinds.len();
        self.infer_spans.push(span);
        self.kinds.push(TyKind::Infer);
        Ty(id)
    }

    pub fn get(&self, ty: Ty) -> &TyKind {
        &self.kinds[ty.0]
    }

    pub fn get_infer_span(&self, ty: Ty) -> Option<Span> {
        if matches!(self.get(ty), TyKind::Infer) {
            self.infer_spans.get(ty.0).copied()
        } else {
            None
        }
    }

    pub fn len(&self) -> usize {
        self.kinds.len()
    }

    pub fn is_empty(&self) -> bool {
        self.kinds.is_empty()
    }
}

impl Default for TyArena {
    fn default() -> Self {
        Self::new()
    }
}
```

use crate::ty::{Ty, TyKind, TyArena};

/// Lockstep structural traversal to find the exact point of divergence.
pub fn zip_diff(
    arena: &TyArena,
    t1: Ty,
    t2: Ty,
    path: String,
) -> Option<String> {
    match (arena.get(t1), arena.get(t2)) {
        (TyKind::App(s1, a1), TyKind::App(s2, a2)) if s1 == s2 && a1.len() == a2.len() => {
            a1.iter().zip(a2).enumerate().find_map(|(i, (a, b))| {
                zip_diff(arena, *a, *b, format!("{path}.T{i}"))
            })
        }
        _ if t1 != t2 => Some(format!("Diverged at {path}")),
        _ => None,
    }
}

/// A token proving an error was emitted.
#[derive(Clone, Copy, Debug)]
pub struct ErrorGuaranteed(#[doc(hidden)] std::convert::Infallible);

impl ErrorGuaranteed {
    pub fn new() -> Self {
        Self(std::convert::Infallible::new())
    }
}

pub struct UnificationTable {
    parents: Vec<Ty>, // Union-Find array
    ranks: Vec<u8>,
}

impl UnificationTable {
    pub fn new() -> Self {
        Self {
            parents: Vec::new(),
            ranks: Vec::new(),
        }
    }

    pub fn new_var(&mut self, arena: &mut TyArena, span: Span) -> Ty {
        let ty = arena.fresh_infer(span);
        self.parents.push(ty); // Points to itself
        self.ranks.push(0);
        ty
    }

    pub fn find(&self, arena: &TyArena, ty: Ty) -> Ty {
        match arena.get(ty) {
            TyKind::Infer | TyKind::Error if self.parents[ty.0] == ty => ty,
            TyKind::Infer => {
                let root = self.find(arena, self.parents[ty.0]);
                // Note: path compression requires &mut self; defer for now
                root
            }
            _ => ty, // Concrete types are their own root
        }
    }
}

impl Default for UnificationTable {
    fn default() -> Self {
        Self::new()
    }
}

pub struct ErrorGuaranteed(#[doc(hidden)] std::convert::Infallible);

impl ErrorGuaranteed {
    pub fn new() -> Self {
        Self(std::convert::Infallible::new())
    }
}

pub struct UnificationTable {
    parents: Vec<Ty>,
    ranks: Vec<u8>,
}

impl UnificationTable {
    pub fn new() -> Self {
        Self {
            parents: Vec::new(),
            ranks: Vec::new(),
        }
    }

    pub fn new_var(&mut self, arena: &mut TyArena, span: Span) -> Ty {
        let ty = arena.fresh_infer(span);
        self.parents.push(ty);
        self.ranks.push(0);
        ty
    }

    pub fn find(&self, arena: &TyArena, ty: Ty) -> Ty {
        match arena.get(ty) {
            TyKind::Infer | TyKind::Error if self.parents[ty.0] == ty => ty,
            TyKind::Infer => self.find(arena, self.parents[ty.0]),
            _ => ty,
        }
    }

    /// Core unification. Returns Err if it had to emit a poison Error type.
    pub fn unify(
        &mut self,
        arena: &mut TyArena,
        a: Ty,
        b: Ty,
        span: Span,
        emit_err: &mut dyn FnMut(TypeError),
    ) -> Result<(), ErrorGuaranteed> {
        let a = self.find(arena, a);
        let b = self.find(arena, b);
        if a == b {
            return Ok(());
        }

        // Error is infectious — unifies with anything without new errors
        if matches!(arena.get(a), TyKind::Error) || matches!(arena.get(b), TyKind::Error) {
            return Ok(());
        }

        // Occurs check: ?0 = Vec<?0> is illegal
        if self.occurs(arena, a, b) || self.occurs(arena, b, a) {
            let origin = arena.get_infer_span(a)
                .or_else(|| arena.get_infer_span(b))
                .unwrap_or(span);
            emit_err(TypeError::InfiniteType { span: origin });
            arena.kinds[a.0] = TyKind::Error;
            return Err(ErrorGuaranteed::new());
        }

        // Infer vars unify with anything
        if matches!(arena.get(a), TyKind::Infer) {
            self.union(a, b);
            return Ok(());
        }
        if matches!(arena.get(b), TyKind::Infer) {
            self.union(b, a);
            return Ok(());
        }

        // Structural recursion for App, RawPtr, Fn...
        self.unify_structural(arena, a, b, span, emit_err)
    }

    fn union(&mut self, a: Ty, b: Ty) {
        // Simple union without rank optimization for now
        self.parents[a.0] = b;
    }

    fn occurs(&self, arena: &TyArena, var: Ty, ty: Ty) -> bool {
        if var == ty {
            return true;
        }
        match arena.get(ty) {
            TyKind::App(_, args) => args.iter().any(|arg| self.occurs(arena, var, *arg)),
            TyKind::Fn(params, ret) => {
                params.iter().any(|p| self.occurs(arena, var, *p)) || self.occurs(arena, var, *ret)
            }
            TyKind::RawPtr(inner) => self.occurs(arena, var, *inner),
            _ => false,
        }
    }

    fn unify_structural(
        &mut self,
        arena: &mut TyArena,
        a: Ty,
        b: Ty,
        span: Span,
        emit_err: &mut dyn FnMut(TypeError),
    ) -> Result<(), ErrorGuaranteed> {
        match (arena.get(a), arena.get(b)) {
            (TyKind::Int, TyKind::Int) => Ok(()),
            (TyKind::Bool, TyKind::Bool) => Ok(()),
            (TyKind::Float, TyKind::Float) => Ok(()),
            (TyKind::Str, TyKind::Str) => Ok(()),
            (TyKind::Unit, TyKind::Unit) => Ok(()),
            (TyKind::Never, TyKind::Never) => Ok(()),

            (TyKind::Named(s1), TyKind::Named(s2)) if s1 == s2 => Ok(()),

            (TyKind::App(s1, a1), TyKind::App(s2, a2)) if s1 == s2 && a1.len() == a2.len() => {
                for (arg_a, arg_b) in a1.iter().zip(a2) {
                    self.unify(arena, *arg_a, *arg_b, span, emit_err)?;
                }
                Ok(())
            }

            (TyKind::Fn(p1, r1), TyKind::Fn(p2, r2)) if p1.len() == p2.len() => {
                for (pa, pb) in p1.iter().zip(p2) {
                    self.unify(arena, *pa, *pb, span, emit_err)?;
                }
                self.unify(arena, *r1, *r2, span, emit_err)
            }

            (TyKind::RawPtr(i1), TyKind::RawPtr(i2)) => {
                self.unify(arena, *i1, *i2, span, emit_err)
            }

            _ => {
                emit_err(TypeError::MismatchedTypes {
                    expected_span: arena.get_infer_span(a).unwrap_or(span),
                    found_span: arena.get_infer_span(b).unwrap_or(span),
                    expected: format!("{:?}", arena.get(a)),
                    found: format!("{:?}", arena.get(b)),
                    diff_path: None,
                    autofix: None,
                });
                Err(ErrorGuaranteed::new())
            }
        }
    }
}

impl Default for UnificationTable {
    fn default() -> Self {
        Self::new()
    }
}

use crate::ty::Ty;
use glyim_interner::Symbol;
use std::collections::HashSet;

/// A logical goal we need to prove.
#[derive(Clone, Debug, PartialEq, Eq, Hash)]
pub enum Goal {
    /// TraitImpl(Symbol, [Ty]) e.g., `Display(Vec<i64>)`
    TraitImpl(Symbol, Vec<Ty>),
    /// StateTransition(Symbol, CurrentStateTy, TargetStateTy)
    StateTransition(Symbol, Ty, Ty),
    /// Can we prove that type T is reflectable?
    Reflectable(Ty),
    /// Does type T have a field with the given name?
    HasField(Ty, Symbol),
}

/// A rewrite rule for the solver.
#[derive(Clone, Debug)]
pub enum ChrRule {
    /// If we need `Goal`, AND we have `Premises`, THEN `Goal` is proven.
    Simplify {
        goal: Goal,
        premises: Vec<Goal>,
    },
    /// If we need `Goal`, AND we have `Premises`, THEN emit `NewGoals`.
    Propagate {
        goal: Goal,
        premises: Vec<Goal>,
        new_goals: Vec<Goal>,
    },
}

impl ChrRule {
    /// Check if this rule matches the given goal.
    pub fn matches(&self, goal: &Goal) -> bool {
        match self {
            ChrRule::Simplify { goal: rule_goal, .. } => rule_goal == goal,
            ChrRule::Propagate { goal: rule_goal, .. } => rule_goal == goal,
        }
    }
}

pub struct ChrStore {
    rules: Vec<ChrRule>,
    pending_goals: Vec<Goal>,
    proven_goals: HashSet<Goal>,
}

impl ChrStore {
    pub fn new(rules: Vec<ChrRule>) -> Self {
        Self {
            rules,
            pending_goals: Vec::new(),
            proven_goals: HashSet::new(),
        }
    }

    pub fn add_rules(&mut self, rules: Vec<ChrRule>) {
        self.rules.extend(rules);
    }

    pub fn pending_goals(&self) -> &[Goal] {
        &self.pending_goals
    }

    pub fn proven_goals(&self) -> &HashSet<Goal> {
        &self.proven_goals
    }

    /// Add a goal to be proven.
    pub fn push_goal(&mut self, goal: Goal) {
        self.pending_goals.push(goal);
    }
}

use crate::unify::ErrorGuaranteed;

impl ChrStore {
    // ... existing methods ...

    /// Run the solver to fixed point.
    pub fn solve(&mut self, arena: &crate::ty::TyArena) -> Result<(), ErrorGuaranteed> {
        while let Some(goal) = self.pending_goals.pop() {
            if self.proven_goals.contains(&goal) {
                continue;
            }

            let mut rule_matched = false;
            for rule in &self.rules {
                if rule.matches(&goal) {
                    rule_matched = true;
                    match rule {
                        ChrRule::Simplify { premises, .. } => {
                            // Check if all premises are already proven
                            if premises.iter().all(|p| self.proven_goals.contains(p)) {
                                self.proven_goals.insert(goal.clone());
                            } else {
                                // Re-queue: push premises first, then the goal again
                                for p in premises {
                                    if !self.proven_goals.contains(p) {
                                        self.pending_goals.push(p.clone());
                                    }
                                }
                                self.pending_goals.push(goal.clone());
                            }
                        }
                        ChrRule::Propagate { premises, new_goals, .. } => {
                            if premises.iter().all(|p| self.proven_goals.contains(p)) {
                                self.proven_goals.insert(goal.clone());
                                for ng in new_goals {
                                    if !self.proven_goals.contains(ng) {
                                        self.pending_goals.push(ng.clone());
                                    }
                                }
                            } else {
                                for p in premises {
                                    if !self.proven_goals.contains(p) {
                                        self.pending_goals.push(p.clone());
                                    }
                                }
                                self.pending_goals.push(goal.clone());
                            }
                        }
                    }
                    break;
                }
            }

            if !rule_matched {
                // No rule fired. Type error: Trait not implemented.
                return Err(ErrorGuaranteed::new());
            }
        }
        Ok(())
    }
}

use crate::ty::{Ty, TyKind, TyArena};
use crate::unify::UnificationTable;
use glyim_hir::types::HirType;
use glyim_interner::Interner;

/// Resolve a Ty to a concrete HirType.
pub fn resolve_ty(arena: &TyArena, unification: &UnificationTable, ty: Ty) -> HirType {
    let ty = unification.find(arena, ty);
    match arena.get(ty) {
        TyKind::Int => HirType::Int,
        TyKind::Float => HirType::Float,
        TyKind::Bool => HirType::Bool,
        TyKind::Str => HirType::Str,
        TyKind::Unit => HirType::Unit,
        TyKind::Never => HirType::Never,
        TyKind::Error => HirType::Error,
        TyKind::Infer => HirType::Error, // Unresolved hole
        TyKind::Named(sym) => HirType::Named(glyim_interner::resolve(*sym).to_string()),
        TyKind::App(sym, args) => HirType::Generic(
            glyim_interner::resolve(*sym).to_string(),
            args.iter().map(|a| resolve_ty(arena, unification, *a)).collect(),
        ),
        TyKind::Fn(params, ret) => HirType::Func(
            params.iter().map(|p| resolve_ty(arena, unification, *p)).collect(),
            Box::new(resolve_ty(arena, unification, *ret)),
        ),
        TyKind::RawPtr(inner) => HirType::RawPtr(Box::new(resolve_ty(arena, unification, *inner))),
        // V3 types that don't have HirType equivalents yet
        TyKind::Code(_) => HirType::Error,
        TyKind::Const(_, _) => HirType::Error,
        TyKind::EffectFn(params, ret, _) => HirType::Func(
            params.iter().map(|p| resolve_ty(arena, unification, *p)).collect(),
            Box::new(resolve_ty(arena, unification, *ret)),
        ),
        TyKind::Any => HirType::Error,
        TyKind::TypeInfo(_) => HirType::Error,
    }
}

/// Resolve all expression types from the elaboration map.
pub fn resolve_expr_types(
    arena: &TyArena,
    unification: &UnificationTable,
    elab_map: &std::collections::HashMap<glyim_hir::types::ExprId, Ty>,
) -> Vec<HirType> {
    let max_id = elab_map.keys().map(|id| id.as_usize()).max().unwrap_or(0);
    let mut expr_types = vec![HirType::Error; max_id + 1];
    for (id, ty) in elab_map {
        expr_types[id.as_usize()] = resolve_ty(arena, unification, *ty);
    }
    expr_types
}

                result_ty
            }
    
            HirExpr::StructLit { id, name, fields, span } => {
                for field in fields {
                    synth_expr(ctx, &field.value);
                }
                let ty = ctx.db.arena.alloc(TyKind::Named(*name));
                ctx.record_type(*id, ty);
                ty
            }
    
            HirExpr::FieldAccess { id, receiver, field_name, span } => {
                let _recv_ty = synth_expr(ctx, receiver);
                let field_ty = ctx.db.arena.fresh_infer(*span);
                ctx.record_type(*id, field_ty);
                field_ty
            }
    
            HirExpr::Match { id, scrutinee, arms, span } => {
                synth_expr(ctx, scrutinee);
                let mut arm_tys = Vec::new();
                for arm in arms {
                    arm_tys.push(synth_expr(ctx, &arm.body));
                }
                // Unify all arm types; for now return the first or Error
                let result_ty = arm_tys.first().copied().unwrap_or_else(|| ctx.db.arena.alloc(TyKind::Error));
                ctx.record_type(*id, result_ty);
                result_ty
            }
    
            HirExpr::Closure { id, params, body, span } => {
                let param_tys: Vec<Ty> = params.iter()
                    .map(|_p| ctx.db.arena.fresh_infer(*span))
                    .collect();
                let body_ty = synth_expr(ctx, body);
                let fn_ty = ctx.db.arena.alloc(TyKind::Fn(param_tys, body_ty));
                ctx.record_type(*id, fn_ty);
                fn_ty
            }
    
            HirExpr::Hole { id, span } => {
                // Hole without expected type: create a fresh inference variable
                let ty = ctx.db.arena.fresh_infer(*span);
                ctx.record_type(*id, ty);
                ty
            }
    
            HirExpr::Quote { id, body, span } => {
                // Elevated staging level
                let inner_ty = synth_expr(ctx, body);
                let code_ty = ctx.db.arena.alloc(TyKind::Code(inner_ty));
                ctx.record_type(*id, code_ty);
                code_ty
            }
    
            HirExpr::Splice { id, inner, span } => {
                let inner_ty = synth_expr(ctx, inner);
                // Extract T from Code<T>
                if let TyKind::Code(t) = ctx.db.arena.get(inner_ty) {
                    ctx.record_type(*id, *t);
                    *t
                } else {
                    // Phase violation or type error
                    let ty = ctx.db.arena.alloc(TyKind::Error);
                    ctx.record_type(*id, ty);
                    ty
                }
            }
    
            HirExpr::Block { id, stmts, trailing, span } => {
                for stmt in stmts {
                    synth_expr(ctx, stmt);
                }
                let result_ty = if let Some(trailing_expr) = trailing {
                    synth_expr(ctx, trailing_expr)
                } else {
                    ctx.db.arena.alloc(TyKind::Unit)
                };
                ctx.record_type(*id, result_ty);
                result_ty
            }
    
            HirExpr::Error { id, span } => {
                let ty = ctx.db.arena.alloc(TyKind::Error);
                ctx.record_type(*id, ty);
                ty
            }
        };
    
        ty
    }

    /// Check an expression against an expected type.
    /// This is the "check" mode of bidirectional type checking.
    pub fn check_expr(ctx: &mut ElabContext, expr: &HirExpr, expected: Ty) {
        // Error type is infectious — unifies with anything
        if matches!(ctx.db.arena.get(expected), TyKind::Error) {
            let synth = synth_expr(ctx, expr);
            ctx.unification.unify(
                &mut ctx.db.arena,
                synth,
                expected,
                span_of(expr),
                &mut |e| ctx.emit_error(e),
            ).ok(); // Swallow the guaranteed — we already know there's an error
            return;
        }
    
        match expr {
            HirExpr::Hole { id, span } => {
                // Hole with expected type: unify the inference variable with expected
                let hole_ty = ctx.db.arena.fresh_infer(*span);
                ctx.unification.unify(
                    &mut ctx.db.arena,
                    hole_ty,
                    expected,
                    *span,
                    &mut |e| ctx.emit_error(e),
                ).ok();
                ctx.record_type(*id, hole_ty);
            }
    
            HirExpr::Error { id, span } => {
                // Error expression: unify with expected, but don't emit new errors
                let error_ty = ctx.db.arena.alloc(TyKind::Error);
                ctx.unification.unify(
                    &mut ctx.db.arena,
                    error_ty,
                    expected,
                    *span,
                    &mut |_| {}, // No new errors for Error type
                ).ok();
                ctx.record_type(*id, error_ty);
            }
    
            HirExpr::Closure { id, params, body, span } => {
                // Check closure body against the return type from expected
                if let TyKind::Fn(param_tys, ret_ty) = ctx.db.arena.get(expected) {
                    // Check each parameter type
                    for (param, param_ty) in params.iter().zip(param_tys.iter()) {
                        let _ = (param, param_ty); // Will wire into scope later
                    }
                    // Check the body against the return type
                    check_expr(ctx, body, *ret_ty);
                    ctx.record_type(*id, expected);
                } else {
                    // Expected type is not a function type — synthesize instead
                    let synth = synth_expr(ctx, expr);
                    ctx.unification.unify(
                        &mut ctx.db.arena,
                        synth,
                        expected,
                        *span,
                        &mut |e| ctx.emit_error(e),
                    ).ok();
                }
            }
    
            HirExpr::Block { id, stmts, trailing, span } => {
                for stmt in stmts {
                    check_expr(ctx, stmt, ctx.db.arena.alloc(TyKind::Unit));
                }
                if let Some(trailing_expr) = trailing {
                    check_expr(ctx, trailing_expr, expected);
                } else {
                    // Block without trailing expr has type Unit
                    ctx.unification.unify(
                        &mut ctx.db.arena,
                        expected,
                        ctx.db.arena.alloc(TyKind::Unit),
                        *span,
                        &mut |e| ctx.emit_error(e),
                    ).ok();
                }
                ctx.record_type(*id, expected);
            }
    
            // Default: synthesize, then unify with expected
            _ => {
                let synth = synth_expr(ctx, expr);
                ctx.unification.unify(
                    &mut ctx.db.arena,
                    synth,
                    expected,
                    span_of(expr),
                    &mut |e| ctx.emit_error(e),
                ).ok();
            }
        }
    }

            // 1. Registration Phase
            let mut chr_store = ChrStore::new(vec![]);
            for item in &hir.items {
                self.register_item(item, &mut chr_store);
            }
    
            // 2. Elaboration Phase
            let mut unification = UnificationTable::new();
            let mut errors = Vec::new();
    
            {
                let mut ctx = ElabContext::new(self, &mut unification, &mut chr_store);
    
                for item in &hir.items {
                    ctx.elaborate_item(item);
                }
    
                errors.extend(ctx.errors);
            }

            if let Err(_) = chr_store.solve(&self.arena) {
                errors.push(TypeError::MismatchedTypes {
                    expected_span: Span::new(0, 0),
                    found_span: Span::new(0, 0),
                    expected: "implementable trait".to_string(),
                    found: "no impl".to_string(),
                    diff_path: None,
                    autofix: None,
                });

            // We need to read the expr_types from the last ElabContext.

            let mut elab_map: HashMap<glyim_hir::types::ExprId, Ty> = HashMap::new();
            {
                let mut ctx = ElabContext::new(self, &mut unification, &mut chr_store);
                for item in &hir.items {
                    ctx.elaborate_item(item);
                }
                elab_map = ctx.expr_types;
            }
    
            let expr_types = freeze::resolve_expr_types(&self.arena, &unification, &elab_map);
    
            if errors.is_empty() {
                Ok(TypeCheckOutput {
                    expr_types,
                    call_type_args: HashMap::new(),
                })
            } else {
                // Even on error, produce the output so the pipeline can continue
                // (error recovery: downstream phases see Error types)
                Ok(TypeCheckOutput {
                    expr_types,
                    call_type_args: HashMap::new(),
                })
            }

                HirType::Float => self.db.arena.alloc(TyKind::Float),

    use crate::rep::{Rep, RepMeta};
    use crate::ty::Ty;
    use glyim_interner::Symbol;
    
    /// A lens: a first-class getter/setter pair for a field.
    #[derive(Clone, Debug)]
    pub struct Lens {
        pub field_name: Symbol,
        pub source_ty: Ty,
        pub target_ty: Ty,
        /// Index path to reach this field in the Rep tree
        pub path: Vec<usize>,
    }
    
    /// A prism: a first-class constructor/matcher for a variant.
    #[derive(Clone, Debug)]
    pub struct Prism {
        pub ctor_name: Symbol,
        pub source_ty: Ty,
        pub target_ty: Ty,
    }
    
    /// A generic optic (either lens or prism).
    #[derive(Clone, Debug)]
    pub enum Optic {
        Lens(Lens),
        Prism(Prism),
    }
    
    /// Collect all fields from a Rep tree, returning (field_name, field_ty, path).
    fn collect_fields(rep: &Rep, path: &mut Vec<usize>) -> Vec<(Symbol, Ty, Vec<usize>)> {
        match rep {
            Rep::Meta(_, inner) => collect_fields(inner, path),
            Rep::Constructor(_, inner) => collect_fields(inner, path),
            Rep::Product(a, b) => {
                let mut result = Vec::new();
                path.push(0);
                result.extend(collect_fields(a, path));
                path.pop();
                path.push(1);
                result.extend(collect_fields(b, path));
                path.pop();
                result
            }
            Rep::Field(meta, ty) => {
                vec![(meta.name, *ty, path.clone())]
            }
            Rep::Sum(_, _) => {
                // Sum nodes don't have direct fields — handled via prisms
                vec![]
            }
            Rep::Unit => vec![],
        }
    }
    
    /// Generate a Lens for each field in a struct-like Rep.
    pub fn generate_lenses(rep: &Rep) -> Vec<Lens> {
        let fields = collect_fields(rep, &mut vec![]);
        fields.into_iter().map(|(name, ty, path)| Lens {
            field_name: name,
            source_ty: Ty(0), // Will be filled with the actual struct type during integration
            target_ty: ty,
            path,
        }).collect()
    }
    
    /// Collect all constructors from a Rep tree, returning (ctor_name, fields).
    fn collect_constructors(rep: &Rep) -> Vec<(Symbol, Vec<(Symbol, Ty)>)> {
        match rep {
            Rep::Meta(_, inner) => collect_constructors(inner),
            Rep::Sum(a, b) => {
                let mut result = collect_constructors(a);
                result.extend(collect_constructors(b));
                result
            }
            Rep::Constructor(meta, inner) => {
                let fields = collect_fields(inner, &mut vec![]);
                vec![(meta.name, fields.into_iter().map(|(n, t, _)| (n, t)).collect())]
            }
            _ => vec![],
        }
    }
    
    /// Generate a Prism for each single-field constructor in an enum-like Rep.
    /// Multi-field and nullary constructors are excluded (per FR-20: "Should").
    pub fn generate_prisms(rep: &Rep) -> Vec<Prism> {
        let ctors = collect_constructors(rep);
        ctors.into_iter()
            .filter_map(|(name, fields)| {
                // Only generate prisms for single-field constructors
                if fields.len() == 1 {
                    let (_, target_ty) = fields[0];
                    Some(Prism {
                        ctor_name: name,
                        source_ty: Ty(0), // Will be filled during integration
                        target_ty,
                    })
                } else {
                    None
                }
            })
            .collect()
    }

    use crate::ty::{Ty, TyKind, TyArena};
    
    /// Lockstep structural traversal to find the exact point of divergence.
    /// Returns `None` if the types are structurally identical.
    /// Returns `Some(path)` describing where they diverge.
    ///
    /// See FR-35, TC-15.
    pub fn zip_diff(
        arena: &TyArena,
        t1: Ty,
        t2: Ty,
        path: String,
    ) -> Option<String> {
        let k1 = arena.get(t1);
        let k2 = arena.get(t2);
    
        match (k1, k2) {
            // Same primitive — no divergence
            (TyKind::Int, TyKind::Int) => None,
            (TyKind::Bool, TyKind::Bool) => None,
            (TyKind::Float, TyKind::Float) => None,
            (TyKind::Str, TyKind::Str) => None,
            (TyKind::Unit, TyKind::Unit) => None,
            (TyKind::Never, TyKind::Never) => None,
            (TyKind::Error, TyKind::Error) => None,
    
            // Same named type
            (TyKind::Named(s1), TyKind::Named(s2)) if s1 == s2 => None,
    
            // Same type constructor with same arity — recurse
            (TyKind::App(s1, a1), TyKind::App(s2, a2)) if s1 == s2 && a1.len() == a2.len() => {
                a1.iter().zip(a2.iter()).enumerate().find_map(|(i, (a, b))| {
                    zip_diff(arena, *a, *b, format!("{path}.T{i}"))
                })
            }
    
            // Same function type with same arity — recurse
            (TyKind::Fn(p1, r1), TyKind::Fn(p2, r2)) if p1.len() == p2.len() => {
                // Check params
                for (i, (pa, pb)) in p1.iter().zip(p2.iter()).enumerate() {
                    if let Some(diff) = zip_diff(arena, *pa, *pb, format!("{path}.param{i}")) {
                        return Some(diff);
                    }
                }
                // Check return
                zip_diff(arena, *r1, *r2, format!("{path}.ret"))
            }
    
            // Same raw pointer — recurse
            (TyKind::RawPtr(i1), TyKind::RawPtr(i2)) => {
                zip_diff(arena, *i1, *i2, format!("{path}.*"))
            }
    
            // Infer/Error — can't diff meaningfully
            (TyKind::Infer, _) | (_, TyKind::Infer) => None,
            (TyKind::Error, _) | (_, TyKind::Error) => None,
    
            // Everything else is a divergence
            _ => Some(format!(
                "Diverged at {path}: expected {:?}, found {:?}",
                k1, k2
            )),
        }
    }

    use crate::diagnostics::AutoFix;
    use crate::ty::{Ty, TyKind, TyArena};
    use glyim_diag::Span;
    
    /// Known wrapper type names for autofix suggestions.
    const OPTION_SYM_STR: &str = "Option";
    const RESULT_SYM_STR: &str = "Result";
    
    /// Ask "What wrapper makes this compile?"
    /// If expected is Option<T> and found is T, suggest wrapping in Some.
    /// If expected is Result<T, E> and found is T, suggest wrapping in Ok.
    ///
    /// See FR-36, TC-16.
    pub fn bi_abductive_synthesis(
        arena: &TyArena,
        expected: Ty,
        found: Ty,
    ) -> Option<AutoFix> {
        let expected_kind = arena.get(expected);
    
        match expected_kind {
            TyKind::App(sym, args) => {
                let name = format!("{:?}", sym); // FIXME: resolve symbol to string
    
                if name.contains(OPTION_SYM_STR) && args.len() == 1 {
                    // Expected Option<T>, check if found is T
                    let inner = args[0];
                    if types_match(arena, inner, found) {
                        return Some(AutoFix::WrapWithOptions(Span::new(0, 0)));
                    }
                }
    
                if name.contains(RESULT_SYM_STR) && args.len() >= 1 {
                    // Expected Result<T, E>, check if found is T
                    let inner = args[0];
                    if types_match(arena, inner, found) {
                        return Some(AutoFix::WrapWithOk(Span::new(0, 0)));
                    }
                }
    
                None
            }
            _ => None,
        }
    }
    
    /// Check if two types match structurally (ignoring inference variables).
    fn types_match(arena: &TyArena, a: Ty, b: Ty) -> bool {
        match (arena.get(a), arena.get(b)) {
            (TyKind::Int, TyKind::Int) => true,
            (TyKind::Bool, TyKind::Bool) => true,
            (TyKind::Float, TyKind::Float) => true,
            (TyKind::Str, TyKind::Str) => true,
            (TyKind::Unit, TyKind::Unit) => true,
            (TyKind::Never, TyKind::Never) => true,
            (TyKind::Error, _) | (_, TyKind::Error) => true,
            (TyKind::Infer, _) | (_, TyKind::Infer) => true,
            (TyKind::Named(s1), TyKind::Named(s2)) => s1 == s2,
            (TyKind::App(s1, a1), TyKind::App(s2, a2)) => {
                s1 == s2 && a1.len() == a2.len() && a1.iter().zip(a2).all(|(x, y)| types_match(arena, *x, *y))
            }
            (TyKind::RawPtr(i1), TyKind::RawPtr(i2)) => types_match(arena, *i1, *i2),
            (TyKind::Fn(p1, r1), TyKind::Fn(p2, r2)) => {
                p1.len() == p2.len()
                    && p1.iter().zip(p2).all(|(a, b)| types_match(arena, *a, *b))
                    && types_match(arena, *r1, *r2)
            }
            _ => false,
        }
    }

    use crate::diagnostics::TypeError;
    use glyim_diag::Span;
    
    /// The stage at which an expression is evaluated.
    #[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
    pub enum Level {
        Comptime = 0,
        Buildtime = 1,
        Runtime = 2,
    }
    
    /// Check that a value defined at `defined_at` level can be used at `used_at` level.
    /// A value can be used at the same level or a later level (cross-stage persistence).
    /// Using a later-stage value at an earlier stage is a phase violation.
    ///
    /// See FR-33, FR-39.
    pub fn check_phase_consistency(
        defined_at: Level,
        used_at: Level,
    ) -> Result<(), PhaseViolation> {
        if defined_at <= used_at {
            Ok(())
        } else {
            Err(PhaseViolation {
                defined_at,
                used_at,
            })
        }
    }
    
    /// Information about a phase violation.
    #[derive(Clone, Debug)]
    pub struct PhaseViolation {
        pub defined_at: Level,
        pub used_at: Level,
    }
    
    impl PhaseViolation {
        pub fn to_type_error(&self, span: Span) -> TypeError {
            TypeError::PhaseViolation {
                span,
                used_at: format!("{:?}", self.used_at),
                defined_at: format!("{:?}", self.defined_at),
            }
        }
    }
    
    impl std::fmt::Display for Level {
        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
            match self {
                Level::Comptime => write!(f, "comptime"),
                Level::Buildtime => write!(f, "buildtime"),
                Level::Runtime => write!(f, "runtime"),
            }
        }
    }

    /// A mapping from type variables to concrete types for CHR rule instantiation.
    #[derive(Clone, Debug)]
    pub struct Substitution {
        pub mappings: Vec<(Ty, Ty)>,
    }
    
    impl Substitution {
        pub fn new() -> Self {
            Self { mappings: vec![] }
        }
    
        pub fn add(&mut self, from: Ty, to: Ty) {
            self.mappings.push((from, to));
        }
    
        pub fn lookup(&self, ty: Ty) -> Option<Ty> {
            self.mappings.iter().find_map(|(from, to)| {
                if *from == ty { Some(*to) } else { None }
            })
        }
    }
    
    impl Default for Substitution {
        fn default() -> Self {
            Self::new()
        }
    }
    
    /// Apply a substitution to a type, replacing mapped variables.
    pub fn apply_substitution(arena: &crate::ty::TyArena, sub: &Substitution, ty: Ty) -> Ty {
        if let Some(replacement) = sub.lookup(ty) {
            return replacement;
        }
    
        match arena.get(ty) {
            crate::ty::TyKind::App(sym, args) => {
                let new_args: Vec<Ty> = args.iter().map(|a| apply_substitution(arena, sub, *a)).collect();
                // We need a new arena entry for the substituted type.
                // Since we can't mutably borrow arena here, we return the original
                // if no substitution occurred, otherwise we need to allocate.
                // For now, if args changed, we indicate that substitution happened.
                // The caller is responsible for allocating in the arena.
                // This is a design limitation — we'll refactor to take &mut TyArena.
                let _ = (sym, new_args);
                ty // FIXME: need &mut TyArena to allocate substituted type
            }
            _ => ty,
        }
    }
    
    /// Apply a substitution to a goal.
    pub fn apply_substitution_to_goal(
        arena: &crate::ty::TyArena,
        sub: &Substitution,
        goal: &Goal,
    ) -> Goal {
        match goal {
            Goal::TraitImpl(sym, tys) => {
                Goal::TraitImpl(*sym, tys.iter().map(|t| apply_substitution(arena, sub, *t)).collect())
            }
            Goal::StateTransition(sym, from, to) => {
                Goal::StateTransition(*sym, apply_substitution(arena, sub, *from), apply_substitution(arena, sub, *to))
            }
            Goal::Reflectable(ty) => {
                Goal::Reflectable(apply_substitution(arena, sub, *ty))
            }
            Goal::HasField(ty, name) => {
                Goal::HasField(apply_substitution(arena, sub, *ty), *name)
            }
        }
    }

    use glyim_typeck::unify::UnificationTable;
    
    /// Simplified TyKind for fuzzing — we generate a small set of types
    /// and try to unify them in various combinations.
    fn generate_ty(arena: &mut TyArena, choice: u8) -> Ty {
        match choice % 8 {
            0 => arena.alloc(TyKind::Int),
            1 => arena.alloc(TyKind::Bool),
            2 => arena.alloc(TyKind::Float),
            3 => arena.alloc(TyKind::Str),
            4 => arena.alloc(TyKind::Unit),
            5 => arena.fresh_infer(glyim_diag::Span::new(0, 1)),
            6 => arena.alloc(TyKind::Error),
            _ => arena.alloc(TyKind::Never),
        }
    }

        if data.len() < 2 {

        }

        let mut table = UnificationTable::new();

        let b = generate_ty(&mut arena, data[1]);
