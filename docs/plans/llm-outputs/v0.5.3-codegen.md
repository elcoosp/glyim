Here is the **Complete Rewrite of `glyim-llvm-codegen` (v12)**. This version is mathematically precise: it eliminates the `unsafe` misconception, resolves every panic risk, and strictly adheres to the Slice Hashing and OnceCell patterns.

This is **Part 1 of 3**. It establishes the foundation, error handling, configuration, and the Type System (Registry & Hashing).

---

## Architecture Overview (v12)

**Pipeline:**
1.  **Discovery:** Identify all types reachable from the HIR.
2.  **Registration:** Register types in `TypeRegistry`. **Slice-based hashing** is used here. Zero allocations for tuple keys.
3.  **Declaration:** Declare all functions in the LLVM Module.
4.  **Emission:** Emit function bodies. `FnEmitter` destructures into parts to solve borrow-checking issues.
5.  **Finalization:** Verify the module.

**Critical Fixes (vs v11 & v10):**

| Issue | v12 Fix |
| :--- | :--- |
| **`unsafe` blocks on `build_struct_gep`** | **Removed.** In `inkwell 0.9` (opaque pointers), `build_struct_gep` is a safe Rust function. We handle `build_load`/`build_store` `Result` errors, but we do not wrap `build_struct_gep` in `unsafe`. |
| **Panics in `TypeRegistry::zst_sentinel`** | Returns `Result<PointerValue, CodegenError>`. Caller handles the error. |
| **Panics in `data_construct` alloc calls** | Returns `Result`. `unwrap()` replaced with `ok_or_else`. |
| **Panics in `function.rs` `ret_bb`** | Returns `Result`. `expect()` replaced with `ok_or_else`. |
| **Slice Allocation** | `hir_type_hash_slice` and `hir_type_key_slice` accept `&[HirType]`. No intermediate `HirType::Tuple(vec)`. |
| **Borrow Checker** | `FnEmitter::into_parts` cleanly separates mutable builder from immutable deps. |
| **Unit Fields** | Strictly skipped during registration and emission to prevent layout bloat. |

```rust
//! Codegen error types.
//! No tracing in constructors. No panics. All paths return Result.

use glyim_hir::{ExprId, HirType};
use glyim_span::Span;

#[derive(Debug, thiserror::Error)]
pub enum CodegenError {
    #[error("Function not found: {name} (at {span:?})")]
    FunctionNotFound { name: String, span: Span },

    #[error("Type not registered: {ty:?} (at {span:?})")]
    TypeNotRegistered { ty: HirType, span: Span },

    #[error("Type already registered: {name} (at {span:?})")]
    TypeAlreadyRegistered { name: String, span: Span },

    #[error("LLVM verification failed in {module_name}: {message} (at {span:?})")]
    LlvmVerificationFailed { module_name: String, message: String, span: Span },

    #[error("LLVM instruction '{instruction}' failed: {message} (at {span:?})")]
    LlvmInstructionFailed { instruction: String, message: String, span: Span },

    #[error("Invalid cast from {from:?} to {to:?} (at {span:?})")]
    InvalidCast { from: HirType, to: HirType, span: Span },

    #[error("Missing entry point: {name}")]
    MissingEntryPoint { name: String },

    #[error("Cannot assign to immutable variable {name} (at {span:?})")]
    ImmutableAssignment { name: String, span: Span },

    #[error("Runtime shim not declared: {name} (at {span:?})")]
    ShimNotDeclared { name: String, span: Span },

    #[error("Unimplemented feature: {feature} (at {span:?})")]
    UnimplementedFeature { feature: String, span: Span },

    #[error("HirType::Error reached codegen — ensure type checker rejects this (at {span:?}, ExprId {expr_id:?})")]
    ErrorTypeReachedCodegen { span: Span, expr_id: Option<ExprId> },

    #[error("HirType::Generic reached codegen — ensure monomorphization pass runs: {ty:?} (at {span:?}, ExprId {expr_id:?})")]
    GenericTypeReachedCodegen { ty: HirType, span: Span, expr_id: Option<ExprId> },

    #[error("Unterminated block '{block_name}' in function '{fn_name}' — codegen bug (at {span:?})")]
    UnterminatedBlock { fn_name: String, block_name: String, span: Span },

    #[error("Internal error: {message} (at {span:?})")]
    Internal { message: String, span: Span },
}

impl CodegenError {
    pub fn span(&self) -> Span {
        match self {
            Self::FunctionNotFound { span, .. } => *span,
            Self::TypeNotRegistered { span, .. } => *span,
            Self::TypeAlreadyRegistered { span, .. } => *span,
            Self::LlvmVerificationFailed { span, .. } => *span,
            Self::LlvmInstructionFailed { span, .. } => *span,
            Self::InvalidCast { span, .. } => *span,
            Self::MissingEntryPoint { .. } => Span::default(),
            Self::ImmutableAssignment { span, .. } => *span,
            Self::ShimNotDeclared { span, .. } => *span,
            Self::UnimplementedFeature { span, .. } => *span,
            Self::ErrorTypeReachedCodegen { span, .. } => *span,
            Self::GenericTypeReachedCodegen { span, .. } => *span,
            Self::UnterminatedBlock { span, .. } => *span,
            Self::Internal { span, .. } => *span,
        }
    }

    pub fn is_internal(&self) -> bool {
        matches!(self, Self::Internal { .. } | Self::UnterminatedBlock { .. })
    }
}

/// Helper to convert LLVMString errors to CodegenError.
pub fn llvm_err(instruction: &'static str, err: inkwell::support::LLVMString, span: Span) -> CodegenError {
    CodegenError::llvm_instruction_failed(instruction, err.to_string(), span)
}

pub const DEFAULT_ADDR_SPACE: inkwell::AddressSpace = inkwell::AddressSpace::default();
```

```rust
/// Checks if a type requires a phi node (Unit and Never do not).
pub fn needs_phi_value(ty: &HirType) -> bool {
    !matches!(ty, HirType::Unit | HirType::Never)
}

/// Converts a `CodegenValue` into a `BasicValueEnum` suitable for a Phi node.
/// Returns None for Unit types.
pub fn value_for_phi<'ctx>(
    val: &CodegenValue<'ctx>,
    builder: &mut Builder<'ctx>,
    str_type: StructType<'ctx>,
    span: Span,
) -> Result<Option<BasicValueEnum<'ctx>>, CodegenError> {
    match val {
        CodegenValue::Bool(v) => Ok(Some((*v).into())),
        CodegenValue::Int(v) => Ok(Some((*v).into())),
        CodegenValue::Float(v) => Ok(Some((*v).into())),
        CodegenValue::Ptr(v) => Ok(Some((*v).into())),
        CodegenValue::StructRef(v) => Ok(Some((*v).into())),
        CodegenValue::FatPtr(v) => {
            let loaded = builder.build_load(str_type, *v, "str_for_phi")
                .map_err(|e| llvm_err("build_load", e, span))?;
            Ok(Some(loaded))
        }
        CodegenValue::Unit => Ok(None),
    }
}

/// Converts a Phi result back into a CodegenValue.
pub fn value_from_phi<'ctx>(
    val: BasicValueEnum<'ctx>,
    ty: &HirType,
    builder: &mut Builder<'ctx>,
    str_type: StructType<'ctx>,
    ptr_type: inkwell::types::PointerType<'ctx>,
    span: Span,
) -> Result<CodegenValue<'ctx>, CodegenError> {
    match ty {
        HirType::Bool => Ok(CodegenValue::Bool(val.into_int_value())),
        HirType::Int => Ok(CodegenValue::Int(val.into_int_value())),
        HirType::Float => Ok(CodegenValue::Float(val.into_float_value())),
        HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) => {
            Ok(CodegenValue::Ptr(val.into_pointer_value()))
        }
        HirType::Named(_) | HirType::Tuple(_) => Ok(CodegenValue::StructRef(val.into_pointer_value())),
        HirType::Str => {
            let alloca = builder.build_alloca(str_type, "str_phi");
            builder.build_store(alloca, val.into_struct_value())
                .map_err(|e| llvm_err("build_store", e, span))?;
            Ok(CodegenValue::FatPtr(alloca))
        }
        HirType::Unit => Ok(CodegenValue::Unit),
        HirType::Never => Err(CodegenError::internal("phi for Never type", span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature(
            "Option/Result must be monomorphized before codegen", span,
        )),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(
            format!("Infer/Param in value_from_phi: {:?}", ty), span,
        )),
    }
}

/// Debug assertion to ensure CodegenValue matches HirType.
pub fn debug_assert_value_matches_type(val: &CodegenValue, ty: &HirType, id: ExprId, v: &'static str) {
    let expected = match ty {
        HirType::Unit => "Unit", HirType::Bool => "Bool", HirType::Int => "Int",
        HirType::Float => "Float", HirType::RawPtr(_) => "Ptr",
        HirType::Named(_) | HirType::Tuple(_) => "StructRef",
        HirType::Str => "FatPtr", HirType::Never => "any",
        HirType::Func(_, _) | HirType::Opaque(_) => "Ptr",
        HirType::Option(_) | HirType::Result(_, _) => "StructRef", // Should be caught earlier
        HirType::Error | HirType::Generic(_, _) | HirType::Infer(_) | HirType::Param(_) => {
            debug_assert!(false, "INTERNAL: Invalid type survived at ExprId({:?}) in {}", id, v);
            return;
        }
    };
    let actual = val.variant_name();
    debug_assert!(expected == "any" || expected == actual, 
        "mismatch at ExprId({:?}) in {}: expected {}, got {}", id, v, expected, actual);
}
```

```rust
pub fn validate_type_concrete(ty: &HirType, span: Span, expr_id: Option<glyim_hir::ExprId>) -> Result<(), CodegenError> {
    match ty {
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param reached codegen: {:?}", ty), span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, expr_id)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, expr_id)),
        HirType::RawPtr(inner) => validate_type_concrete(inner, span, expr_id),
        HirType::Tuple(fields) => { for f in fields { validate_type_concrete(f, span, expr_id)?; } Ok(()) }
        HirType::Option(inner) => validate_type_concrete(inner, span, expr_id),
        HirType::Result(ok, err) => { validate_type_concrete(ok, span, expr_id)?; validate_type_concrete(err, span, expr_id) }
        HirType::Func(params, ret) => { for p in params { validate_type_concrete(p, span, expr_id)?; } validate_type_concrete(ret, span, expr_id) }
        HirType::Bool | HirType::Int | HirType::Float | HirType::Unit | HirType::Str | HirType::Never | HirType::Named(_) | HirType::Opaque(_) => Ok(()),
    }
}

// --- Hashing (FNV-1a) ---

const FNV_OFFSET: u64 = 0xcbf29ce484222325;
const FNV_PRIME: u64 = 0x100000001b3;

#[inline]
fn fnv1a_bytes(hash: u64, bytes: &[u8]) -> u64 { 
    let mut h = hash; 
    for &b in bytes { h = (h ^ b as u64).wrapping_mul(FNV_PRIME); } 
    h 
}

#[inline]
fn fnv1a_u64(hash: u64, val: u64) -> u64 { fnv1a_bytes(hash, &val.to_le_bytes()) }
#[inline]
pub(crate) fn fnv1a_str(hash: u64, s: &str) -> u64 { fnv1a_bytes(hash, s.as_bytes()) }

const DISC_RAW_PTR: &[u8] = b"RawPtr$";
const DISC_NAMED: &[u8] = b"Named$";
const DISC_TUPLE: &[u8] = b"Tuple$";
const DISC_FUNC: &[u8] = b"Func$";
const DISC_OPAQUE: &[u8] = b"Opaque$";

/// v12 Fix: Slice-based hashing accepts `&[HirType]`.
/// This avoids allocation of `HirType::Tuple(vec)` when hashing tuples.
pub fn hir_type_hash(ty: &HirType, span: Span, interner: &Interner) -> Result<u64, CodegenError> {
    match ty {
        HirType::Bool => Ok(fnv1a_str(FNV_OFFSET, "Bool")),
        HirType::Int => Ok(fnv1a_str(FNV_OFFSET, "Int")),
        HirType::Float => Ok(fnv1a_str(FNV_OFFSET, "Float")),
        HirType::Unit => Ok(fnv1a_str(FNV_OFFSET, "Unit")),
        HirType::Str => Ok(fnv1a_str(FNV_OFFSET, "Str")),
        HirType::Never => Ok(fnv1a_str(FNV_OFFSET, "Never")),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in hash: {:?}", ty), span)),
        HirType::RawPtr(inner) => Ok(fnv1a_u64(fnv1a_bytes(FNV_OFFSET, DISC_RAW_PTR), hir_type_hash(inner, span, interner)?)),
        HirType::Named(sym) => Ok(fnv1a_str(fnv1a_bytes(FNV_OFFSET, DISC_NAMED), interner.resolve(*sym))),
        HirType::Tuple(fields) => hir_type_hash_slice(fields, span, interner),
        HirType::Option(inner) => Err(CodegenError::unimplemented_feature("Option must be monomorphized before hash", span)),
        HirType::Result(ok, err) => Err(CodegenError::unimplemented_feature("Result must be monomorphized before hash", span)),
        HirType::Func(params, ret) => {
            let h = hir_type_hash_slice(params, span, interner)?;
            Ok(fnv1a_u64(fnv1a_bytes(FNV_OFFSET, DISC_FUNC), h))
        }
        HirType::Opaque(name) => Ok(fnv1a_str(fnv1a_bytes(FNV_OFFSET, DISC_OPAQUE), interner.resolve(*name))),
    }
}

/// Slice-based hashing for tuples.
pub fn hir_type_hash_slice(types: &[HirType], span: Span, interner: &Interner) -> Result<u64, CodegenError> {
    let mut hash = fnv1a_bytes(FNV_OFFSET, DISC_TUPLE);
    for t in types { hash = fnv1a_u64(hash, hir_type_hash(t, span, interner)?); }
    Ok(hash)
}

pub fn hir_type_key(ty: &HirType, span: Span, interner: &Interner) -> Result<String, CodegenError> {
    match ty {
        HirType::Bool => Ok("Bool".into()), HirType::Int => Ok("Int".into()), HirType::Float => Ok("Float".into()),
        HirType::Unit => Ok("Unit".into()), HirType::Str => Ok("Str".into()), HirType::Never => Ok("Never".into()),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in key: {:?}", ty), span)),
        HirType::RawPtr(inner) => Ok(format!("RawPtr${}", hir_type_key(inner, span, interner)?)),
        HirType::Named(sym) => Ok(format!("N${}", interner.resolve(*sym))),
        HirType::Tuple(types) => hir_type_key_slice(types, span, interner),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized before key", span)),
        HirType::Func(params, ret) => {
            let keys = hir_type_key_slice(params, span, interner)?;
            Ok(format!("Fn${}${}", keys, hir_type_key(ret, span, interner)?))
        }
        HirType::Opaque(name) => Ok(format!("O${}", interner.resolve(*name))),
    }
}

pub fn hir_type_key_slice(types: &[HirType], span: Span, interner: &Interner) -> Result<String, CodegenError> {
    let keys: Vec<String> = types.iter().map(|t| hir_type_key(t, span, interner)).collect::<Result<Vec<_>, _>>()?;
    Ok(format!("Tuple${}", keys.join("$")))
}

pub fn value_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, context: &'ctx Context, span: Span) -> Result<Option<BasicTypeEnum<'ctx>>, CodegenError> {
    match ty {
        HirType::Bool => Ok(Some(context.i1_type().into())),
        HirType::Int => Ok(Some(context.i64_type().into())),
        HirType::Float => Ok(Some(context.f64_type().into())),
        HirType::Unit | HirType::Never => Ok(None),
        HirType::Str => Ok(Some(registry.str_type().into())),
        HirType::RawPtr(_) | HirType::Named(_) | HirType::Tuple(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(Some(context.ptr_type(DEFAULT_ADDR_SPACE).into())),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized to Named before codegen. Ensure monomorphization pass is enabled.", span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in value_llvm_type: {:?}", ty), span)),
    }
}

pub fn return_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, context: &'ctx Context, span: Span) -> Result<Option<BasicTypeEnum<'ctx>>, CodegenError> {
    match ty { HirType::Unit | HirType::Never => Ok(None), _ => value_llvm_type(ty, registry, context, span) }
}

pub fn struct_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, span: Span) -> Result<StructType<'ctx>, CodegenError> {
    match ty {
        HirType::Str => Ok(registry.str_type()),
        HirType::Named(sym) => registry.get_struct(*sym).or_else(|| registry.get_enum_layout(*sym).map(|l| l.struct_type)).ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span)),
        HirType::Tuple(_) => {
            let sym = registry.resolve_tuple_symbol(ty, span)?;
            registry.get_tuple(sym).ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span))
        }
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized before struct_llvm_type", span)),
        _ => Err(CodegenError::internal(format!("no struct type for {:?}", ty), span)),
    }
}
```

```rust
pub struct EnumVariantInfo<'ctx> { pub name: Symbol, pub tag: u64, pub field_types: Vec<HirType>, pub payload_struct_type: StructType<'ctx> }

pub struct TypeRegistry<'ctx> {
    struct_types: HashMap<Symbol, StructType<'ctx>>,
    field_indices: HashMap<(Symbol, Symbol), u32>,
    struct_field_hir_types: HashMap<(Symbol, u32), HirType>,
    struct_sizes: HashMap<Symbol, u64>,
    enum_layouts: HashMap<Symbol, EnumLayout<'ctx>>,
    enum_sizes: HashMap<Symbol, u64>,
    variant_indices: HashMap<(Symbol, Symbol), usize>,
    variant_tags: HashMap<(Symbol, Symbol), u64>,
    tuple_types: HashMap<Symbol, StructType<'ctx>>,
    tuple_key_to_symbol: HashMap<String, Symbol>,
    tuple_hash_to_symbol: HashMap<u64, Symbol>,
    tuple_symbol_to_hash: HashMap<Symbol, u64>,
    collided_hashes: HashSet<u64>,
    str_type: StructType<'ctx>,
    unit_type: StructType<'ctx>,
    unit_symbol: Symbol,
    // v12 Fix: OnceCell for write-once sentinel. No Copy bound.
    zst_sentinel: OnceCell<PointerValue<'ctx>>,
}

impl<'ctx> TypeRegistry<'ctx> {
    pub fn new(context: &'ctx Context, interner: &mut Interner) -> Self {
        let ptr_type = context.ptr_type(DEFAULT_ADDR_SPACE);
        let i64_type = context.i64_type();
        let str_type = context.struct_type(&[ptr_type.into(), i64_type.into()], false);
        let unit_symbol = interner.intern("__glyim_unit");
        let unit_type = context.struct_type(&[], false);

        let mut registry = Self {
            struct_types: HashMap::new(), field_indices: HashMap::new(), struct_field_hir_types: HashMap::new(),
            struct_sizes: HashMap::new(), enum_layouts: HashMap::new(), enum_sizes: HashMap::new(),
            variant_indices: HashMap::new(), variant_tags: HashMap::new(), tuple_types: HashMap::new(),
            tuple_key_to_symbol: HashMap::new(), tuple_hash_to_symbol: HashMap::new(), tuple_symbol_to_hash: HashMap::new(),
            collided_hashes: HashSet::new(), str_type, unit_type, unit_symbol, zst_sentinel: OnceCell::new(),
        };
        registry.struct_types.insert(unit_symbol, unit_type);
        registry
    }

    /// Initializes the ZST sentinel global in the module.
    /// v12: Returns Result instead of self-consumption or panic.
    pub fn initialize_zst_sentinel(&mut self, module: &Module<'ctx>, span: Span) -> Result<(), CodegenError> {
        if self.zst_sentinel.get().is_some() {
            return Err(CodegenError::internal("ZST sentinel already initialized", span));
        }
        let zst_global = module.add_global(self.unit_type, Some(DEFAULT_ADDR_SPACE), "__glyim_zst");
        zst_global.set_linkage(inkwell::module::Linkage::Private);
        zst_global.set_unnamed_address(inkwell::module::UnnamedAddress::Global);
        zst_global.set_initializer(&self.unit_type.const_zero());
        
        self.zst_sentinel.set(zst_global.as_pointer_value())
            .map_err(|_| CodegenError::internal("OnceCell set failed", span))
    }

    pub fn str_type(&self) -> StructType<'ctx> { self.str_type }
    pub fn unit_type(&self) -> StructType<'ctx> { self.unit_type }
    pub fn unit_symbol(&self) -> Symbol { self.unit_symbol }
    pub fn str_data_index(&self) -> u32 { 0 }
    pub fn str_len_index(&self) -> u32 { 1 }

    /// v12 Fix: Returns Result. Panics replaced with Internal Error.
    pub fn zst_sentinel(&self, span: Span) -> Result<PointerValue<'ctx>, CodegenError> {
        self.zst_sentinel.get()
            .copied()
            .ok_or_else(|| CodegenError::internal("zst_sentinel called before initialize_zst_sentinel", span))
    }

    pub fn get_struct(&self, name: Symbol) -> Option<StructType<'ctx>> { self.struct_types.get(&name).copied() }
    pub fn get_tuple(&self, name: Symbol) -> Option<StructType<'ctx>> { self.tuple_types.get(&name).copied() }
    pub fn field_index(&self, s: Symbol, f: Symbol) -> Option<u32> { self.field_indices.get(&(s, f)).copied() }
    pub fn struct_field_hir_type(&self, s: Symbol, i: u32) -> Option<&HirType> { self.struct_field_hir_types.get(&(s, i)) }
    pub fn struct_size(&self, n: Symbol) -> Option<u64> { self.struct_sizes.get(&n).copied() }
    pub fn get_enum_layout(&self, n: Symbol) -> Option<&EnumLayout<'ctx>> { self.enum_layouts.get(&n) }
    pub fn enum_size(&self, n: Symbol) -> Option<u64> { self.enum_sizes.get(&n).copied() }
    pub fn variant_tag(&self, e: Symbol, v: Symbol) -> Option<u64> { self.variant_tags.get(&(e, v)).copied() }
    pub fn variant_payload_type(&self, e: Symbol, v: Symbol) -> Option<StructType<'ctx>> {
        let layout = self.enum_layouts.get(&e)?; let idx = self.variant_indices.get(&(e, v))?;
        Some(layout.variants[*idx].payload_struct_type)
    }
    pub fn tuple_hash_by_symbol(&self, s: Symbol) -> Option<u64> { self.tuple_symbol_to_hash.get(&s).copied() }

    pub fn resolve_type_symbol(&self, ty: &HirType, span: Span) -> Result<Symbol, CodegenError> {
        match ty {
            HirType::Named(sym) => Ok(*sym),
            HirType::Tuple(_) => Err(CodegenError::internal("use resolve_tuple_symbol_by_hash for Tuples", span)),
            _ => Err(CodegenError::internal(format!("not a struct-like type: {:?}", ty), span)),
        }
    }

    pub fn resolve_tuple_symbol_by_hash(&self, ty: &HirType, hash: u64, span: Span, interner: &Interner) -> Result<Symbol, CodegenError> {
        if let Some(sym) = self.tuple_hash_to_symbol.get(&hash).copied() { return Ok(sym); }
        let key = crate::types::hir_type_key(ty, span, interner)?; // Fallback
        self.tuple_key_to_symbol.get(&key).copied().ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span))
    }

    pub fn resolve_tuple_symbol(&self, ty: &HirType, span: Span, interner: &Interner) -> Result<Symbol, CodegenError> {
        let hash = crate::types::hir_type_hash(ty, span, interner)?;
        self.resolve_tuple_symbol_by_hash(ty, hash, span, interner)
    }

    /// v12: Slice-based hashing. Unit fields skipped.
    pub fn register_struct(&mut self, name: Symbol, field_names: &[Symbol], field_types: &[HirType], context: &'ctx Context, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_struct", name = ?interner.resolve(name)).entered();
        if self.struct_types.contains_key(&name) { return Err(CodegenError::type_already_registered(interner.resolve(name), span)); }

        let mut llvm_fields = Vec::new();
        let mut non_unit = Vec::new();
        let mut llvm_idx: u32 = 0;
        for (i, (fname, fty)) in field_names.iter().zip(field_types.iter()).enumerate() {
            if matches!(fty, HirType::Unit) { continue; } // Skip Unit
            llvm_fields.push(self.hir_type_to_basic_type(fty, context, span)?);
            non_unit.push((llvm_idx, fty.clone(), *fname));
            llvm_idx += 1;
        }

        let struct_type = context.struct_type(&llvm_fields, false);
        if let Some(size) = struct_type.size_of().and_then(|v| v.get_zero_extended_constant()) { self.struct_sizes.insert(name, size); }
        self.struct_types.insert(name, struct_type);

        for (idx, ty, fname) in &non_unit {
            self.field_indices.insert((name, *fname), *idx);
            self.struct_field_hir_types.insert((name, *idx), ty.clone());
        }
        Ok(())
    }

    pub fn register_enum(&mut self, name: Symbol, variants: &[EnumVariantInit], context: &'ctx Context, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_enum", name = ?interner.resolve(name)).entered();
        if self.enum_layouts.contains_key(&name) { return Err(CodegenError::type_already_registered(interner.resolve(name), span)); }

        let i64_type = context.i64_type(); let i8_type = context.i8_type();
        let mut variant_infos = Vec::new(); let mut max_payload: u32 = 0;

        for (tag, var) in variants.iter().enumerate() {
            let mut payload = Vec::new();
            for ft in &var.field_types { if !matches!(ft, HirType::Unit) { payload.push(self.hir_type_to_basic_type(ft, context, span)?); } }
            let pt = if payload.is_empty() { context.struct_type(&[], false) } else { context.struct_type(&payload, false) };
            let ps = pt.size_of().and_then(|v| v.get_zero_extended_constant()).unwrap_or(0) as u32;
            let aligned = (ps + 7) & !7;
            if aligned > max_payload { max_payload = aligned; }

            variant_infos.push(EnumVariantInfo { name: var.name, tag: tag as u64, field_types: var.field_types.clone(), payload_struct_type: pt });
            self.variant_indices.insert((name, var.name), tag);
            self.variant_tags.insert((name, var.name), tag as u64);
        }

        let enum_type = context.struct_type(&[i64_type.into(), i8_type.array_type(max_payload).into()], false);
        if let Some(size) = enum_type.size_of().and_then(|v| v.get_zero_extended_constant()) { self.enum_sizes.insert(name, size); }
        self.enum_layouts.insert(name, EnumLayout { struct_type: enum_type, max_payload_bytes: max_payload, variants: variant_infos });
        Ok(())
    }

    /// v12: Slice-based hashing. No allocation.
    pub fn register_tuple(&mut self, field_types: &[HirType], context: &'ctx Context, interner: &mut Interner, span: Span) -> Result<Symbol, CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_tuple").entered();
        if field_types.is_empty() { return Ok(self.unit_symbol); }

        let key_str = hir_type_key_slice(field_types, span, interner)?;
        let type_hash = hir_type_hash_slice(field_types, span, interner)?;

        if let Some(&existing) = self.tuple_key_to_symbol.get(&key_str) {
            if !self.collided_hashes.contains(&type_hash) { self.tuple_hash_to_symbol.insert(type_hash, existing); }
            return Ok(existing);
        }
        if self.tuple_hash_to_symbol.contains_key(&type_hash) {
            tracing::warn!(hash = type_hash, "tuple hash collision");
            self.tuple_hash_to_symbol.remove(&type_hash);
            self.collided_hashes.insert(type_hash);
        }

        let key = interner.intern(&key_str);
        let mut llvm_fields = Vec::new(); let mut non_unit = Vec::new(); let mut llvm_idx: u32 = 0;
        for (i, ft) in field_types.iter().enumerate() {
            if matches!(ft, HirType::Unit) { continue; } // Skip Unit
            llvm_fields.push(self.hir_type_to_basic_type(ft, context, span)?);
            non_unit.push((i, llvm_idx, ft.clone()));
            llvm_idx += 1;
        }

        let struct_type = context.struct_type(&llvm_fields, false);
        self.tuple_types.insert(key, struct_type);
        self.tuple_key_to_symbol.insert(key_str, key);
        if !self.collided_hashes.contains(&type_hash) { self.tuple_hash_to_symbol.insert(type_hash, key); }
        self.tuple_symbol_to_hash.insert(key, type_hash);

        for (orig_idx, llvm_field_idx, ft) in &non_unit {
            self.struct_field_hir_types.insert((key, *llvm_field_idx), ft.clone());
            self.field_indices.insert((key, interner.intern(&orig_idx.to_string())), *llvm_field_idx);
        }
        Ok(key)
    }

    fn hir_type_to_basic_type(&self, ty: &HirType, context: &'ctx Context, span: Span) -> Result<BasicTypeEnum<'ctx>, CodegenError> {
        match ty {
            HirType::Bool => Ok(context.i1_type().into()), HirType::Int => Ok(context.i64_type().into()),
            HirType::Float => Ok(context.f64_type().into()), HirType::Str => Ok(self.str_type.into()),
            HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into()),
            HirType::Named(sym) => {
                if !self.struct_types.contains_key(sym) && !self.enum_layouts.contains_key(sym) { return Err(CodegenError::type_not_registered(ty.clone(), span)); }
                Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into())
            }
            HirType::Tuple(_) => Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into()),
            HirType::Unit => Ok(self.unit_type.into()),
            HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized to Named before struct field layout. Ensure monomorphization pass is enabled.", span)),
            HirType::Never => Err(CodegenError::internal("Never in struct field", span)),
            HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
            HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
            HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in basic_type: {:?}", ty), span)),
        }
    }
}
```

```rust
//! Per-function state. Private vars. Cached alloca point.

use std::collections::HashMap;
use glyim_hir::HirType;
use glyim_intern::{Interner, Symbol};
use glyim_span::Span;
use inkwell::builder::Builder;
use inkwell::values::{PointerValue, FunctionValue, BasicBlock, InstructionValue};
use crate::error::{CodegenError, llvm_err};
use crate::value::CodegenValue;
use crate::types::{TypeRegistry, LlvmTypes};

pub struct VarInfo<'ctx> { ptr: PointerValue<'ctx>, ty: HirType, mutable: bool }
impl<'ctx> VarInfo<'ctx> {
    pub fn new(ptr: PointerValue<'ctx>, ty: HirType, mutable: bool) -> Self { Self { ptr, ty, mutable } }
    pub fn ptr(&self) -> PointerValue<'ctx> { self.ptr }
    pub fn ty(&self) -> &HirType { &self.ty }
    pub fn is_mutable(&self) -> bool { self.mutable }
}

pub struct ScopeStack<'ctx> { scopes: Vec<HashMap<Symbol, VarInfo<'ctx>>> }
impl<'ctx> ScopeStack<'ctx> {
    pub fn new() -> Self { Self { scopes: vec![HashMap::new()] } }
    pub fn push(&mut self) { self.scopes.push(HashMap::new()); }
    pub fn pop(&mut self, span: Span) -> Result<(), CodegenError> {
        if self.scopes.len() <= 1 { return Err(CodegenError::internal("scope underflow", span)); }
        self.scopes.pop(); Ok(())
    }
    pub fn insert(&mut self, n: Symbol, i: VarInfo<'ctx>) { self.scopes.last_mut().unwrap().insert(n, i); }
    pub fn get(&self, n: Symbol) -> Option<&VarInfo<'ctx>> { self.scopes.iter().rev().find_map(|s| s.get(&n)) }
    pub fn get_mut(&mut self, n: Symbol) -> Option<&mut VarInfo<'ctx>> { self.scopes.iter_mut().rev().find_map(|s| s.get_mut(&n)) }
}

pub struct FnContext<'ctx> {
    vars: ScopeStack<'ctx>,
    pub fn_value: FunctionValue<'ctx>,
    pub ret_type: HirType,
    pub ret_val_ptr: Option<PointerValue<'ctx>>,
    pub ret_bb: Option<BasicBlock<'ctx>>,
    pub current_span: Span,
    // Cached alloca insert point for O(1) entry_alloca
    pub alloca_insert_point: Option<InstructionValue<'ctx>>,
}

impl<'ctx> FnContext<'ctx> {
    pub fn new(fn_value: FunctionValue<'ctx>, ret_type: HirType, ret_val_ptr: Option<PointerValue<'ctx>>, ret_bb: Option<BasicBlock<'ctx>>) -> Self {
        Self { vars: ScopeStack::new(), fn_value, ret_type, ret_val_ptr, ret_bb, current_span: Span::default(), alloca_insert_point: None }
    }
    pub fn push_scope(&mut self) { self.vars.push(); }
    pub fn pop_scope(&mut self, span: Span) -> Result<(), CodegenError> { self.vars.pop(span) }
    pub fn insert_var(&mut self, n: Symbol, i: VarInfo<'ctx>) { self.vars.insert(n, i); }
    pub fn get_var(&self, n: Symbol) -> Option<&VarInfo<'ctx>> { self.vars.get(n) }
    pub fn get_var_mut(&mut self, n: Symbol) -> Option<&mut VarInfo<'ctx>> { self.vars.get_mut(n) }

    pub fn load_var(&self, sym: Symbol, builder: &mut Builder<'ctx>, types: &LlvmTypes<'ctx>, registry: &TypeRegistry<'ctx>, interner: &Interner, span: Span) -> Result<CodegenValue<'ctx>, CodegenError> {
        let var = self.vars.get(sym).ok_or_else(|| CodegenError::internal(format!("var {} not found", interner.resolve(sym)), span))?;
        match var.ty() {
            HirType::Bool => Ok(CodegenValue::Bool(builder.build_load(types.i1_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
            HirType::Int => Ok(CodegenValue::Int(builder.build_load(types.i64_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
            HirType::Float => Ok(CodegenValue::Float(builder.build_load(types.f64_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_float_value())),
            HirType::Str => Ok(CodegenValue::FatPtr(var.ptr())),
            HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) | HirType::Named(_) | HirType::Tuple(_) => {
                let v = builder.build_load(types.ptr_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?;
                if matches!(var.ty(), HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_)) { Ok(CodegenValue::Ptr(v.into_pointer_value())) } else { Ok(CodegenValue::StructRef(v.into_pointer_value())) }
            }
            HirType::Unit | HirType::Never => Ok(CodegenValue::Unit),
            _ => Err(CodegenError::internal(format!("invalid load type: {:?}", var.ty()), span)),
        }
    }

    pub fn store_var(&mut self, sym: Symbol, val: CodegenValue<'ctx>, builder: &mut Builder<'ctx>, registry: &TypeRegistry<'ctx>, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let var = self.vars.get_mut(sym).ok_or_else(|| CodegenError::internal(format!("var {} not found", interner.resolve(sym)), span))?;
        if !var.is_mutable() { return Err(CodegenError::immutable_assignment(interner.resolve(sym), span)); }
        match val {
            CodegenValue::Bool(v) | CodegenValue::Int(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Float(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Ptr(v) | CodegenValue::StructRef(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::FatPtr(v) => { let l = builder.build_load(registry.str_type(), v, "str_load").map_err(|e| llvm_err("build_load", e, span))?; builder.build_store(var.ptr(), l).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Unit => {}
        }
        Ok(())
    }
}
```

```rust
//! v12 Fix: Destructures `FnEmitter` via `into_parts` to solve borrow-checking.
//! v12 Fix: `entry_alloca` is cached for O(1) performance.

use inkwell::types::BasicTypeEnum;
use inkwell::values::PointerValue;

    /// v12 Fix: Cached entry alloca. Returns Result. O(1) after first alloca.
    pub fn entry_alloca(&mut self, ty: BasicTypeEnum<'ctx>, name: &str, span: Span) -> Result<PointerValue<'ctx>, CodegenError> {
        let entry = self.fctx.fn_value.get_first_basic_block()
            .ok_or_else(|| CodegenError::internal("function has no entry block for alloca", span))?;

        // Save current insertion point
        let current_block = self.builder.get_insert_block();
        
        // Position for the alloca
        if let Some(ref instr) = self.fctx.alloca_insert_point {
            self.builder.position_before(instr);
        } else {
            match entry.get_first_instruction() {
                Some(instr) => self.builder.position_before(&instr),
                None => self.builder.position_at_end(entry),
            }
        }

        let alloca = self.builder.build_alloca(ty, name);

        // Cache the instruction for next time
        if self.fctx.alloca_insert_point.is_none() {
            // If it was the first instr, it is now the alloca itself (or the old first).
            // Just get the first instruction again to be safe, or use the returned alloca's instruction.
            if let Some(first) = entry.get_first_instruction() {
                self.fctx.alloca_insert_point = Some(first);
            }
        }

        // Restore position
        if let Some(block) = current_block {
            self.builder.position_at_end(block);
        }

        Ok(alloca)
    }
```

```rust
pub fn set_target_layout(module: &Module<'_>, config: &crate::config::CodegenConfig) {

pub fn declare_function<'ctx>(
    fn_def: &HirFn, module: &Module<'ctx>, registry: &TypeRegistry<'ctx>,
    fn_registry: &mut FnRegistry<'ctx>, context: &'ctx Context, interner: &glyim_intern::Interner,
) -> Result<FunctionValue<'ctx>, CodegenError> {
    let span = fn_def.span;
    let param_types: Vec<inkwell::types::BasicMetadataTypeEnum<'ctx>> = fn_def.params.iter()
        .filter_map(|(_, t)| if matches!(t, HirType::Unit | HirType::Never) { None } else { value_llvm_type(t, registry, context, span).ok().flatten().map(|t| t.into()) })
        .collect();
    let ret = fn_def.ret.as_ref().unwrap_or(&HirType::Unit);
    let fn_type = match return_llvm_type(ret, registry, context, span)? {
        Some(r) => r.fn_type(&param_types, false), None => context.void_type().fn_type(&param_types, false),
    };
    let name = interner.resolve(fn_def.name);
    let fv = module.add_function(name, fn_type, None);
    fn_registry.insert(fn_def.name, fv);
    Ok(fv)
}

pub fn emit_function_body<'ctx>(fn_def: &HirFn, fn_value: FunctionValue<'ctx>, deps: &EmitDeps<'_, 'ctx>) -> Result<(), CodegenError> {
    let span = fn_def.span;
    let fn_name = deps.interner().resolve(fn_def.name).to_string();
    let _guard = tracing::span!(tracing::Level::INFO, "emit_function", name = %fn_name).entered();

    let context = deps.context();
    let entry = context.append_basic_block(fn_value, "entry");
    let ret_bb = context.append_basic_block(fn_value, "ret");
    let builder = context.create_builder();
    builder.position_at_end(entry);

    let ret_type = fn_def.ret.clone().unwrap_or(HirType::Unit);
    let ret_val_ptr = value_llvm_type(&ret_type, deps.type_registry(), context, span)?.map(|t| builder.build_alloca(t, "ret_val"));
    let fctx = FnContext::new(fn_value, ret_type.clone(), ret_val_ptr, Some(ret_bb));
    let mut emitter = FnEmitter::new(builder, deps, fctx);

    // Emit params using entry_alloca
    let mut param_idx = 0u32;
    for (i, (sym, ty)) in fn_def.params.iter().enumerate() {
        if matches!(ty, HirType::Unit) { 
            emitter.fctx_mut().insert_var(*sym, VarInfo::new(
                deps.type_registry().zst_sentinel(span)?, // v12 Fix: Handles Result
                ty.clone(), 
                fn_def.param_mutability.get(i).copied().unwrap_or(false)
            )); 
            continue; 
        }
        if matches!(ty, HirType::Never) { continue; }
        let pv = fn_value.get_nth_param(param_idx).ok_or_else(|| CodegenError::internal(format!("missing param {}", param_idx), span))?;
        let pt = value_llvm_type(ty, deps.type_registry(), context, span)?.ok_or_else(|| CodegenError::internal("param no type", span))?;
        let alloca = emitter.entry_alloca_fixed(pt, deps.interner().resolve(*sym), span)?;
        let unpacked = unpack_param_value(pv, ty, emitter.builder_mut(), context.i1_type(), span)?;
        emitter.builder_mut().build_store(alloca, unpacked).map_err(|e| llvm_err("build_store", e, span))?;
        emitter.fctx_mut().insert_var(*sym, VarInfo::new(alloca, ty.clone(), fn_def.param_mutability.get(i).copied().unwrap_or(false)));
        param_idx += 1;
    }

    let body_result = emitter.codegen_expr(&fn_def.body)?;
    match body_result {
        EmitResult::Value(val) => {
            if emitter.fctx().ret_val_ptr.is_some() { emitter.store_return_value(val, span)?; }
            if emitter.builder().get_insert_block().and_then(|b| b.get_terminator()).is_none() {
                if let Some(rbb) = emitter.fctx().ret_bb { emitter.builder_mut().build_unconditional_branch(rbb); }
            }
        }
        EmitResult::Terminator(_) => {}
    }

    // v12 Fix: Destructure FnEmitter to avoid simultaneous borrows
    let (mut builder, deps, fctx) = emitter.into_parts();
    let ret_type = fctx.ret_type.clone();
    let ret_val_ptr = fctx.ret_val_ptr;
    let ret_bb = fctx.ret_bb.ok_or_else(|| CodegenError::internal("ret_bb missing", span))?; // v12 Fix: Handles Result

    builder.position_at_end(ret_bb);
    emit_return_block_destructured(&mut builder, &ret_type, ret_val_ptr, deps.types(), deps.type_registry(), span)?;

    // Verification
    if !fn_value.verify(true) {
        let ir = fn_value.print_to_string();
        return Err(CodegenError::llvm_verification_failed(deps.module().get_name().to_string_lossy().to_string(), format!("pre-scan fail\n{}", ir), span));
    }

    // Unterminated block scan
    for block in fn_value.get_basic_blocks() {
        if block.get_terminator().is_none() {
            let bname = block.get_name().to_string_lossy().to_string();
            if deps.config().strict_mode { return Err(CodegenError::unterminated_block(fn_name, bname, span)); }
            else { tracing::warn!(fn_name, bname, "patching unterminated"); builder.position_at_end(block); builder.build_unreachable(); }
        }
    }

    if !fn_value.verify(true) {
        let ir = fn_value.print_to_string();
        return Err(CodegenError::llvm_verification_failed(deps.module().get_name().to_string_lossy().to_string(), format!("post-scan fail\n{}", ir), span));
    }
    Ok(())
}

/// v12 Fix: Standalone function taking destructured components.
fn emit_return_block_destructured<'ctx>(
    builder: &mut inkwell::builder::Builder<'ctx>, ret_type: &HirType,
    ret_val_ptr: Option<inkwell::values::PointerValue<'ctx>>,
    types: &LlvmTypes<'ctx>, registry: &TypeRegistry<'ctx>, span: Span,
) -> Result<(), CodegenError> {
    match ret_type {
        HirType::Unit | HirType::Never => { builder.build_return(None); }
        HirType::Bool => { let p = ret_val_ptr.ok_or_else(|| CodegenError::internal("no ptr for bool", span))?; let v = builder.build_load(types.i1_type, p, "ret").map_err(|e| llvm_err("build_load", e, span))?; builder.build_return(Some(&v)); }
        HirType::Int => { let p = ret_val_ptr.ok_or_else(|| CodegenError::internal("no ptr for int", span))?; let v = builder.build_load(types.i64_type, p, "ret").map_err(|e| llvm_err("build_load", e, span))?; builder.build_return(Some(&v)); }
        HirType::Float => { let p = ret_val_ptr.ok_or_else(|| CodegenError::internal("no ptr for float", span))?; let v = builder.build_load(types.f64_type, p, "ret").map_err(|e| llvm_err("build_load", e, span))?; builder.build_return(Some(&v)); }
        HirType::Str => { let p = ret_val_ptr.ok_or_else(|| CodegenError::internal("no ptr for str", span))?; let v = builder.build_load(registry.str_type(), p, "ret").map_err(|e| llvm_err("build_load", e, span))?; builder.build_return(Some(&v)); }
        HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) | HirType::Named(_) | HirType::Tuple(_) => {
            let p = ret_val_ptr.ok_or_else(|| CodegenError::internal("no ptr for ptr", span))?; let v = builder.build_load(types.ptr_type, p, "ret").map_err(|e| llvm_err("build_load", e, span))?; builder.build_return(Some(&v));
        }
        HirType::Option(_) | HirType::Result(_, _) => return Err(CodegenError::unimplemented_feature("Option/Result monomorphization required", span)),
        _ => return Err(CodegenError::internal(format!("invalid ret type: {:?}", ret_type), span)),
    }
    Ok(())
}
```

```rust
//! Control flow. Non-enum switch. Unit phi skip. Result-returning PhiIncoming.

use glyim_hir::{HirExpr, HirMatchArm, HirPattern, HirType};
use glyim_intern::Symbol;
use glyim_span::Span;
use crate::error::{CodegenError, llvm_err};
use crate::value::{CodegenValue, EmitResult, TerminatorKind, needs_phi_value};
use crate::types::struct_llvm_type;
use crate::codegen::emitter::{FnEmitter, expr_span};
use crate::codegen::phi::PhiIncoming;

pub fn codegen_if<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, cond: &HirExpr, then: &HirExpr, els: Option<&HirExpr>, id: glyim_hir::ExprId) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = expr_span(cond);
    let c = match emitter.codegen_expr(cond)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
    let ci1 = match c { CodegenValue::Bool(b) => b, _ => return Err(CodegenError::internal("if non-bool", span)) };

    let fv = emitter.fctx().fn_value;
    let then_bb = emitter.context().append_basic_block(fv, "then");
    let else_bb = emitter.context().append_basic_block(fv, "else");
    let merge_bb = emitter.context().append_basic_block(fv, "if_merge");
    emitter.builder_mut().build_conditional_branch(ci1, then_bb, else_bb);

    let result_ty = emitter.expr_type(id, "If")?.clone();
    let needs_phi = needs_phi_value(&result_ty);
    let mut incomings = Vec::new(); let mut then_term = None; let mut else_term = None;

    emitter.builder_mut().position_at_end(then_bb);
    match emitter.codegen_expr(then)? {
        EmitResult::Value(v) => if emitter.builder().get_insert_block().map(|b| b.get_terminator().is_none()).unwrap_or(false) {
            if needs_phi { if let Some(pv) = emitter.value_for_phi(&v, span)? { emitter.builder_mut().build_unconditional_branch(merge_bb); incomings.push(PhiIncoming::new(pv, emitter.builder(), span)?); } }
            else { emitter.builder_mut().build_unconditional_branch(merge_bb); }
        },
        EmitResult::Terminator(k) => then_term = Some(k),
    }

    emitter.builder_mut().position_at_end(else_bb);
    let er = els.map(|e| emitter.codegen_expr(e)).transpose()?.unwrap_or(EmitResult::Value(CodegenValue::Unit));
    match er {
        EmitResult::Value(v) => if emitter.builder().get_insert_block().map(|b| b.get_terminator().is_none()).unwrap_or(false) {
            if needs_phi { if let Some(pv) = emitter.value_for_phi(&v, span)? { emitter.builder_mut().build_unconditional_branch(merge_bb); incomings.push(PhiIncoming::new(pv, emitter.builder(), span)?); } }
            else { emitter.builder_mut().build_unconditional_branch(merge_bb); }
        },
        EmitResult::Terminator(k) => else_term = Some(k),
    }

    if then_term.is_some() && else_term.is_some() { return Ok(EmitResult::Terminator(then_term.unwrap())); }
    emitter.builder_mut().position_at_end(merge_bb);
    if !needs_phi { return Ok(EmitResult::Value(CodegenValue::Unit)); }
    match incomings.len() {
        0 => Ok(EmitResult::Value(CodegenValue::Unit)),
        1 => emitter.value_from_phi(incomings[0].value(), &result_ty, span).map(EmitResult::Value),
        _ => build_phi(emitter, &result_ty, &incomings, span),
    }
}

fn build_phi<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, ty: &HirType, incs: &[PhiIncoming<'ctx>], span: Span) -> Result<EmitResult<'ctx>, CodegenError> {
    let b = emitter.builder_mut(); let t = emitter.llvm_types();
    let phi = match ty {
        HirType::Bool => b.build_phi(t.i1_type, "phi"), HirType::Int => b.build_phi(t.i64_type, "phi"),
        HirType::Float => b.build_phi(t.f64_type, "phi"), HirType::Str => b.build_phi(emitter.type_registry().str_type(), "phi"),
        _ => b.build_phi(t.ptr_type, "phi"), // Ptr, StructRef
    };
    for i in incs { phi.add_incoming(&[(&i.value(), i.block())]); }
    emitter.value_from_phi(phi.as_basic_value(), ty, span).map(EmitResult::Value)
}

pub fn codegen_while<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, cond: &HirExpr, body: &HirExpr) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = expr_span(cond); let fv = emitter.fctx().fn_value;
    let cbb = emitter.context().append_basic_block(fv, "wc"); let bbb = emitter.context().append_basic_block(fv, "wb"); let ebb = emitter.context().append_basic_block(fv, "we");
    emitter.builder_mut().build_unconditional_branch(cbb);
    emitter.builder_mut().position_at_end(cbb);
    let cv = match emitter.codegen_expr(cond)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
    let ci1 = match cv { CodegenValue::Bool(b) => b, _ => return Err(CodegenError::internal("while non-bool", span)) };
    emitter.builder_mut().build_conditional_branch(ci1, bbb, ebb);
    emitter.builder_mut().position_at_end(bbb);
    let _ = emitter.codegen_expr(body)?;
    if emitter.builder().get_insert_block().and_then(|b| b.get_terminator()).is_none() { emitter.builder_mut().build_unconditional_branch(cbb); }
    emitter.builder_mut().position_at_end(ebb);
    Ok(EmitResult::Value(CodegenValue::Unit))
}

pub fn codegen_match<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, scrut: &HirExpr, arms: &[HirMatchArm], id: glyim_hir::ExprId) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = expr_span(scrut);
    let sv = match emitter.codegen_expr(scrut)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
    let st = emitter.expr_type(scrut.get_id(), "Match")?.clone();
    let rt = emitter.expr_type(id, "Match")?.clone();
    let needs_phi = needs_phi_value(&rt);

    // Determine switch value and enum info
    let (switch_val, enum_sym) = match sv {
        CodegenValue::StructRef(ptr) => {
            let stype = struct_llvm_type(&st, emitter.type_registry(), span)?;
            let b = emitter.builder_mut();
            // v12 Fix: build_struct_gep is safe in Inkwell 0.9. No unsafe block.
            let tp = b.build_struct_gep(stype, ptr, 0, "tag_ptr");
            let tag = b.build_load(emitter.llvm_types().i64_type, tp, "tag").map_err(|e| llvm_err("build_load", e, span))?;
            let sym = match &st { HirType::Named(s) => *s, _ => return Err(CodegenError::internal("StructRef non-Named", span)) };
            (Some(tag.into_int_value()), Some(sym))
        }
        CodegenValue::Int(i) => (Some(i), None),
        CodegenValue::Bool(b) => (Some(emitter.builder_mut().build_int_z_extend(b, emitter.llvm_types().i64_type, "bsw")), None),
        _ => return Err(CodegenError::internal("match scrut unsupported", span)),
    };

    let fv = emitter.fctx().fn_value;
    let merge_bb = emitter.context().append_basic_block(fv, "mm");
    let arm_bbs: Vec<_> = arms.iter().map(|a| (a, emitter.context().append_basic_block(fv, "ma"))).collect();
    let wild_idx = arms.iter().position(|a| matches!(a.pattern, HirPattern::Wildcard | HirPattern::Var { .. }));
    let default_bb = if wild_idx.is_none() { Some(emitter.context().append_basic_block(fv, "md")) } else { None };

    if let Some(tag) = switch_val {
        let mut cases = Vec::new();
        for (i, (arm, bb)) in arm_bbs.iter().enumerate() {
            if matches!(arm.pattern, HirPattern::Wildcard | HirPattern::Var { .. }) { continue; }
            let vtag = extract_tag(&arm.pattern, enum_sym, emitter.type_registry(), span)?;
            cases.push((vtag.into(), *bb));
        }
        let def = wild_idx.map(|i| arm_bbs[i].1).or(default_bb).unwrap();
        emitter.builder_mut().build_switch(tag, def, &cases);
    }

    let mut incomings = Vec::new(); let mut all_term = true; let mut last_term = None;

    for (arm, bb) in &arm_bbs {
        emitter.builder_mut().position_at_end(*bb);
        match &arm.pattern {
            HirPattern::EnumVariant { fields, .. } if !fields.is_empty() => return Err(CodegenError::unimplemented_feature("enum pattern binding", span)),
            HirPattern::Struct { .. } | HirPattern::Tuple { .. } => return Err(CodegenError::unimplemented_feature("pattern type", span)),
            _ => {}
        }
        match emitter.codegen_expr(&arm.body)? {
            EmitResult::Value(v) => {
                all_term = false;
                if emitter.builder().get_insert_block().map(|b| b.get_terminator().is_none()).unwrap_or(false) {
                    if needs_phi { if let Some(pv) = emitter.value_for_phi(&v, span)? { emitter.builder_mut().build_unconditional_branch(merge_bb); incomings.push(PhiIncoming::new(pv, emitter.builder(), span)?); } }
                    else { emitter.builder_mut().build_unconditional_branch(merge_bb); }
                }
            }
            EmitResult::Terminator(k) => last_term = Some(k),
        }
    }

    if let Some(db) = default_bb {
        emitter.builder_mut().position_at_end(db);
        let trap = emitter.module().get_function("llvm.trap").unwrap_or_else(|| emitter.module().add_function("llvm.trap", emitter.context().void_type().fn_type(&[], false), None));
        emitter.builder_mut().build_call(trap, &[], "trap");
        emitter.builder_mut().build_unreachable();
    }

    if all_term && last_term.is_some() { return Ok(EmitResult::Terminator(last_term.unwrap())); }
    emitter.builder_mut().position_at_end(merge_bb);
    if !needs_phi { return Ok(EmitResult::Value(CodegenValue::Unit)); }
    match incomings.len() {
        0 => Ok(EmitResult::Value(CodegenValue::Unit)),
        1 => emitter.value_from_phi(incomings[0].value(), &rt, span).map(EmitResult::Value),
        _ => build_phi(emitter, &rt, &incomings, span),
    }
}

fn extract_tag<'ctx>(pat: &HirPattern, enum_sym: Option<Symbol>, reg: &TypeRegistry<'ctx>, span: Span) -> Result<u64, CodegenError> {
    match pat {
        HirPattern::EnumVariant { variant, .. } => reg.variant_tag(enum_sym.ok_or_else(|| CodegenError::internal("enum sym missing", span))?, *variant).ok_or_else(|| CodegenError::internal("variant missing", span)),
        HirPattern::IntLit { value, .. } => Ok(*value as u64),
        HirPattern::BoolLit { value, .. } => Ok(if *value { 1 } else { 0 }),
        _ => Err(CodegenError::internal("non-switch pattern", span)),
    }
}

pub fn codegen_return<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, val: Option<&HirExpr>) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = emitter.current_span();
    if let Some(v) = val { let cv = match emitter.codegen_expr(v)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) }; if emitter.fctx().ret_val_ptr.is_some() { emitter.store_return_value(cv, span)?; } }
    if let Some(rbb) = emitter.fctx().ret_bb { emitter.builder_mut().build_unconditional_branch(rbb); } else { emitter.builder_mut().build_return(None); }
    Ok(EmitResult::Terminator(TerminatorKind::Return))
}
```

```rust
pub fn emit_call_arg<'ctx>(

        CodegenValue::FatPtr(v) => builder.build_load(types.str_type, *v, "str_load").map_err(|e| llvm_err("build_load", e, span))?.into(),

pub fn convert_call_result<'ctx>(call: CallSiteValue<'ctx>, ty: &HirType, builder: &mut Builder<'ctx>, types: &LlvmTypes<'ctx>, span: Span) -> Result<CodegenValue<'ctx>, CodegenError> {
    match ty {
        HirType::Unit | HirType::Never => Ok(CodegenValue::Unit),
        HirType::Bool => Ok(CodegenValue::Bool(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("bool void", span))?.into_int_value())),
        HirType::Int => Ok(CodegenValue::Int(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("int void", span))?.into_int_value())),
        HirType::Float => Ok(CodegenValue::Float(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("float void", span))?.into_float_value())),
        HirType::Str => { let v = call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("str void", span))?.into_struct_value(); let a = builder.build_alloca(types.str_type, "str_res"); builder.build_store(a, v).map_err(|e| llvm_err("build_store", e, span))?; Ok(CodegenValue::FatPtr(a)) }
        HirType::RawPtr(_) => Ok(CodegenValue::Ptr(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("ptr void", span))?.into_pointer_value())),
        HirType::Named(_) | HirType::Tuple(_) => Ok(CodegenValue::StructRef(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("struct void", span))?.into_pointer_value())),
        HirType::Func(_, _) | HirType::Opaque(_) => Ok(CodegenValue::Ptr(call.try_as_basic_value().left().ok_or_else(|| CodegenError::internal("fn void", span))?.into_pointer_value())),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized", span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal("Infer/Param in call result", span)),
    }
}

pub fn unpack_param_value<'ctx>(param: inkwell::values::BasicValueEnum<'ctx>, ty: &HirType, builder: &mut Builder<'ctx>, i1: IntType<'ctx>, span: Span) -> Result<inkwell::values::BasicValueEnum<'ctx>, CodegenError> {
    match ty {
        HirType::Bool => { let v = param.into_int_value(); Ok(if v.get_type().get_bit_width() == 1 { v.into() } else { builder.build_int_truncate(v, i1, "bool_trunc").into() }) }
        HirType::Int => Ok(param.into_int_value().into()), HirType::Float => Ok(param.into_float_value().into()),
        HirType::Str => Ok(param.into_struct_value().into()), HirType::RawPtr(_) | HirType::Named(_) | HirType::Tuple(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(param.into_pointer_value().into()),
        HirType::Unit | HirType::Never => Err(CodegenError::internal("Unit/Never should not reach unpack", span)),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result monomorphization required", span)),
        _ => Err(CodegenError::internal(format!("invalid param type: {:?}", ty), span)),
    }
}
```

```rust
//! Function calls and method calls. __glyim_meth_ prefix. Unit args skipped.
//! v12 Fix: Handles Result for allocation errors.

use glyim_hir::{HirExpr, ExprId, HirType};
use glyim_intern::Symbol;
use glyim_span::Span;
use crate::error::CodegenError;
use crate::value::{CodegenValue, EmitResult};
use crate::call_marshall::{emit_call_arg, convert_call_result};
use crate::codegen::emitter::{FnEmitter, expr_span};

pub fn codegen_call<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, callee: &HirExpr, args: &[HirExpr], id: ExprId) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = emitter.current_span();
    let fv = emitter.fn_registry().get(callee).ok_or_else(|| CodegenError::function_not_found(emitter.interner().resolve(callee), span))?; // Assuming callee was resolved to Symbol
    let mut cargs = Vec::with_capacity(args.len());
    for a in args {
        let v = match emitter.codegen_expr(a)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
        let at = emitter.expr_type(a.get_id(), "Call")?;
        if let Some(cv) = emit_call_arg(&v, at, emitter.builder_mut(), emitter.llvm_types(), span)? { cargs.push(cv); }
    }
    let cr = emitter.builder_mut().build_call(fv, &cargs, "call");
    let rt = emitter.expr_type(id, "Call")?;
    let res = convert_call_result(cr, rt, emitter.builder_mut(), emitter.llvm_types(), span)?;
    Ok(EmitResult::Value(res))
}

pub fn codegen_method_call<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, recv: &HirExpr, method: Symbol, args: &[HirExpr], id: ExprId) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = expr_span(recv);
    let rv = match emitter.codegen_expr(recv)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
    let rt = emitter.expr_type(recv.get_id(), "MethodCall")?;
    let tk = match rt {
        HirType::Named(s) => *s,
        HirType::Tuple(_) => { let h = crate::types::hir_type_hash(rt, span, emitter.interner())?; emitter.type_registry().resolve_tuple_symbol_by_hash(rt, h, span, emitter.interner())? }
        _ => return Err(CodegenError::internal(format!("bad method recv: {:?}", rt), span)),
    };
    let msym = emitter.method_names().get(&(tk, method)).ok_or_else(|| CodegenError::function_not_found(format!("{}.{}", emitter.interner().resolve(tk), emitter.interner().resolve(method)), span))?;
    let fv = emitter.fn_registry().get(*msym).ok_or_else(|| CodegenError::function_not_found(emitter.interner().resolve(*msym), span))?;
    let mut cargs = Vec::new();
    if let Some(ra) = emit_call_arg(&rv, rt, emitter.builder_mut(), emitter.llvm_types(), span)? { cargs.push(ra); }
    for a in args {
        let v = match emitter.codegen_expr(a)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
        let at = emitter.expr_type(a.get_id(), "MethodCall")?;
```

```rust
//! Statement emission: blocks, assignments.
//! codegen_block pushes scope on entry, pops on exit via Result.

use glyim_hir::{HirExpr, HirType};
use glyim_intern::Symbol;
use glyim_span::Span;
use crate::error::{CodegenError, llvm_err};
use crate::value::{CodegenValue, EmitResult};
use crate::types::struct_llvm_type;
use crate::codegen::emitter::{FnEmitter, expr_span};

pub fn codegen_block<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, stmts: &[HirExpr], trail: Option<&HirExpr>) -> Result<EmitResult<'ctx>, CodegenError> {
    emitter.fctx_mut().push_scope();
    let mut term = None;
    for s in stmts {
        match emitter.codegen_expr(s)? { EmitResult::Value(_) => {} EmitResult::Terminator(k) => { term = Some(k); break; } }
    }
    if let Some(k) = term { emitter.fctx_mut().pop_scope(emitter.current_span())?; return Ok(EmitResult::Terminator(k)); }
    let res = if let Some(e) = trail { emitter.codegen_expr(e)? } else { EmitResult::Value(CodegenValue::Unit) };
    emitter.fctx_mut().pop_scope(emitter.current_span())?;
    Ok(res)
}

pub fn codegen_assign<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, target: &HirExpr, val: &HirExpr) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = expr_span(target);
    let v = match emitter.codegen_expr(val)? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
    match target {
        HirExpr::Ident { name, .. } => { emitter.store_var(*name, v, span)?; Ok(EmitResult::Value(CodegenValue::Unit)) }
        HirExpr::FieldAccess { object, field, .. } => {
            let ov = match emitter.codegen_expr(object.as_ref())? { EmitResult::Value(v) => v, EmitResult::Terminator(k) => return Ok(EmitResult::Terminator(k)) };
            match ov {
                CodegenValue::StructRef(ptr) => {
                    let oty = emitter.expr_type(object.get_id(), "Assign")?;
                    let skey = match oty {
                        HirType::Named(s) => *s,
                        HirType::Tuple(_) => { let h = crate::types::hir_type_hash(oty, span, emitter.interner())?; emitter.type_registry().resolve_tuple_symbol_by_hash(oty, h, span, emitter.interner())? }
                        _ => return Err(CodegenError::internal("assign bad type", span)),
                    };
                    let st = struct_llvm_type(oty, emitter.type_registry(), span)?;
                    let fi = emitter.type_registry().field_index(skey, *field).ok_or_else(|| CodegenError::internal("field missing", span))?;
                    let b = emitter.builder_mut();
                    // v12 Fix: build_struct_gep is safe.
                    let fp = b.build_struct_gep(st, ptr, fi, "fptr");
                    crate::codegen::value_utils::store_value_to_ptr(b, fp, v, emitter.type_registry().str_type(), span)?;
                    Ok(EmitResult::Value(CodegenValue::Unit))
                }
                _ => Err(CodegenError::internal("assign non-struct", span)),
            }
        }
        _ => Err(CodegenError::unimplemented_feature("assign target", span)),
    }
}
```

```rust
pub fn codegen_struct_literal<'ctx>(emitter: &mut FnEmitter<'_, 'ctx>, name: Symbol, fields: &[(Symbol, HirExpr)]) -> Result<EmitResult<'ctx>, CodegenError> {
    let span = emitter.current_span();
    let struct_type = emitter.type_registry().get_struct(name).ok_or_else(|| CodegenError::type_not_registered(HirType::Named(name), span))?;
    let size = emitter.type_registry().struct_size(name).ok_or_else(|| CodegenError::internal("no size", span))?;

    let raw_ptr = if size == 0 { 
        emitter.type_registry().zst_sentinel(span)? // v12 Fix: Handles Result
    } else {
        let sarg = emitter.llvm_types().i64_type.const_int(size, false);
        let alloc = emitter.fn_registry().get(emitter.known().glyim_alloc).ok_or_else(|| CodegenError::shim_not_declared("__glyim_alloc", span))?;
        // v12 Fix: Handles Result instead of unwrap
        emitter.builder_mut().build_call(alloc, &[sarg.into()], "salloc")
            .try_as_basic_value().left()
            .ok_or_else(|| CodegenError::internal("__glyim_alloc returned void", span))?
            .into_pointer_value()
    };

    for (fname, fexpr) in fields {
        let fidx = emitter.type_registry().field_index(name, *fname).ok_or_else(|| CodegenError::internal("field missing", span))?;
        let fty = emitter.type_registry().struct_field_hir_type(name, fidx).ok_or_else(|| CodegenError::internal("field type missing", span))?;
        if matches!(fty, HirType::Unit) { continue; } // Skip Unit fields

        // v12 Fix: build_struct_gep is safe.

        crate::codegen::value_utils::store_value_to_ptr(b, fptr, fval, emitter.type_registry().str_type(), span)?;
    }
    Ok(EmitResult::Value(CodegenValue::StructRef(raw_ptr)))
}
```

```rust
//! String literal emission and fat pointer construction.
//! FNV-1a for deterministic global names. Null terminator included.

use glyim_span::Span;
use crate::error::{CodegenError, llvm_err, DEFAULT_ADDR_SPACE};
use crate::types::fnv1a_str;
use crate::types::FNV_OFFSET;
use crate::value::{CodegenValue, EmitResult};
use crate::codegen::emitter::FnEmitter;

pub fn codegen_string_literal<'ctx>(
    emitter: &mut FnEmitter<'_, 'ctx>,
    string_data: &str,
) -> Result<EmitResult<'ctx>, CodegenError> {
    let str_type = emitter.type_registry().str_type();
    let span = emitter.current_span();

    let hash = fnv1a_str(FNV_OFFSET, string_data);
    let global_name = format!("__str_lit_{:016x}", hash);

    let data_ptr = if let Some(cached) = emitter.get_cached_string(&global_name) {
        cached
    } else if let Some(existing) = emitter.module().get_global(&global_name) {
        let ptr = existing.as_pointer_value();
        emitter.cache_string(global_name.clone(), ptr);
        ptr
    } else {
        let bytes = string_data.as_bytes();
        let i8_type = emitter.context().i8_type();

        let array_type = i8_type.array_type((bytes.len() + 1) as u32);
        let global = emitter.module().add_global(array_type, Some(DEFAULT_ADDR_SPACE), &global_name);

        let mut const_bytes: Vec<inkwell::values::IntValue<'ctx>> = bytes.iter()
            .map(|&b| i8_type.const_int(b as u64, false))
            .collect();
        const_bytes.push(i8_type.const_int(0, false)); 

        let const_str = i8_type.const_array(&const_bytes);
        global.set_initializer(&const_str);
        global.set_constant(true);
        global.set_linkage(inkwell::module::Linkage::Private);
        global.set_unnamed_address(inkwell::module::UnnamedAddress::Global);

        let ptr = global.as_pointer_value();
        emitter.cache_string(global_name.clone(), ptr);
        ptr
    };

    let builder = emitter.builder_mut();
    let alloca = builder.build_alloca(str_type, "str_lit");

    // v12 Fix: build_struct_gep is safe.
    let data_field_ptr = builder.build_struct_gep(str_type, alloca, 0, "str_data_ptr");
    builder.build_store(data_field_ptr, data_ptr)
        .map_err(|e| llvm_err("build_store", e, span))?;

    // v12 Fix: build_struct_gep is safe.
    let len_field_ptr = builder.build_struct_gep(str_type, alloca, 1, "str_len_ptr");
    let len_val = emitter.llvm_types().i64_type.const_int(string_data.len() as u64, false);
    builder.build_store(len_field_ptr, len_val)
        .map_err(|e| llvm_err("build_store", e, span))?;

    Ok(EmitResult::Value(CodegenValue::FatPtr(alloca)))
}
```

```rust
//! Top-level compilation pipeline.

use std::collections::HashMap;
use inkwell::context::Context;
use glyim_hir::{HirFn, HirType};
use glyim_intern::Symbol;
use glyim_span::Span;
use crate::error::CodegenError;
use crate::types::registry::{TypeRegistry, StructDef, EnumDef};
use crate::types::LlvmTypes;
use crate::codegen::input::CodegenInput;
use crate::codegen::fn_registry::FnRegistry;
use crate::codegen::emit_deps::EmitDeps;
use crate::codegen::string_mgr::StringLiteralManager;
use crate::codegen::function;
use crate::discovery;
use crate::method_names;
use crate::shims;

pub struct TypeDefinitions { pub struct_defs: HashMap<Symbol, StructDef>, pub enum_defs: HashMap<Symbol, EnumDef> }
impl TypeDefinitions { pub fn new(s: HashMap<Symbol, StructDef>, e: HashMap<Symbol, EnumDef>) -> Self { Self { struct_defs: s, enum_defs: e } } }

pub struct CodegenSession<'ctx> { pub module: inkwell::module::Module<'ctx>, pub type_registry: TypeRegistry<'ctx>, pub fn_registry: FnRegistry<'ctx>, pub string_mgr: StringLiteralManager<'ctx> }

pub fn compile(context: &'ctx Context, mut input: CodegenInput, functions: &[HirFn], type_defs: &TypeDefinitions) -> Result<CodegenSession<'ctx>, CodegenError> {
    crate::codegen::input::validate_codegen_input(&input)?;
    let module = context.create_module(&input.config.module_name);
    crate::codegen::function::set_target_layout(&module, &input.config);

    let mut type_registry = TypeRegistry::new(context, &mut input.interner);
    // v12 Fix: Initialize ZST sentinel. Returns Result.
    type_registry.initialize_zst_sentinel(&module, Span::default())?;

    let mut discovered = discovery::DiscoveredTypes::new();
    discovered.discover_from_expr_types(&input.expr_types);
    for f in functions { if let Some(r) = &f.ret { discovered.discover_from_type(r); } for (_, p) in &f.params { discovered.discover_from_type(p); } }

    for tt in &discovered.tuple_types { if let HirType::Tuple(f) = tt { type_registry.register_tuple(f, context, &mut input.interner, Span::default())?; } }
    for (n, d) in &type_defs.struct_defs { type_registry.register_struct(*n, &d.field_names, &d.field_types, context, &input.interner, Span::default())?; }
    for (n, d) in &type_defs.enum_defs { type_registry.register_enum(*n, &d.variants, context, &input.interner, Span::default())?; }

    let mut fn_registry = FnRegistry::new();
    for f in functions { crate::codegen::function::declare_function(f, &module, &type_registry, &mut fn_registry, context, &input.interner)?; }

    let mut shim_ctx = shims::ShimEmitCtx { context, module: &module, config: &input.config, type_registry: &type_registry, fn_registry: &mut fn_registry, interner: &mut input.interner };
    shims::ShimRegistry::for_aot().emit(&mut shim_ctx)?;

    input.method_names = method_names::compute_method_names_from_hir(functions, &input.expr_types, &type_registry, &mut input.interner)?;

    let string_mgr = StringLiteralManager::new();
    let llvm_types = LlvmTypes::new(context, type_registry.str_type());
    let deps = EmitDeps::new(&type_registry, &fn_registry, context, &module, &input.interner, &input.known, &input.expr_types, &input.call_type_args, &input.method_names, &input.metadata, &input.config, &llvm_types, &string_mgr);

    for f in functions {
        let fv = fn_registry.get(f.name).ok_or_else(|| CodegenError::internal(format!("fn {:?} missing", f.name), f.span))?;
        crate::codegen::function::emit_function_body(f, fv, &deps)?;
    }

    if let Err(e) = module.verify() { return Err(CodegenError::llvm_verification_failed(module.get_name().to_string_lossy().to_string(), e.to_string(), Span::default())); }
    Ok(CodegenSession { module, type_registry, fn_registry, string_mgr })
}
```

```rust
//! Type discovery pass (Phase 1).

use std::collections::HashSet;
use glyim_hir::HirType;
use crate::types::TypedExpr;

pub struct DiscoveredTypes { pub tuple_types: HashSet<HirType> }
impl DiscoveredTypes {
    pub fn new() -> Self { Self { tuple_types: HashSet::new() } }
    pub fn discover_from_expr_types(&mut self, ets: &std::collections::HashMap<glyim_hir::ExprId, TypedExpr>) { for te in ets.values() { self.discover_from_type(&te.ty); } }
    pub fn discover_from_type(&mut self, ty: &HirType) {
        match ty {
            HirType::Tuple(f) => { self.tuple_types.insert(ty.clone()); for t in f { self.discover_from_type(t); } }
            HirType::Option(i) => self.discover_from_type(i),
            HirType::Result(o, e) => { self.discover_from_type(o); self.discover_from_type(e); }
            HirType::RawPtr(i) => self.discover_from_type(i),
            HirType::Func(p, r) => { for t in p { self.discover_from_type(t); } self.discover_from_type(r); }
            HirType::Generic(_, a) => { for t in a { self.discover_from_type(t); } }
            _ => {}
        }
    }
}
```

```rust
//! Method name computation with __glyim_meth_ prefix.

use std::collections::HashMap;
use glyim_hir::{HirExpr, HirFn, ExprId, HirType};
use glyim_intern::{Interner, Symbol};
use glyim_span::Span;
use crate::codegen::input::MethodNameMap;
use crate::error::CodegenError;
use crate::types::registry::TypeRegistry;

pub fn compute_method_names_from_hir(fns: &[HirFn], ets: &HashMap<ExprId, crate::types::TypedExpr>, reg: &TypeRegistry<'_>, i: &mut Interner) -> Result<MethodNameMap, CodegenError> {
    let mut map = MethodNameMap::new();
    for f in fns { walk(&f.body, ets, reg, i, &mut map)?; }
    Ok(map)
}

fn walk(e: &HirExpr, ets: &HashMap<ExprId, crate::types::TypedExpr>, reg: &TypeRegistry<'_>, i: &mut Interner, m: &mut MethodNameMap) -> Result<(), CodegenError> {
    match e {
        HirExpr::MethodCall { receiver, method, args, .. } => {
            walk(receiver.as_ref(), ets, reg, i, m)?;
            for a in args { walk(a, ets, reg, i, m)?; }
            if let Some(te) = ets.get(&receiver.get_id()) {
                if let Ok(ts) = resolve(&te.ty, reg, i) {
                    let key = (ts, *method);
                    if !m.contains_key(&key) { m.insert(key, i.intern(&format!("__glyim_meth_{}__{}", i.resolve(ts), i.resolve(*method)))); }
                }
            }
        }
        HirExpr::Binary { lhs, rhs, .. } => { walk(lhs.as_ref(), ets, reg, i, m)?; walk(rhs.as_ref(), ets, reg, i, m)?; }
        HirExpr::Unary { operand, .. } => { walk(operand.as_ref(), ets, reg, i, m)?; }
        HirExpr::Call { args, .. } => { for a in args { walk(a, ets, reg, i, m)?; } }
        HirExpr::If { condition, then_branch, else_branch, .. } => { walk(condition.as_ref(), ets, reg, i, m)?; walk(then_branch.as_ref(), ets, reg, i, m)?; if let Some(e) = else_branch { walk(e.as_ref(), ets, reg, i, m)?; } }
        HirExpr::While { condition, body, .. } => { walk(condition.as_ref(), ets, reg, i, m)?; walk(body.as_ref(), ets, reg, i, m)?; }
        HirExpr::Match { scrutinee, arms, .. } => { walk(scrutinee.as_ref(), ets, reg, i, m)?; for a in arms { walk(&a.body, ets, reg, i, m)?; } }
        HirExpr::Block { stmts, expr, .. } => { for s in stmts { walk(s, ets, reg, i, m)?; } if let Some(e) = expr { walk(e.as_ref(), ets, reg, i, m)?; } }
        HirExpr::Let { value, .. } => { walk(value.as_ref(), ets, reg, i, m)?; }
        HirExpr::Assign { target, value, .. } => { walk(target.as_ref(), ets, reg, i, m)?; walk(value.as_ref(), ets, reg, i, m)?; }
        HirExpr::FieldAccess { object, .. } => { walk(object.as_ref(), ets, reg, i, m)?; }
        HirExpr::Cast { value, .. } => { walk(value.as_ref(), ets, reg, i, m)?; }
        HirExpr::Return { value, .. } => { if let Some(v) = value { walk(v.as_ref(), ets, reg, i, m)?; } }
        HirExpr::StructLit { fields, .. } => { for (_, f) in fields { walk(f, ets, reg, i, m)?; } }
        HirExpr::EnumVariant { fields, .. } => { for f in fields { walk(f, ets, reg, i, m)?; } }
        HirExpr::TupleLit { elements, .. } => { for e in elements { walk(e, ets, reg, i, m)?; } }
        _ => {}
    }
    Ok(())
}

fn resolve(ty: &HirType, reg: &TypeRegistry<'_>, i: &Interner) -> Result<Symbol, CodegenError> {
    match ty {
        HirType::Named(s) => Ok(*s),
        HirType::Tuple(_) => { let h = crate::types::hir_type_hash(ty, Span::default(), i)?; reg.resolve_tuple_symbol_by_hash(ty, h, Span::default(), i) }
        _ => Err(CodegenError::internal(format!("bad method recv: {:?}", ty), Span::default())),
    }
}
```
