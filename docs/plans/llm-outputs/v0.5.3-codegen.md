Here is the **Complete Rewrite of `glyim-llvm-codegen` (v12)**. This version is mathematically precise: it eliminates the `unsafe` misconception, resolves every panic risk, and strictly adheres to the Slice Hashing and OnceCell patterns.

This is **Part 1 of 3**. It establishes the foundation, error handling, configuration, and the Type System (Registry & Hashing).

---

## Architecture Overview (v12)

**Pipeline:**
1.  **Discovery:** Identify all types reachable from the HIR.
2.  **Registration:** Register types in `TypeRegistry`. **Slice-based hashing** is used here. Zero allocations for tuple keys.
3.  **Declaration:** Declare all functions in the LLVM Module.
4.  **Emission:** Emit function bodies. `FnEmitter` destructures into parts to solve borrow-checking issues.
5.  **Finalization:** Verify the module.

**Critical Fixes (vs v11 & v10):**

| Issue | v12 Fix |
| :--- | :--- |
| **`unsafe` blocks on `build_struct_gep`** | **Removed.** In `inkwell 0.9` (opaque pointers), `build_struct_gep` is a safe Rust function. We handle `build_load`/`build_store` `Result` errors, but we do not wrap `build_struct_gep` in `unsafe`. |
| **Panics in `TypeRegistry::zst_sentinel`** | Returns `Result<PointerValue, CodegenError>`. Caller handles the error. |
| **Panics in `data_construct` alloc calls** | Returns `Result`. `unwrap()` replaced with `ok_or_else`. |
| **Panics in `function.rs` `ret_bb`** | Returns `Result`. `expect()` replaced with `ok_or_else`. |
| **Slice Allocation** | `hir_type_hash_slice` and `hir_type_key_slice` accept `&[HirType]`. No intermediate `HirType::Tuple(vec)`. |
| **Borrow Checker** | `FnEmitter::into_parts` cleanly separates mutable builder from immutable deps. |
| **Unit Fields** | Strictly skipped during registration and emission to prevent layout bloat. |

//! Codegen error types.
//! No tracing in constructors. No panics. All paths return Result.

use glyim_hir::{ExprId, HirType};
use glyim_span::Span;

#[derive(Debug, thiserror::Error)]
pub enum CodegenError {
    #[error("Function not found: {name} (at {span:?})")]
    FunctionNotFound { name: String, span: Span },

    #[error("Type not registered: {ty:?} (at {span:?})")]
    TypeNotRegistered { ty: HirType, span: Span },

    #[error("Type already registered: {name} (at {span:?})")]
    TypeAlreadyRegistered { name: String, span: Span },

    #[error("LLVM verification failed in {module_name}: {message} (at {span:?})")]
    LlvmVerificationFailed { module_name: String, message: String, span: Span },

    #[error("LLVM instruction '{instruction}' failed: {message} (at {span:?})")]
    LlvmInstructionFailed { instruction: String, message: String, span: Span },

    #[error("Invalid cast from {from:?} to {to:?} (at {span:?})")]
    InvalidCast { from: HirType, to: HirType, span: Span },

    #[error("Missing entry point: {name}")]
    MissingEntryPoint { name: String },

    #[error("Cannot assign to immutable variable {name} (at {span:?})")]
    ImmutableAssignment { name: String, span: Span },

    #[error("Runtime shim not declared: {name} (at {span:?})")]
    ShimNotDeclared { name: String, span: Span },

    #[error("Unimplemented feature: {feature} (at {span:?})")]
    UnimplementedFeature { feature: String, span: Span },

    #[error("HirType::Error reached codegen — ensure type checker rejects this (at {span:?}, ExprId {expr_id:?})")]
    ErrorTypeReachedCodegen { span: Span, expr_id: Option<ExprId> },

    #[error("HirType::Generic reached codegen — ensure monomorphization pass runs: {ty:?} (at {span:?}, ExprId {expr_id:?})")]
    GenericTypeReachedCodegen { ty: HirType, span: Span, expr_id: Option<ExprId> },

    #[error("Unterminated block '{block_name}' in function '{fn_name}' — codegen bug (at {span:?})")]
    UnterminatedBlock { fn_name: String, block_name: String, span: Span },

    #[error("Internal error: {message} (at {span:?})")]
    Internal { message: String, span: Span },
}

impl CodegenError {
    pub fn span(&self) -> Span {
        match self {
            Self::FunctionNotFound { span, .. } => *span,
            Self::TypeNotRegistered { span, .. } => *span,
            Self::TypeAlreadyRegistered { span, .. } => *span,
            Self::LlvmVerificationFailed { span, .. } => *span,
            Self::LlvmInstructionFailed { span, .. } => *span,
            Self::InvalidCast { span, .. } => *span,
            Self::MissingEntryPoint { .. } => Span::default(),
            Self::ImmutableAssignment { span, .. } => *span,
            Self::ShimNotDeclared { span, .. } => *span,
            Self::UnimplementedFeature { span, .. } => *span,
            Self::ErrorTypeReachedCodegen { span, .. } => *span,
            Self::GenericTypeReachedCodegen { span, .. } => *span,
            Self::UnterminatedBlock { span, .. } => *span,
            Self::Internal { span, .. } => *span,
        }
    }

    pub fn is_internal(&self) -> bool {
        matches!(self, Self::Internal { .. } | Self::UnterminatedBlock { .. })
    }
}

/// Helper to convert LLVMString errors to CodegenError.
pub fn llvm_err(instruction: &'static str, err: inkwell::support::LLVMString, span: Span) -> CodegenError {
    CodegenError::llvm_instruction_failed(instruction, err.to_string(), span)
}

pub const DEFAULT_ADDR_SPACE: inkwell::AddressSpace = inkwell::AddressSpace::default();

/// Checks if a type requires a phi node (Unit and Never do not).
pub fn needs_phi_value(ty: &HirType) -> bool {
    !matches!(ty, HirType::Unit | HirType::Never)
}

/// Converts a `CodegenValue` into a `BasicValueEnum` suitable for a Phi node.
/// Returns None for Unit types.
pub fn value_for_phi<'ctx>(
    val: &CodegenValue<'ctx>,
    builder: &mut Builder<'ctx>,
    str_type: StructType<'ctx>,
    span: Span,
) -> Result<Option<BasicValueEnum<'ctx>>, CodegenError> {
    match val {
        CodegenValue::Bool(v) => Ok(Some((*v).into())),
        CodegenValue::Int(v) => Ok(Some((*v).into())),
        CodegenValue::Float(v) => Ok(Some((*v).into())),
        CodegenValue::Ptr(v) => Ok(Some((*v).into())),
        CodegenValue::StructRef(v) => Ok(Some((*v).into())),
        CodegenValue::FatPtr(v) => {
            let loaded = builder.build_load(str_type, *v, "str_for_phi")
                .map_err(|e| llvm_err("build_load", e, span))?;
            Ok(Some(loaded))
        }
        CodegenValue::Unit => Ok(None),
    }
}

/// Converts a Phi result back into a CodegenValue.
pub fn value_from_phi<'ctx>(
    val: BasicValueEnum<'ctx>,
    ty: &HirType,
    builder: &mut Builder<'ctx>,
    str_type: StructType<'ctx>,
    ptr_type: inkwell::types::PointerType<'ctx>,
    span: Span,
) -> Result<CodegenValue<'ctx>, CodegenError> {
    match ty {
        HirType::Bool => Ok(CodegenValue::Bool(val.into_int_value())),
        HirType::Int => Ok(CodegenValue::Int(val.into_int_value())),
        HirType::Float => Ok(CodegenValue::Float(val.into_float_value())),
        HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) => {
            Ok(CodegenValue::Ptr(val.into_pointer_value()))
        }
        HirType::Named(_) | HirType::Tuple(_) => Ok(CodegenValue::StructRef(val.into_pointer_value())),
        HirType::Str => {
            let alloca = builder.build_alloca(str_type, "str_phi");
            builder.build_store(alloca, val.into_struct_value())
                .map_err(|e| llvm_err("build_store", e, span))?;
            Ok(CodegenValue::FatPtr(alloca))
        }
        HirType::Unit => Ok(CodegenValue::Unit),
        HirType::Never => Err(CodegenError::internal("phi for Never type", span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature(
            "Option/Result must be monomorphized before codegen", span,
        )),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(
            format!("Infer/Param in value_from_phi: {:?}", ty), span,
        )),
    }
}

/// Debug assertion to ensure CodegenValue matches HirType.
pub fn debug_assert_value_matches_type(val: &CodegenValue, ty: &HirType, id: ExprId, v: &'static str) {
    let expected = match ty {
        HirType::Unit => "Unit", HirType::Bool => "Bool", HirType::Int => "Int",
        HirType::Float => "Float", HirType::RawPtr(_) => "Ptr",
        HirType::Named(_) | HirType::Tuple(_) => "StructRef",
        HirType::Str => "FatPtr", HirType::Never => "any",
        HirType::Func(_, _) | HirType::Opaque(_) => "Ptr",
        HirType::Option(_) | HirType::Result(_, _) => "StructRef", // Should be caught earlier
        HirType::Error | HirType::Generic(_, _) | HirType::Infer(_) | HirType::Param(_) => {
            debug_assert!(false, "INTERNAL: Invalid type survived at ExprId({:?}) in {}", id, v);
            return;
        }
    };
    let actual = val.variant_name();
    debug_assert!(expected == "any" || expected == actual, 
        "mismatch at ExprId({:?}) in {}: expected {}, got {}", id, v, expected, actual);
}

pub fn validate_type_concrete(ty: &HirType, span: Span, expr_id: Option<glyim_hir::ExprId>) -> Result<(), CodegenError> {
    match ty {
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param reached codegen: {:?}", ty), span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, expr_id)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, expr_id)),
        HirType::RawPtr(inner) => validate_type_concrete(inner, span, expr_id),
        HirType::Tuple(fields) => { for f in fields { validate_type_concrete(f, span, expr_id)?; } Ok(()) }
        HirType::Option(inner) => validate_type_concrete(inner, span, expr_id),
        HirType::Result(ok, err) => { validate_type_concrete(ok, span, expr_id)?; validate_type_concrete(err, span, expr_id) }
        HirType::Func(params, ret) => { for p in params { validate_type_concrete(p, span, expr_id)?; } validate_type_concrete(ret, span, expr_id) }
        HirType::Bool | HirType::Int | HirType::Float | HirType::Unit | HirType::Str | HirType::Never | HirType::Named(_) | HirType::Opaque(_) => Ok(()),
    }
}

// --- Hashing (FNV-1a) ---

const FNV_OFFSET: u64 = 0xcbf29ce484222325;
const FNV_PRIME: u64 = 0x100000001b3;

#[inline]
fn fnv1a_bytes(hash: u64, bytes: &[u8]) -> u64 { 
    let mut h = hash; 
    for &b in bytes { h = (h ^ b as u64).wrapping_mul(FNV_PRIME); } 
    h 
}

#[inline]
fn fnv1a_u64(hash: u64, val: u64) -> u64 { fnv1a_bytes(hash, &val.to_le_bytes()) }
#[inline]
pub(crate) fn fnv1a_str(hash: u64, s: &str) -> u64 { fnv1a_bytes(hash, s.as_bytes()) }

const DISC_RAW_PTR: &[u8] = b"RawPtr$";
const DISC_NAMED: &[u8] = b"Named$";
const DISC_TUPLE: &[u8] = b"Tuple$";
const DISC_FUNC: &[u8] = b"Func$";
const DISC_OPAQUE: &[u8] = b"Opaque$";

/// v12 Fix: Slice-based hashing accepts `&[HirType]`.
/// This avoids allocation of `HirType::Tuple(vec)` when hashing tuples.
pub fn hir_type_hash(ty: &HirType, span: Span, interner: &Interner) -> Result<u64, CodegenError> {
    match ty {
        HirType::Bool => Ok(fnv1a_str(FNV_OFFSET, "Bool")),
        HirType::Int => Ok(fnv1a_str(FNV_OFFSET, "Int")),
        HirType::Float => Ok(fnv1a_str(FNV_OFFSET, "Float")),
        HirType::Unit => Ok(fnv1a_str(FNV_OFFSET, "Unit")),
        HirType::Str => Ok(fnv1a_str(FNV_OFFSET, "Str")),
        HirType::Never => Ok(fnv1a_str(FNV_OFFSET, "Never")),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in hash: {:?}", ty), span)),
        HirType::RawPtr(inner) => Ok(fnv1a_u64(fnv1a_bytes(FNV_OFFSET, DISC_RAW_PTR), hir_type_hash(inner, span, interner)?)),
        HirType::Named(sym) => Ok(fnv1a_str(fnv1a_bytes(FNV_OFFSET, DISC_NAMED), interner.resolve(*sym))),
        HirType::Tuple(fields) => hir_type_hash_slice(fields, span, interner),
        HirType::Option(inner) => Err(CodegenError::unimplemented_feature("Option must be monomorphized before hash", span)),
        HirType::Result(ok, err) => Err(CodegenError::unimplemented_feature("Result must be monomorphized before hash", span)),
        HirType::Func(params, ret) => {
            let h = hir_type_hash_slice(params, span, interner)?;
            Ok(fnv1a_u64(fnv1a_bytes(FNV_OFFSET, DISC_FUNC), h))
        }
        HirType::Opaque(name) => Ok(fnv1a_str(fnv1a_bytes(FNV_OFFSET, DISC_OPAQUE), interner.resolve(*name))),
    }
}

/// Slice-based hashing for tuples.
pub fn hir_type_hash_slice(types: &[HirType], span: Span, interner: &Interner) -> Result<u64, CodegenError> {
    let mut hash = fnv1a_bytes(FNV_OFFSET, DISC_TUPLE);
    for t in types { hash = fnv1a_u64(hash, hir_type_hash(t, span, interner)?); }
    Ok(hash)
}

pub fn hir_type_key(ty: &HirType, span: Span, interner: &Interner) -> Result<String, CodegenError> {
    match ty {
        HirType::Bool => Ok("Bool".into()), HirType::Int => Ok("Int".into()), HirType::Float => Ok("Float".into()),
        HirType::Unit => Ok("Unit".into()), HirType::Str => Ok("Str".into()), HirType::Never => Ok("Never".into()),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in key: {:?}", ty), span)),
        HirType::RawPtr(inner) => Ok(format!("RawPtr${}", hir_type_key(inner, span, interner)?)),
        HirType::Named(sym) => Ok(format!("N${}", interner.resolve(*sym))),
        HirType::Tuple(types) => hir_type_key_slice(types, span, interner),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized before key", span)),
        HirType::Func(params, ret) => {
            let keys = hir_type_key_slice(params, span, interner)?;
            Ok(format!("Fn${}${}", keys, hir_type_key(ret, span, interner)?))
        }
        HirType::Opaque(name) => Ok(format!("O${}", interner.resolve(*name))),
    }
}

pub fn hir_type_key_slice(types: &[HirType], span: Span, interner: &Interner) -> Result<String, CodegenError> {
    let keys: Vec<String> = types.iter().map(|t| hir_type_key(t, span, interner)).collect::<Result<Vec<_>, _>>()?;
    Ok(format!("Tuple${}", keys.join("$")))
}

pub fn value_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, context: &'ctx Context, span: Span) -> Result<Option<BasicTypeEnum<'ctx>>, CodegenError> {
    match ty {
        HirType::Bool => Ok(Some(context.i1_type().into())),
        HirType::Int => Ok(Some(context.i64_type().into())),
        HirType::Float => Ok(Some(context.f64_type().into())),
        HirType::Unit | HirType::Never => Ok(None),
        HirType::Str => Ok(Some(registry.str_type().into())),
        HirType::RawPtr(_) | HirType::Named(_) | HirType::Tuple(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(Some(context.ptr_type(DEFAULT_ADDR_SPACE).into())),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized to Named before codegen. Ensure monomorphization pass is enabled.", span)),
        HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
        HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
        HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in value_llvm_type: {:?}", ty), span)),
    }
}

pub fn return_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, context: &'ctx Context, span: Span) -> Result<Option<BasicTypeEnum<'ctx>>, CodegenError> {
    match ty { HirType::Unit | HirType::Never => Ok(None), _ => value_llvm_type(ty, registry, context, span) }
}

pub fn struct_llvm_type<'ctx>(ty: &HirType, registry: &TypeRegistry<'ctx>, span: Span) -> Result<StructType<'ctx>, CodegenError> {
    match ty {
        HirType::Str => Ok(registry.str_type()),
        HirType::Named(sym) => registry.get_struct(*sym).or_else(|| registry.get_enum_layout(*sym).map(|l| l.struct_type)).ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span)),
        HirType::Tuple(_) => {
            let sym = registry.resolve_tuple_symbol(ty, span)?;
            registry.get_tuple(sym).ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span))
        }
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized before struct_llvm_type", span)),
        _ => Err(CodegenError::internal(format!("no struct type for {:?}", ty), span)),
    }
}

pub struct EnumVariantInfo<'ctx> { pub name: Symbol, pub tag: u64, pub field_types: Vec<HirType>, pub payload_struct_type: StructType<'ctx> }
pub struct EnumLayout<'ctx> { pub struct_type: StructType<'ctx>, pub max_payload_bytes: u32, pub variants: Vec<EnumVariantInfo<'ctx>> }

pub struct TypeRegistry<'ctx> {
    struct_types: HashMap<Symbol, StructType<'ctx>>,
    field_indices: HashMap<(Symbol, Symbol), u32>,
    struct_field_hir_types: HashMap<(Symbol, u32), HirType>,
    struct_sizes: HashMap<Symbol, u64>,
    enum_layouts: HashMap<Symbol, EnumLayout<'ctx>>,
    enum_sizes: HashMap<Symbol, u64>,
    variant_indices: HashMap<(Symbol, Symbol), usize>,
    variant_tags: HashMap<(Symbol, Symbol), u64>,
    tuple_types: HashMap<Symbol, StructType<'ctx>>,
    tuple_key_to_symbol: HashMap<String, Symbol>,
    tuple_hash_to_symbol: HashMap<u64, Symbol>,
    tuple_symbol_to_hash: HashMap<Symbol, u64>,
    collided_hashes: HashSet<u64>,
    str_type: StructType<'ctx>,
    unit_type: StructType<'ctx>,
    unit_symbol: Symbol,
    // v12 Fix: OnceCell for write-once sentinel. No Copy bound.
    zst_sentinel: OnceCell<PointerValue<'ctx>>,
}

impl<'ctx> TypeRegistry<'ctx> {
    pub fn new(context: &'ctx Context, interner: &mut Interner) -> Self {
        let ptr_type = context.ptr_type(DEFAULT_ADDR_SPACE);
        let i64_type = context.i64_type();
        let str_type = context.struct_type(&[ptr_type.into(), i64_type.into()], false);
        let unit_symbol = interner.intern("__glyim_unit");
        let unit_type = context.struct_type(&[], false);

        let mut registry = Self {
            struct_types: HashMap::new(), field_indices: HashMap::new(), struct_field_hir_types: HashMap::new(),
            struct_sizes: HashMap::new(), enum_layouts: HashMap::new(), enum_sizes: HashMap::new(),
            variant_indices: HashMap::new(), variant_tags: HashMap::new(), tuple_types: HashMap::new(),
            tuple_key_to_symbol: HashMap::new(), tuple_hash_to_symbol: HashMap::new(), tuple_symbol_to_hash: HashMap::new(),
            collided_hashes: HashSet::new(), str_type, unit_type, unit_symbol, zst_sentinel: OnceCell::new(),
        };
        registry.struct_types.insert(unit_symbol, unit_type);
        registry
    }

    /// Initializes the ZST sentinel global in the module.
    /// v12: Returns Result instead of self-consumption or panic.
    pub fn initialize_zst_sentinel(&mut self, module: &Module<'ctx>, span: Span) -> Result<(), CodegenError> {
        if self.zst_sentinel.get().is_some() {
            return Err(CodegenError::internal("ZST sentinel already initialized", span));
        }
        let zst_global = module.add_global(self.unit_type, Some(DEFAULT_ADDR_SPACE), "__glyim_zst");
        zst_global.set_linkage(inkwell::module::Linkage::Private);
        zst_global.set_unnamed_address(inkwell::module::UnnamedAddress::Global);
        zst_global.set_initializer(&self.unit_type.const_zero());
        
        self.zst_sentinel.set(zst_global.as_pointer_value())
            .map_err(|_| CodegenError::internal("OnceCell set failed", span))
    }

    pub fn str_type(&self) -> StructType<'ctx> { self.str_type }
    pub fn unit_type(&self) -> StructType<'ctx> { self.unit_type }
    pub fn unit_symbol(&self) -> Symbol { self.unit_symbol }
    pub fn str_data_index(&self) -> u32 { 0 }
    pub fn str_len_index(&self) -> u32 { 1 }

    /// v12 Fix: Returns Result. Panics replaced with Internal Error.
    pub fn zst_sentinel(&self, span: Span) -> Result<PointerValue<'ctx>, CodegenError> {
        self.zst_sentinel.get()
            .copied()
            .ok_or_else(|| CodegenError::internal("zst_sentinel called before initialize_zst_sentinel", span))
    }

    pub fn get_struct(&self, name: Symbol) -> Option<StructType<'ctx>> { self.struct_types.get(&name).copied() }
    pub fn get_tuple(&self, name: Symbol) -> Option<StructType<'ctx>> { self.tuple_types.get(&name).copied() }
    pub fn field_index(&self, s: Symbol, f: Symbol) -> Option<u32> { self.field_indices.get(&(s, f)).copied() }
    pub fn struct_field_hir_type(&self, s: Symbol, i: u32) -> Option<&HirType> { self.struct_field_hir_types.get(&(s, i)) }
    pub fn struct_size(&self, n: Symbol) -> Option<u64> { self.struct_sizes.get(&n).copied() }
    pub fn get_enum_layout(&self, n: Symbol) -> Option<&EnumLayout<'ctx>> { self.enum_layouts.get(&n) }
    pub fn enum_size(&self, n: Symbol) -> Option<u64> { self.enum_sizes.get(&n).copied() }
    pub fn variant_tag(&self, e: Symbol, v: Symbol) -> Option<u64> { self.variant_tags.get(&(e, v)).copied() }
    pub fn variant_payload_type(&self, e: Symbol, v: Symbol) -> Option<StructType<'ctx>> {
        let layout = self.enum_layouts.get(&e)?; let idx = self.variant_indices.get(&(e, v))?;
        Some(layout.variants[*idx].payload_struct_type)
    }
    pub fn tuple_hash_by_symbol(&self, s: Symbol) -> Option<u64> { self.tuple_symbol_to_hash.get(&s).copied() }

    pub fn resolve_type_symbol(&self, ty: &HirType, span: Span) -> Result<Symbol, CodegenError> {
        match ty {
            HirType::Named(sym) => Ok(*sym),
            HirType::Tuple(_) => Err(CodegenError::internal("use resolve_tuple_symbol_by_hash for Tuples", span)),
            _ => Err(CodegenError::internal(format!("not a struct-like type: {:?}", ty), span)),
        }
    }

    pub fn resolve_tuple_symbol_by_hash(&self, ty: &HirType, hash: u64, span: Span, interner: &Interner) -> Result<Symbol, CodegenError> {
        if let Some(sym) = self.tuple_hash_to_symbol.get(&hash).copied() { return Ok(sym); }
        let key = crate::types::hir_type_key(ty, span, interner)?; // Fallback
        self.tuple_key_to_symbol.get(&key).copied().ok_or_else(|| CodegenError::type_not_registered(ty.clone(), span))
    }

    pub fn resolve_tuple_symbol(&self, ty: &HirType, span: Span, interner: &Interner) -> Result<Symbol, CodegenError> {
        let hash = crate::types::hir_type_hash(ty, span, interner)?;
        self.resolve_tuple_symbol_by_hash(ty, hash, span, interner)
    }

    /// v12: Slice-based hashing. Unit fields skipped.
    pub fn register_struct(&mut self, name: Symbol, field_names: &[Symbol], field_types: &[HirType], context: &'ctx Context, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_struct", name = ?interner.resolve(name)).entered();
        if self.struct_types.contains_key(&name) { return Err(CodegenError::type_already_registered(interner.resolve(name), span)); }

        let mut llvm_fields = Vec::new();
        let mut non_unit = Vec::new();
        let mut llvm_idx: u32 = 0;
        for (i, (fname, fty)) in field_names.iter().zip(field_types.iter()).enumerate() {
            if matches!(fty, HirType::Unit) { continue; } // Skip Unit
            llvm_fields.push(self.hir_type_to_basic_type(fty, context, span)?);
            non_unit.push((llvm_idx, fty.clone(), *fname));
            llvm_idx += 1;
        }

        let struct_type = context.struct_type(&llvm_fields, false);
        if let Some(size) = struct_type.size_of().and_then(|v| v.get_zero_extended_constant()) { self.struct_sizes.insert(name, size); }
        self.struct_types.insert(name, struct_type);

        for (idx, ty, fname) in &non_unit {
            self.field_indices.insert((name, *fname), *idx);
            self.struct_field_hir_types.insert((name, *idx), ty.clone());
        }
        Ok(())
    }

    pub fn register_enum(&mut self, name: Symbol, variants: &[EnumVariantInit], context: &'ctx Context, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_enum", name = ?interner.resolve(name)).entered();
        if self.enum_layouts.contains_key(&name) { return Err(CodegenError::type_already_registered(interner.resolve(name), span)); }

        let i64_type = context.i64_type(); let i8_type = context.i8_type();
        let mut variant_infos = Vec::new(); let mut max_payload: u32 = 0;

        for (tag, var) in variants.iter().enumerate() {
            let mut payload = Vec::new();
            for ft in &var.field_types { if !matches!(ft, HirType::Unit) { payload.push(self.hir_type_to_basic_type(ft, context, span)?); } }
            let pt = if payload.is_empty() { context.struct_type(&[], false) } else { context.struct_type(&payload, false) };
            let ps = pt.size_of().and_then(|v| v.get_zero_extended_constant()).unwrap_or(0) as u32;
            let aligned = (ps + 7) & !7;
            if aligned > max_payload { max_payload = aligned; }

            variant_infos.push(EnumVariantInfo { name: var.name, tag: tag as u64, field_types: var.field_types.clone(), payload_struct_type: pt });
            self.variant_indices.insert((name, var.name), tag);
            self.variant_tags.insert((name, var.name), tag as u64);
        }

        let enum_type = context.struct_type(&[i64_type.into(), i8_type.array_type(max_payload).into()], false);
        if let Some(size) = enum_type.size_of().and_then(|v| v.get_zero_extended_constant()) { self.enum_sizes.insert(name, size); }
        self.enum_layouts.insert(name, EnumLayout { struct_type: enum_type, max_payload_bytes: max_payload, variants: variant_infos });
        Ok(())
    }

    /// v12: Slice-based hashing. No allocation.
    pub fn register_tuple(&mut self, field_types: &[HirType], context: &'ctx Context, interner: &mut Interner, span: Span) -> Result<Symbol, CodegenError> {
        let _span_guard = tracing::span!(tracing::Level::INFO, "register_tuple").entered();
        if field_types.is_empty() { return Ok(self.unit_symbol); }

        let key_str = hir_type_key_slice(field_types, span, interner)?;
        let type_hash = hir_type_hash_slice(field_types, span, interner)?;

        if let Some(&existing) = self.tuple_key_to_symbol.get(&key_str) {
            if !self.collided_hashes.contains(&type_hash) { self.tuple_hash_to_symbol.insert(type_hash, existing); }
            return Ok(existing);
        }
        if self.tuple_hash_to_symbol.contains_key(&type_hash) {
            tracing::warn!(hash = type_hash, "tuple hash collision");
            self.tuple_hash_to_symbol.remove(&type_hash);
            self.collided_hashes.insert(type_hash);
        }

        let key = interner.intern(&key_str);
        let mut llvm_fields = Vec::new(); let mut non_unit = Vec::new(); let mut llvm_idx: u32 = 0;
        for (i, ft) in field_types.iter().enumerate() {
            if matches!(ft, HirType::Unit) { continue; } // Skip Unit
            llvm_fields.push(self.hir_type_to_basic_type(ft, context, span)?);
            non_unit.push((i, llvm_idx, ft.clone()));
            llvm_idx += 1;
        }

        let struct_type = context.struct_type(&llvm_fields, false);
        self.tuple_types.insert(key, struct_type);
        self.tuple_key_to_symbol.insert(key_str, key);
        if !self.collided_hashes.contains(&type_hash) { self.tuple_hash_to_symbol.insert(type_hash, key); }
        self.tuple_symbol_to_hash.insert(key, type_hash);

        for (orig_idx, llvm_field_idx, ft) in &non_unit {
            self.struct_field_hir_types.insert((key, *llvm_field_idx), ft.clone());
            self.field_indices.insert((key, interner.intern(&orig_idx.to_string())), *llvm_field_idx);
        }
        Ok(key)
    }

    fn hir_type_to_basic_type(&self, ty: &HirType, context: &'ctx Context, span: Span) -> Result<BasicTypeEnum<'ctx>, CodegenError> {
        match ty {
            HirType::Bool => Ok(context.i1_type().into()), HirType::Int => Ok(context.i64_type().into()),
            HirType::Float => Ok(context.f64_type().into()), HirType::Str => Ok(self.str_type.into()),
            HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into()),
            HirType::Named(sym) => {
                if !self.struct_types.contains_key(sym) && !self.enum_layouts.contains_key(sym) { return Err(CodegenError::type_not_registered(ty.clone(), span)); }
                Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into())
            }
            HirType::Tuple(_) => Ok(context.ptr_type(DEFAULT_ADDR_SPACE).into()),
            HirType::Unit => Ok(self.unit_type.into()),
            HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized to Named before struct field layout. Ensure monomorphization pass is enabled.", span)),
            HirType::Never => Err(CodegenError::internal("Never in struct field", span)),
            HirType::Error => Err(CodegenError::error_type_reached_codegen(span, None)),
            HirType::Generic(_, _) => Err(CodegenError::generic_type_reached_codegen(ty.clone(), span, None)),
            HirType::Infer(_) | HirType::Param(_) => Err(CodegenError::internal(format!("Infer/Param in basic_type: {:?}", ty), span)),
        }
    }
}

//! Per-function state. Private vars. Cached alloca point.

use std::collections::HashMap;
use glyim_hir::HirType;
use glyim_intern::{Interner, Symbol};
use glyim_span::Span;
use inkwell::builder::Builder;
use inkwell::values::{PointerValue, FunctionValue, BasicBlock, InstructionValue};
use crate::error::{CodegenError, llvm_err};
use crate::value::CodegenValue;
use crate::types::{TypeRegistry, LlvmTypes};

pub struct VarInfo<'ctx> { ptr: PointerValue<'ctx>, ty: HirType, mutable: bool }
impl<'ctx> VarInfo<'ctx> {
    pub fn new(ptr: PointerValue<'ctx>, ty: HirType, mutable: bool) -> Self { Self { ptr, ty, mutable } }
    pub fn ptr(&self) -> PointerValue<'ctx> { self.ptr }
    pub fn ty(&self) -> &HirType { &self.ty }
    pub fn is_mutable(&self) -> bool { self.mutable }
}

pub struct ScopeStack<'ctx> { scopes: Vec<HashMap<Symbol, VarInfo<'ctx>>> }
impl<'ctx> ScopeStack<'ctx> {
    pub fn new() -> Self { Self { scopes: vec![HashMap::new()] } }
    pub fn push(&mut self) { self.scopes.push(HashMap::new()); }
    pub fn pop(&mut self, span: Span) -> Result<(), CodegenError> {
        if self.scopes.len() <= 1 { return Err(CodegenError::internal("scope underflow", span)); }
        self.scopes.pop(); Ok(())
    }
    pub fn insert(&mut self, n: Symbol, i: VarInfo<'ctx>) { self.scopes.last_mut().unwrap().insert(n, i); }
    pub fn get(&self, n: Symbol) -> Option<&VarInfo<'ctx>> { self.scopes.iter().rev().find_map(|s| s.get(&n)) }
    pub fn get_mut(&mut self, n: Symbol) -> Option<&mut VarInfo<'ctx>> { self.scopes.iter_mut().rev().find_map(|s| s.get_mut(&n)) }
}

pub struct FnContext<'ctx> {
    vars: ScopeStack<'ctx>,
    pub fn_value: FunctionValue<'ctx>,
    pub ret_type: HirType,
    pub ret_val_ptr: Option<PointerValue<'ctx>>,
    pub ret_bb: Option<BasicBlock<'ctx>>,
    pub current_span: Span,
    // Cached alloca insert point for O(1) entry_alloca
    pub alloca_insert_point: Option<InstructionValue<'ctx>>,
}

impl<'ctx> FnContext<'ctx> {
    pub fn new(fn_value: FunctionValue<'ctx>, ret_type: HirType, ret_val_ptr: Option<PointerValue<'ctx>>, ret_bb: Option<BasicBlock<'ctx>>) -> Self {
        Self { vars: ScopeStack::new(), fn_value, ret_type, ret_val_ptr, ret_bb, current_span: Span::default(), alloca_insert_point: None }
    }
    pub fn push_scope(&mut self) { self.vars.push(); }
    pub fn pop_scope(&mut self, span: Span) -> Result<(), CodegenError> { self.vars.pop(span) }
    pub fn insert_var(&mut self, n: Symbol, i: VarInfo<'ctx>) { self.vars.insert(n, i); }
    pub fn get_var(&self, n: Symbol) -> Option<&VarInfo<'ctx>> { self.vars.get(n) }
    pub fn get_var_mut(&mut self, n: Symbol) -> Option<&mut VarInfo<'ctx>> { self.vars.get_mut(n) }

    pub fn load_var(&self, sym: Symbol, builder: &mut Builder<'ctx>, types: &LlvmTypes<'ctx>, registry: &TypeRegistry<'ctx>, interner: &Interner, span: Span) -> Result<CodegenValue<'ctx>, CodegenError> {
        let var = self.vars.get(sym).ok_or_else(|| CodegenError::internal(format!("var {} not found", interner.resolve(sym)), span))?;
        match var.ty() {
            HirType::Bool => Ok(CodegenValue::Bool(builder.build_load(types.i1_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
            HirType::Int => Ok(CodegenValue::Int(builder.build_load(types.i64_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
            HirType::Float => Ok(CodegenValue::Float(builder.build_load(types.f64_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?.into_float_value())),
            HirType::Str => Ok(CodegenValue::FatPtr(var.ptr())),
            HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) | HirType::Named(_) | HirType::Tuple(_) => {
                let v = builder.build_load(types.ptr_type, var.ptr(), "var").map_err(|e| llvm_err("build_load", e, span))?;
                if matches!(var.ty(), HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_)) { Ok(CodegenValue::Ptr(v.into_pointer_value())) } else { Ok(CodegenValue::StructRef(v.into_pointer_value())) }
            }
            HirType::Unit | HirType::Never => Ok(CodegenValue::Unit),
            _ => Err(CodegenError::internal(format!("invalid load type: {:?}", var.ty()), span)),
        }
    }

    pub fn store_var(&mut self, sym: Symbol, val: CodegenValue<'ctx>, builder: &mut Builder<'ctx>, registry: &TypeRegistry<'ctx>, interner: &Interner, span: Span) -> Result<(), CodegenError> {
        let var = self.vars.get_mut(sym).ok_or_else(|| CodegenError::internal(format!("var {} not found", interner.resolve(sym)), span))?;
        if !var.is_mutable() { return Err(CodegenError::immutable_assignment(interner.resolve(sym), span)); }
        match val {
            CodegenValue::Bool(v) | CodegenValue::Int(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Float(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Ptr(v) | CodegenValue::StructRef(v) => { builder.build_store(var.ptr(), v).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::FatPtr(v) => { let l = builder.build_load(registry.str_type(), v, "str_load").map_err(|e| llvm_err("build_load", e, span))?; builder.build_store(var.ptr(), l).map_err(|e| llvm_err("build_store", e, span))?; }
            CodegenValue::Unit => {}
        }
        Ok(())
    }
}

//! Shared value storage and loading utilities.

use inkwell::builder::Builder;
use inkwell::types::StructType;
use inkwell::values::PointerValue;
use glyim_hir::HirType;
use glyim_span::Span;
use crate::error::{CodegenError, llvm_err};
use crate::value::CodegenValue;
use crate::types::LlvmTypes;

pub fn store_value_to_ptr<'ctx>(
    builder: &mut Builder<'ctx>, ptr: PointerValue<'ctx>, val: CodegenValue<'ctx>, str_type: StructType<'ctx>, span: Span,
) -> Result<(), CodegenError> {
    match val {
        CodegenValue::Bool(v) | CodegenValue::Int(v) => { builder.build_store(ptr, v).map_err(|e| llvm_err("build_store", e, span))?; }
        CodegenValue::Float(v) => { builder.build_store(ptr, v).map_err(|e| llvm_err("build_store", e, span))?; }
        CodegenValue::Ptr(v) | CodegenValue::StructRef(v) => { builder.build_store(ptr, v).map_err(|e| llvm_err("build_store", e, span))?; }
        CodegenValue::FatPtr(v) => {
            let l = builder.build_load(str_type, v, "str_for_field").map_err(|e| llvm_err("build_load", e, span))?;
            builder.build_store(ptr, l).map_err(|e| llvm_err("build_store", e, span))?;
        }
        CodegenValue::Unit => {}
    }
    Ok(())
}

/// HirType-directed loading. No clone needed for &HirType.
pub fn load_value_from_ptr<'ctx>(
    builder: &mut Builder<'ctx>, ptr: PointerValue<'ctx>, ty: &HirType, types: &LlvmTypes<'ctx>, span: Span,
) -> Result<CodegenValue<'ctx>, CodegenError> {
    match ty {
        HirType::Bool => Ok(CodegenValue::Bool(builder.build_load(types.i1_type, ptr, "fv").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
        HirType::Int => Ok(CodegenValue::Int(builder.build_load(types.i64_type, ptr, "fv").map_err(|e| llvm_err("build_load", e, span))?.into_int_value())),
        HirType::Float => Ok(CodegenValue::Float(builder.build_load(types.f64_type, ptr, "fv").map_err(|e| llvm_err("build_load", e, span))?.into_float_value())),
        HirType::Str => Ok(CodegenValue::FatPtr(ptr)),
        HirType::RawPtr(_) | HirType::Func(_, _) | HirType::Opaque(_) => Ok(CodegenValue::Ptr(builder.build_load(types.ptr_type, ptr, "fv").map_err(|e| llvm_err("build_load", e, span))?.into_pointer_value())),
        HirType::Named(_) | HirType::Tuple(_) => Ok(CodegenValue::StructRef(builder.build_load(types.ptr_type, ptr, "fv").map_err(|e| llvm_err("build_load", e, span))?.into_pointer_value())),
        HirType::Unit | HirType::Never => Ok(CodegenValue::Unit),
        HirType::Option(_) | HirType::Result(_, _) => Err(CodegenError::unimplemented_feature("Option/Result must be monomorphized", span)),
        _ => Err(CodegenError::internal(format!("invalid load type: {:?}", ty), span)),
    }
}
