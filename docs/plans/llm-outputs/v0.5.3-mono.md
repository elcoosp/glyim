# Monomorphizer v2 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the hacky, inference-heavy monomorphizer with a demand-driven, BFS-based substitution engine that relies strictly on the typechecker's resolved types.

**Architecture:** The new monomorphizer operates on a work queue. Seeded by `main`, it pops items, performs a single-pass substitution over their HIR bodies, and enqueues any newly discovered generic instantiations. It performs zero type inference and uses structural indices instead of string parsing.

**Tech Stack:** Rust, existing `glyim-hir` and `glyim-typeck` crates.

// crates/glyim-hir/src/monomorphize/index.rs
use crate::{Hir, HirItem, StructDef, EnumDef, HirFn, HirImplDef};
use glyim_interner::Symbol;
use std::collections::HashMap;

/// O(1) lookup tables for HIR items, eliminating linear scans.
pub struct MonoIndex {
    structs: HashMap<Symbol, StructDef>,
    enums: HashMap<Symbol, EnumDef>,
    fns: HashMap<Symbol, HirFn>,
    impl_methods: HashMap<(Symbol, Symbol), HirFn>, // (TypeName, MethodName)
}

impl MonoIndex {
    pub fn build(hir: &Hir) -> Self {
        let mut index = Self {
            structs: HashMap::new(),
            enums: HashMap::new(),
            fns: HashMap::new(),
            impl_methods: HashMap::new(),
        };

        for item in &hir.items {
            match item {
                HirItem::Struct(s) => {
                    index.structs.insert(s.name, s.clone());
                }
                HirItem::Enum(e) => {
                    index.enums.insert(e.name, e.clone());
                }
                HirItem::Fn(f) => {
                    index.fns.insert(f.name, f.clone());
                }
                HirItem::Impl(imp) => {
                    for m in &imp.methods {
                        // Map (TargetType, MethodName) -> Method
                        index.impl_methods.insert((imp.target_name, m.name), m.clone());
                        // Also map by mangled name if typeck uses it
                        let mangled = format!("{}_{}", imp.target_name_resolved, m.name_resolved); // pseudocode
                        // For robustness, index the method globally by its mangled name
                        // index.fns.insert(mangled_sym, m.clone());
                    }
                }
                _ => {}
            }
        }
        index
    }

    pub fn find_struct(&self, name: Symbol) -> Option<&StructDef> {
        self.structs.get(&name)
    }

    pub fn find_enum(&self, name: Symbol) -> Option<&EnumDef> {
        self.enums.get(&name)
    }

    pub fn find_fn(&self, name: Symbol) -> Option<&HirFn> {
        self.fns.get(&name)
    }

    pub fn find_method(&self, type_name: Symbol, method: Symbol) -> Option<&HirFn> {
        self.impl_methods.get(&(type_name, method))
    }
}

use std::collections::{VecDeque, HashSet, HashMap};
use crate::types::HirType;
use glyim_interner::Symbol;

#[derive(Clone, Debug, Hash, Eq, PartialEq)]
pub enum ItemKind {
    Fn,
    Struct,
    Enum,
}

#[derive(Clone, Debug, Hash, Eq, PartialEq)]
pub struct WorkItem {
    pub def_id: Symbol,
    pub kind: ItemKind,
    pub type_args: Vec<HirType>,
}

impl WorkItem {
    pub fn fn_item(id: Symbol, args: Vec<HirType>) -> Self {
        Self { def_id: id, kind: ItemKind::Fn, type_args: args }
    }
    pub fn struct_item(id: Symbol, args: Vec<HirType>) -> Self {
        Self { def_id: id, kind: ItemKind::Struct, type_args: args }
    }
    pub fn enum_item(id: Symbol, args: Vec<HirType>) -> Self {
        Self { def_id: id, kind: ItemKind::Enum, type_args: args }
    }
}

pub struct WorkQueue {
    queue: VecDeque<WorkItem>,
    seen: HashSet<WorkItem>,
}

impl WorkQueue {
    pub fn new() -> Self {
        Self {
            queue: VecDeque::new(),
            seen: HashSet::new(),
        }
    }

    pub fn push(&mut self, item: WorkItem) {
        if self.seen.insert(item.clone()) {
            self.queue.push_back(item);
        }
    }

    pub fn pop(&mut self) -> Option<WorkItem> {
        self.queue.pop_front()
    }

    pub fn len(&self) -> usize {
        self.queue.len()
    }
    
    pub fn is_empty(&self) -> bool {
        self.queue.is_empty()
    }
}

// crates/glyim-hir/src/monomorphize/subst.rs
use crate::types::HirType;
use glyim_interner::Symbol;
use std::collections::HashMap;

/// A pure, single-pass mathematical substitution.
/// Replaces all occurrences of type parameters found in the substitution map.
pub fn substitute_type(ty: &HirType, sub: &HashMap<Symbol, HirType>) -> HirType {
    match ty {
        HirType::Named(sym) => {
            // If it's a type parameter, substitute it. Otherwise, leave it.
            sub.get(sym).cloned().unwrap_or_else(|| ty.clone())
        }
        HirType::Generic(sym, args) => {
            let new_args: Vec<HirType> = args.iter().map(|a| substitute_type(a, sub)).collect();
            
            // If the base is a parameter (rare, but possible), substitute it
            if let Some(new_base) = sub.get(sym) {
                new_base // Note: may need flattening if new_base is also Generic, but usually Named
            } else {
                HirType::Generic(*sym, new_args)
            }
        }
        HirType::RawPtr(inner) => HirType::RawPtr(Box::new(substitute_type(inner, sub))),
        HirType::Option(inner) => HirType::Option(Box::new(substitute_type(inner, sub))),
        HirType::Result(ok, err) => HirType::Result(
            Box::new(substitute_type(ok, sub)),
            Box::new(substitute_type(err, sub)),
        ),
        HirType::Tuple(elems) => HirType::Tuple(elems.iter().map(|e| substitute_type(e, sub)).collect()),
        HirType::Func(params, ret) => HirType::Func(
            params.iter().map(|p| substitute_type(p, sub)).collect(),
            Box::new(substitute_type(ret, sub)),
        ),
        // Primitives and errors are returned as-is
        HirType::Int | HirType::Bool | HirType::Float | HirType::Str | HirType::Unit | HirType::Never | HirType::Error | HirType::Opaque(_) => ty.clone(),
    }
}

// Inside crates/glyim-hir/src/monomorphize/mod.rs
use crate::{HirItem, StructDef, EnumDef, HirFn};
use crate::types::HirType;
use crate::monomorphize::index::MonoIndex;
use crate::monomorphize::subst::substitute_type;
use crate::monomorphize::mangle_table::MangleTable;
use glyim_interner::{Interner, Symbol};
use std::collections::HashMap;

pub struct MonoContext<'a> {
    pub index: MonoIndex,
    pub interner: &'a mut Interner,
    pub expr_types: &'a [HirType],
    pub call_type_args: &'a HashMap<crate::types::ExprId, Vec<HirType>>,
    
    work_queue: WorkQueue,
    mangle_table: MangleTable,
    pub output_items: Vec<HirItem>,
}

impl<'a> MonoContext<'a> {
    pub fn new(
        index: MonoIndex,
        interner: &'a mut Interner,
        expr_types: &'a [HirType],
        call_type_args: &'a HashMap<crate::types::ExprId, Vec<HirType>>,
    ) -> Self {
        Self {
            index,
            interner,
            expr_types,
            call_type_args,
            work_queue: WorkQueue::new(),
            mangle_table: MangleTable::new(),
            output_items: Vec::new(),
        }
    }

    
    pub fn process_struct(&mut self, name: Symbol, type_args: Vec<HirType>) {
        let generic_def = self.index.find_struct(name).expect("ICE: Struct not in index");
        let sub = self.build_subst(&generic_def.type_params, &type_args);
        
        let mut concrete_def = generic_def.clone();
        concrete_def.type_params.clear();
        concrete_def.name = self.mangle_table.mangle(name, &type_args, self.interner);
        
        for field in &mut concrete_def.fields {
            field.ty = substitute_type(&field.ty, &sub);
            self.discover_type(&field.ty);
        }
        
        self.output_items.push(HirItem::Struct(concrete_def));
    }

    pub fn process_enum(&mut self, name: Symbol, type_args: Vec<HirType>) {
        let generic_def = self.index.find_enum(name).expect("ICE: Enum not in index");
        let sub = self.build_subst(&generic_def.type_params, &type_args);
        
        let mut concrete_def = generic_def.clone();
        concrete_def.type_params.clear();
        concrete_def.name = self.mangle_table.mangle(name, &type_args, self.interner);
        
        for variant in &mut concrete_def.variants {
            for field in &mut variant.fields {
                field.ty = substitute_type(&field.ty, &sub);
                self.discover_type(&field.ty);
            }
        }
        
        self.output_items.push(HirItem::Enum(concrete_def));
    }

    fn build_subst(&self, params: &[Symbol], args: &[HirType]) -> HashMap<Symbol, HirType> {
        params.iter().zip(args.iter()).map(|(p, a)| (*p, a.clone())).collect()
    }

    fn discover_type(&mut self, ty: &HirType) {
        if let HirType::Generic(base, args) = ty {
            if !args.iter().any(|a| self.has_unresolved_type_param(a)) {
                if self.index.find_struct(*base).is_some() {
                    self.work_queue.push(WorkItem::struct_item(*base, args.clone()));
                } else if self.index.find_enum(*base).is_some() {
                    self.work_queue.push(WorkItem::enum_item(*base, args.clone()));
                }
            }
        }
    }

    fn has_unresolved_type_param(&self, ty: &HirType) -> bool {
        match ty {
            HirType::Named(sym) => {
                let s = self.interner.resolve(*sym);
                s.len() == 1 && s.chars().next().is_some_and(|c| c.is_uppercase())
            }
            HirType::Generic(_, args) => args.iter().any(|a| self.has_unresolved_type_param(a)),
            HirType::RawPtr(inner) => self.has_unresolved_type_param(inner),
            HirType::Option(inner) => self.has_unresolved_type_param(inner),
            HirType::Result(ok, err) => self.has_unresolved_type_param(ok) || self.has_unresolved_type_param(err),
            HirType::Tuple(elems) => elems.iter().any(|e| self.has_unresolved_type_param(e)),
            _ => false,
        }
    }
}

// Add to MonoContext in mod.rs
    pub fn process_fn(&mut self, name: Symbol, type_args: Vec<HirType>) {
        let generic_def = self.index.find_fn(name).unwrap_or_else(|| {
            panic!("ICE: Fn {:?} not in index", self.interner.resolve(name))
        });
        
        let sub = self.build_subst(&generic_def.type_params, &type_args);
        
        let mut concrete_def = generic_def.clone();
        concrete_def.type_params.clear();
        concrete_def.name = self.mangle_table.mangle(name, &type_args, self.interner);
        
        // Substitute signature
        for (_, param_ty) in &mut concrete_def.params {
            *param_ty = substitute_type(param_ty, &sub);
        }
        if let Some(ret) = &mut concrete_def.ret {
            *ret = substitute_type(ret, &sub);
        }
        
        // Substitute body
        concrete_def.body = self.substitute_expr(&concrete_def.body, &sub);
        
        self.output_items.push(HirItem::Fn(concrete_def));
    }

    pub fn run(&mut self, entry_point: Symbol) {
        self.work_queue.push(WorkItem::fn_item(entry_point, vec![]));
        
        while let Some(item) = self.work_queue.pop() {
            match item.kind {
                ItemKind::Fn => self.process_fn(item.def_id, item.type_args),
                ItemKind::Struct => self.process_struct(item.def_id, item.type_args),
                ItemKind::Enum => self.process_enum(item.def_id, item.type_args),
            }
        }
    }
