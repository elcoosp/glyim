```rust
//! Core types for the package index.

use serde::{Deserialize, Serialize};

/// A single published version of a package in the index.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub struct VersionEntry {
    /// Semver version string
    pub version: String,
    /// Content hash of the package archive in the CAS (BLAKE3 hex)
    pub content_hash: String,
    /// Whether this version contains macro code (Wasm)
    pub is_macro: bool,
    /// Direct dependencies
    pub deps: Vec<IndexDep>,
    /// Declared capabilities (for capability-addressed discovery)
    pub capabilities: Vec<String>,
    /// Trust level of this version
    pub trust_level: TrustLevel,
    /// Granularity manifest hash (if function-level splitting was performed)
    pub granularity_manifest_hash: Option<String>,
    /// Timestamp of publication (ISO 8601)
    pub published_at: String,
    /// Publisher identity (public key hash)
    pub publisher: String,
    /// SLSA provenance attestation hash (if available)
    pub provenance_hash: Option<String>,
    /// Whether this version is yanked (do-not-use)
    pub yanked: bool,
    /// Deprecation message (if deprecated)
    pub deprecated: Option<String>,
}

/// A dependency declared by a package.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub struct IndexDep {
    /// Package name
    pub name: String,
    /// Version constraint (e.g., "^1.0.0", "*", "1.2.3")
    pub version_constraint: String,
    /// Whether this dependency is a macro (Wasm)
    pub is_macro: bool,
}

/// Progressive trust level for package versions.
///
/// Ordered: New < Established < Verified < Certified
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub enum TrustLevel {
    /// Just published, minimal verification
    New,
    /// Passes automated audit, >100 downloads, >7 days old
    Established,
    /// Full audit suite, reproducible build, community attestation
    Verified,
    /// Formal verification of critical paths, manual security review
    Certified,
}
```

```markdown
# ADR-0001: Event Log as Source of Truth

- **Status**: Accepted
- **Date**: 2025-07-11
- **Decision Drivers**: ASR-04 (event log replay ≤2s for 10K events), ASR-06 (append-only, no deletion API)

## Context

The package index needs to support temporal resolution (REQ-FUNC-011) — reconstructing the index state as it existed at any past timestamp. The index also needs an immutable audit trail (REQ-NFR-SEC-003).

## Decision

Use an append-only event log (JSONL) as the source of truth for the package index. The in-memory index (`HashMap<String, Vec<VersionEntry>>`) is a materialized view, rebuilt on startup by replaying the event log.

## Alternatives Considered

- **SQLite database**: Simpler queries, but harder to implement temporal snapshots and append-only semantics at the application level.
- **In-memory only with periodic snapshots**: Fast but loses durability across process restarts.
- **Event sourcing with Kafka**: Overkill for single-server deployment; adds operational complexity.

## Consequences

**Positive:**
- Temporal resolution is natural — replay events up to any timestamp.
- Audit trail is inherent — every mutation is recorded.
- Simple persistence model — append to a file.
- No schema migrations — event schema evolves independently.

**Negative:**
- Startup time depends on event log size (mitigated by checkpoint snapshots).
- Need checkpoint mechanism for logs with >100K events.
- In-memory index consumes RAM proportional to package count.

## Mitigations

- Implement checkpoint snapshots: periodically serialize the full in-memory index to disk and record the event ID. On startup, load the latest checkpoint and only replay events after it.
- For very large scales, consider migrating to a database-backed event store.
```

```rust
// src/event_log.rs
//! Append-only event log for temporal resolution.
//!
//! Every mutation to the index (publish, yank, unyank, deprecate, transfer)
//! is recorded as an event. The log can be replayed to any point in time,
//! enabling `resolve-as-of(timestamp)` without lockfiles.

use crate::version_entry::VersionEntry;
use serde::{Deserialize, Serialize};
use std::io::{BufRead, Write};

/// The append-only event log.
///
/// Stores events as JSONL (one JSON object per line) in `{dir}/event_log.jsonl`.
/// Events are never deleted or modified.
pub struct EventLog {
    dir: std::path::PathBuf,
    next_id: u64,
}

/// A single event in the log.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Event {
    /// Sequential event ID
    pub id: u64,
    /// ISO 8601 timestamp
    pub timestamp: String,
    /// The event payload
    pub kind: EventKind,
}

/// The types of mutations that can occur on the index.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum EventKind {
    /// A new version was published
    Publish { name: String, entry: VersionEntry },
    /// A version was yanked (marked do-not-use)
    Yank { name: String, version: String },
    /// A version was un-yanked
    Unyank { name: String, version: String },
    /// A version was deprecated with a message
    Deprecate { name: String, version: String, message: String },
    /// Package ownership was transferred
    Transfer {
        name: String,
        from_publisher: String,
        to_publisher: String,
    },
}

/// A point-in-time snapshot of the index.
#[derive(Debug, Clone)]
pub struct Snapshot {
    packages: std::collections::HashMap<String, Vec<VersionEntry>>,
}

impl Snapshot {
    /// Get all versions for a package in this snapshot.
    pub fn get_versions(&self, name: &str) -> Option<&Vec<VersionEntry>> {
        self.packages.get(name)
    }

    /// Get a specific version for a package in this snapshot.
    pub fn get_version(&self, name: &str, version: &str) -> Option<&VersionEntry> {
        self.packages
            .get(name)?
            .iter()
            .find(|v| v.version == version)
    }

    /// Convert the snapshot into the inner HashMap.
    pub fn into_packages(self) -> std::collections::HashMap<String, Vec<VersionEntry>> {
        self.packages
    }

    /// Check if the snapshot is empty.
    pub fn is_empty(&self) -> bool {
        self.packages.is_empty()
    }

    /// Get a reference to all packages in the snapshot.
    pub fn packages(&self) -> &std::collections::HashMap<String, Vec<VersionEntry>> {
        &self.packages
    }
}

impl EventLog {
    /// Open or create an event log in the given directory.
    pub fn open(dir: std::path::PathBuf) -> std::io::Result<Self> {
        std::fs::create_dir_all(&dir)?;
        let next_id = Self::count_existing_events(&dir)?;
        Ok(Self { dir, next_id })
    }

    /// Get the next event ID that will be assigned.
    pub fn next_id(&self) -> u64 {
        self.next_id
    }

    /// Append an event to the log. Returns the assigned event ID.
    pub fn append(&mut self, kind: EventKind) -> std::io::Result<u64> {
        let path = self.dir.join("event_log.jsonl");
        let mut file = std::fs::OpenOptions::new()
            .create(true)
            .append(true)
            .open(&path)?;

        let event = Event {
            id: self.next_id,
            timestamp: chrono::Utc::now().to_rfc3339(),
            kind,
        };

        let json = serde_json::to_string(&event)
            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))?;
        writeln!(file, "{}", json)?;
        file.flush()?;

        let id = self.next_id;
        self.next_id += 1;
        Ok(id)
    }

    /// Replay all events from the beginning to get the current index state.
    pub fn replay_all(&self) -> std::io::Result<Snapshot> {
        self.replay_until("9999-12-31T23:59:59Z")
    }

    /// Replay events up to (and including) a given timestamp.
    /// Returns a snapshot of the index as it existed at that moment.
    pub fn replay_until(&self, timestamp: &str) -> std::io::Result<Snapshot> {
        let mut packages: std::collections::HashMap<String, Vec<VersionEntry>> =
            std::collections::HashMap::new();

        let path = self.dir.join("event_log.jsonl");
        if !path.exists() {
            return Ok(Snapshot { packages });
        }

        let reader = std::io::BufReader::new(std::fs::File::open(&path)?);

        for line in reader.lines() {
            let line = line?;
            if line.trim().is_empty() {
                continue;
            }
            let event: Event = serde_json::from_str(&line)
                .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))?;

            if event.timestamp > timestamp {
                break;
            }

            Self::apply_event(&mut packages, event.kind);
        }

        Ok(Snapshot { packages })
    }

    /// Apply a single event to the packages map.
    fn apply_event(
        packages: &mut std::collections::HashMap<String, Vec<VersionEntry>>,
        kind: EventKind,
    ) {
        match kind {
            EventKind::Publish { name, entry } => {
                packages.entry(name).or_default().push(entry);
            }
            EventKind::Yank { name, version } => {
                if let Some(versions) = packages.get_mut(&name) {
                    for v in versions.iter_mut() {
                        if v.version == version {
                            v.yanked = true;
                        }
                    }
                }
            }
            EventKind::Unyank { name, version } => {
                if let Some(versions) = packages.get_mut(&name) {
                    for v in versions.iter_mut() {
                        if v.version == version {
                            v.yanked = false;
                        }
                    }
                }
            }
            EventKind::Deprecate {
                name,
                version,
                message,
            } => {
                if let Some(versions) = packages.get_mut(&name) {
                    for v in versions.iter_mut() {
                        if v.version == version {
                            v.deprecated = Some(message.clone());
                        }
                    }
                }
            }
            EventKind::Transfer {
                name,
                from_publisher: _,
                to_publisher,
            } => {
                if let Some(versions) = packages.get_mut(&name) {
                    for v in versions.iter_mut() {
                        v.publisher = to_publisher.clone();
                    }
                }
            }
        }
    }

    /// Count existing events in the log file.
    fn count_existing_events(dir: &std::path::Path) -> std::io::Result<u64> {
        let path = dir.join("event_log.jsonl");
        if !path.exists() {
            return Ok(0);
        }
        let count = std::io::BufReader::new(std::fs::File::open(&path)?)
            .lines()
            .filter(|line| line.as_ref().map(|l| !l.trim().is_empty()).unwrap_or(false))
            .count() as u64;
        Ok(count)
    }
}
```

```rust
// src/index.rs
//! The package index — maps package names to their version entries.
//!
//! Backed by an append-only event log for temporal resolution.
//! The in-memory index is a materialized view rebuilt on startup.

use crate::event_log::{EventKind, EventLog, Snapshot};
use crate::version_entry::VersionEntry;
use std::collections::HashMap;

/// The package index.
///
/// Stores an in-memory mapping of package names to their version entries.
/// All mutations are recorded in an append-only event log.
pub struct PackageIndex {
    /// In-memory cache: package name → version entries
    packages: HashMap<String, Vec<VersionEntry>>,
    /// Append-only event log
    event_log: EventLog,
}

impl PackageIndex {
    /// Open or create the index, backed by the given data directory.
    ///
    /// On startup, replays the event log to rebuild the in-memory index.
    pub fn open(data_dir: &std::path::Path) -> std::io::Result<Self> {
        let event_log = EventLog::open(data_dir.join("events"))?;
        let snapshot = event_log.replay_all()?;
        let packages = snapshot.into_packages();

        // Filter out yanked versions from the default view
        let packages: HashMap<String, Vec<VersionEntry>> = packages
            .into_iter()
            .map(|(name, versions)| {
                let filtered: Vec<VersionEntry> = versions
                    .into_iter()
                    .filter(|v| !v.yanked)
                    .collect();
                (name, filtered)
            })
            .collect();

        Ok(Self { packages, event_log })
    }

    /// Publish a new version.
    ///
    /// Returns an error if the version already exists.
    /// The publish is first written to the event log (write-ahead),
    /// then applied to the in-memory index.
    pub fn publish(
        &mut self,
        name: &str,
        entry: VersionEntry,
    ) -> Result<u64, IndexError> {
        // Check for duplicate version
        if let Some(versions) = self.packages.get(name) {
            if versions.iter().any(|v| v.version == entry.version) {
                return Err(IndexError::VersionExists {
                    name: name.to_string(),
                    version: entry.version,
                });
            }
        }

        // Write to event log first (durability)
        let event_id = self.event_log.append(EventKind::Publish {
            name: name.to_string(),
            entry: entry.clone(),
        })?;

        // Then update in-memory index
        self.packages
            .entry(name.to_string())
            .or_default()
            .push(entry);

        Ok(event_id)
    }

    /// Yank a version (mark as do-not-use).
    pub fn yank(&mut self, name: &str, version: &str) -> Result<u64, IndexError> {
        // Find the entry in the full (including yanked) set
        let found = self.find_version_any(name, version).is_some();

        if !found {
            return Err(IndexError::VersionNotFound {
                name: name.to_string(),
                version: version.to_string(),
            });
        }

        let event_id = self.event_log.append(EventKind::Yank {
            name: name.to_string(),
            version: version.to_string(),
        })?;

        // Remove from the non-yanked view
        if let Some(versions) = self.packages.get_mut(name) {
            versions.retain(|v| v.version != version);
        }

        Ok(event_id)
    }

    /// Get all non-yanked versions for a package.
    pub fn get_versions(&self, name: &str) -> Vec<&VersionEntry> {
        self.packages
            .get(name)
            .map(|v| v.iter().collect())
            .unwrap_or_default()
    }

    /// Get a specific (non-yanked) version.
    pub fn get_version(&self, name: &str, version: &str) -> Option<&VersionEntry> {
        self.packages
            .get(name)?
            .iter()
            .find(|v| v.version == version)
    }

    /// Get the index state as of a timestamp (temporal resolution).
    pub fn snapshot_at(&self, timestamp: &str) -> std::io::Result<Snapshot> {
        self.event_log.replay_until(timestamp)
    }

    /// Get the current snapshot (including yanked versions).
    pub fn snapshot_current(&self) -> Snapshot {
        self.event_log.replay_all().unwrap_or_else(|_| Snapshot {
            packages: HashMap::new(),
        })
    }

    /// Helper: find a version even if it's yanked (by replaying events).
    fn find_version_any(&self, name: &str, version: &str) -> Option<VersionEntry> {
        let snapshot = self.event_log.replay_all().ok()?;
        snapshot
            .packages()
            .get(name)?
            .iter()
            .find(|v| v.version == version)
            .cloned()
    }
}

/// Errors that can occur during index operations.
#[derive(Debug, thiserror::Error)]
pub enum IndexError {
    /// A version already exists for this package
    #[error("version {version} already exists for package {name}")]
    VersionExists { name: String, version: String },
    /// A package was not found
    #[error("package {name} not found")]
    NotFound { name: String },
    /// A specific version was not found
    #[error("version {version} not found for package {name}")]
    VersionNotFound { name: String, version: String },
    /// An I/O error occurred
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
}
```

```rust
//! Search functions for the package index.

use crate::index::PackageIndex;
use crate::version_entry::VersionEntry;
use serde::Serialize;

/// A search result with relevance scoring.
#[derive(Debug, Clone, Serialize)]
pub struct SearchResult {
    /// Package name
    pub name: String,
    /// Latest version string
    pub latest_version: String,
    /// Package description (if available)
    pub description: Option<String>,
    /// Trust level as string
    pub trust_level: String,
    /// Relevance score (0.0–1.0)
    pub relevance: f64,
}

impl PackageIndex {
    /// Search by exact name match or name prefix.
    ///
    /// Returns package names that start with the given prefix.
    pub fn search_by_name(&self, prefix: &str) -> Vec<&str> {
        self.packages()
            .keys()
            .filter(|k| k.starts_with(prefix))
            .map(|k| k.as_str())
            .collect()
    }

    /// Search packages by declared capability.
    ///
    /// Returns (package_name, version_entry) pairs for packages
    /// that declare the given capability.
    pub fn search_by_capability(&self, capability: &str) -> Vec<(&str, &VersionEntry)> {
        self.packages()
            .iter()
            .flat_map(|(name, versions)| {
                versions
                    .iter()
                    .filter(|v| v.capabilities.contains(&capability.to_string()))
                    .map(|v| (name.as_str(), v))
            })
            .collect()
    }

    /// Combined search: exact match + name prefix + capability match.
    ///
    /// Returns results sorted by relevance (descending).
    pub fn search(&self, query: &str) -> Vec<SearchResult> {
        let mut results: Vec<SearchResult> = Vec::new();
        let mut seen = std::collections::HashSet::new();

        // 1. Exact name match (highest relevance = 1.0)
        if let Some(versions) = self.packages().get(query) {
            if let Some(latest) = versions.first() {
                seen.insert(query.to_string());
                results.push(SearchResult {
                    name: query.to_string(),
                    latest_version: latest.version.clone(),
                    description: None,
                    trust_level: format!("{:?}", latest.trust_level),
                    relevance: 1.0,
                });
            }
        }

        // 2. Name prefix match (relevance = 0.8 * query_len / name_len)
        for (name, versions) in self.packages() {
            if name.starts_with(query) && name != query {
                if seen.insert(name.clone()) {
                    if let Some(latest) = versions.first() {
                        let relevance = 0.8 * (query.len() as f64 / name.len() as f64);
                        results.push(SearchResult {
                            name: name.clone(),
                            latest_version: latest.version.clone(),
                            description: None,
                            trust_level: format!("{:?}", latest.trust_level),
                            relevance,
                        });
                    }
                }
            }
        }

        // 3. Capability match (relevance = 0.5)
        for (name, versions) in self.packages() {
            if let Some(latest) = versions.first() {
                if latest.capabilities.iter().any(|c| c.contains(query)) {
                    if seen.insert(name.clone()) {
                        results.push(SearchResult {
                            name: name.clone(),
                            latest_version: latest.version.clone(),
                            description: None,
                            trust_level: format!("{:?}", latest.trust_level),
                            relevance: 0.5,
                        });
                    }
                }
            }
        }

        // Sort by relevance descending
        results.sort_by(|a, b| {
            b.relevance
                .partial_cmp(&a.relevance)
                .unwrap_or(std::cmp::Ordering::Equal)
        });

        results
    }
```

```rust
    pub fn search_by_name(&self, prefix: &str) -> Vec<&str> {
        self.packages_ref()
            .keys()
            .filter(|k| k.starts_with(prefix))
            .map(|k| k.as_str())
            .collect()
    }

    pub fn search_by_capability(&self, capability: &str) -> Vec<(&str, &VersionEntry)> {
        self.packages_ref()
            .iter()
            .flat_map(|(name, versions)| {
                versions
                    .iter()
                    .filter(|v| v.capabilities.contains(&capability.to_string()))
                    .map(|v| (name.as_str(), v))
            })
            .collect()
    }

    pub fn search(&self, query: &str) -> Vec<SearchResult> {
        let mut results: Vec<SearchResult> = Vec::new();
        let mut seen = std::collections::HashSet::new();

        // 1. Exact name match
        if let Some(versions) = self.packages_ref().get(query) {
            if let Some(latest) = versions.first() {
                seen.insert(query.to_string());
                results.push(SearchResult {
                    name: query.to_string(),
                    latest_version: latest.version.clone(),
                    description: None,
                    trust_level: format!("{:?}", latest.trust_level),
                    relevance: 1.0,
                });
            }
        }

        // 2. Name prefix match
        for (name, versions) in self.packages_ref() {
            if name.starts_with(query) && name != query {
                if seen.insert(name.clone()) {
                    if let Some(latest) = versions.first() {
                        let relevance = 0.8 * (query.len() as f64 / name.len() as f64);
                        results.push(SearchResult {
                            name: name.clone(),
                            latest_version: latest.version.clone(),
                            description: None,
                            trust_level: format!("{:?}", latest.trust_level),
                            relevance,
                        });
                    }
                }
            }
        }

        // 3. Capability match
        for (name, versions) in self.packages_ref() {
            if let Some(latest) = versions.first() {
                if latest.capabilities.iter().any(|c| c.contains(query)) {
                    if seen.insert(name.clone()) {
                        results.push(SearchResult {
                            name: name.clone(),
                            latest_version: latest.version.clone(),
                            description: None,
                            trust_level: format!("{:?}", latest.trust_level),
                            relevance: 0.5,
                        });
                    }
                }
            }
        }

        results.sort_by(|a, b| {
            b.relevance
                .partial_cmp(&a.relevance)
                .unwrap_or(std::cmp::Ordering::Equal)
        });

        results
    }
}
```

```rust
    pub trust_level: String,
    pub content_hash: String,

/// GET /api/v1/packages/{name}
///
/// Returns all non-yanked versions of a package.
pub async fn get_package(
    State(state): State<Arc<AppState>>,
    Path(name): Path<String>,
) -> impl IntoResponse {
    let index = state.index.read().await;
    let versions = index.get_versions(&name);

    if versions.is_empty() {
        return (StatusCode::NOT_FOUND, "package not found").into_response();
    }

    let response = PackageResponse {
        name: name.clone(),
        versions: versions
            .iter()
            .map(|v| VersionSummary {
                version: v.version.clone(),
                is_macro: v.is_macro,
                trust_level: format!("{:?}", v.trust_level),
                content_hash: v.content_hash.clone(),
                capabilities: v.capabilities.clone(),
                deprecated: v.deprecated.clone(),
            })
            .collect(),
    };

    Json(response).into_response()
}

/// GET /api/v1/packages/{name}/{version}

/// Returns the full VersionEntry for a specific version.
pub async fn get_version(
    State(state): State<Arc<AppState>>,
    Path((name, version)): Path<(String, String)>,
) -> impl IntoResponse {
    let index = state.index.read().await;
    match index.get_version(&name, &version) {
        Some(entry) => Json(entry).into_response(),
```

```rust
// src/constraints.rs
//! Extended version constraint types.

use semver::Version;

/// A version constraint that can be satisfied by a semver version.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum VersionConstraint {
    /// Exact version match: "1.2.3"
    Exact(Version),
    /// Caret: "^1.2.3" — >=1.2.3, <2.0.0
    Caret(Version),
    /// Tilde: "~1.2.3" — >=1.2.3, <1.3.0
    Tilde(Version),
    /// Wildcard: "*" — any version
    Wildcard,
}

impl VersionConstraint {
    /// Parse a version constraint string.
    pub fn parse(s: &str) -> Result<Self, String> {
        let s = s.trim();

        if s == "*" || s.is_empty() {
            return Ok(VersionConstraint::Wildcard);
        }

        if let Some(rest) = s.strip_prefix('^') {
            let v = Version::parse(rest.trim())
                .map_err(|e| format!("invalid caret version '{}': {}", rest, e))?;
            return Ok(VersionConstraint::Caret(v));
        }

        if let Some(rest) = s.strip_prefix('~') {
            let v = Version::parse(rest.trim())
                .map_err(|e| format!("invalid tilde version '{}': {}", rest, e))?;
            return Ok(VersionConstraint::Tilde(v));
        }

        let v = Version::parse(s)
            .map_err(|e| format!("invalid version '{}': {}", s, e))?;
        Ok(VersionConstraint::Exact(v))
    }

    /// Check if a version string satisfies this constraint.
    pub fn satisfies(&self, version_str: &str) -> bool {
        let version = match Version::parse(version_str) {
            Ok(v) => v,
            Err(_) => return false,
        };

        match self {
            VersionConstraint::Exact(v) => &version == v,
            VersionConstraint::Caret(v) => {
                version >= *v && version.major == v.major
            }
            VersionConstraint::Tilde(v) => {
                version >= *v && version.major == v.major && version.minor == v.minor
            }
            VersionConstraint::Wildcard => true,
        }
    }
}
```

```rust
// src/solver.rs
//! Server-side dependency resolver.

use crate::constraints::VersionConstraint;
use glyim_registry_index::{
    version_entry::VersionEntry, PackageIndex,
};
use hlock::{
    DepType, Dependency, HashAlgorithm, IntegrityHash, Lockfile, Package, Source,
};
use std::collections::HashMap;

/// Server-side dependency resolver.
pub struct ServerResolver {
    index: PackageIndex,
}

/// What to optimize for during resolution.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum ResolutionObjective {
    /// Minimal versions (like traditional minimal version selection)
    MinimalVersions,
    /// Latest compatible versions (better security patches)
    LatestVersions,
    /// Highest trust scores
    MaximalTrust,
}

/// A complete resolution result.
pub struct ServerResolution {
    /// The resolved hlock Lockfile
    pub lockfile: Lockfile,
    /// Cache key for this resolution
    pub cache_key: String,
    /// Time taken to resolve
    pub resolve_time_ms: u64,
}

impl ServerResolver {
    /// Create a new resolver backed by the given index.
    pub fn new(index: PackageIndex) -> Self {
        Self { index }
    }

    /// Resolve dependencies and produce an hlock Lockfile.
    pub fn resolve(
        &self,
        root_deps: &[(String, String)],
        registry_url: &str,
        objective: ResolutionObjective,
        _resolve_as_of: Option<&str>,
    ) -> Result<ServerResolution, ResolverError> {
        let start = std::time::Instant::now();

        let resolved = self.solve(root_deps, &objective)?;

        let source = Source::CasHttp(registry_url.to_string());
        let mut packages = Vec::new();

        for (name, entry) in &resolved {
            let deps: Vec<Dependency> = entry.deps.iter().map(|d| Dependency {
                name: d.name.clone(),
                dep_type: DepType::Runtime,
                requested_features: vec![],
            }).collect();

            let (major, minor, patch) = parse_version(&entry.version);

            packages.push(Package {
                name: name.clone(),
                logical_name: None,
                source_idx: 0,
                major,
                minor,
                patch,
                hashes: vec![IntegrityHash {
                    algo: HashAlgorithm::Blake3,
                    digest: hex_to_bytes(&entry.content_hash),
                    attestation: hlock::Attestation::None,
                }],
                features: vec![],
                resolved_peers: vec![],
                dependencies: deps,
                peer_requirements: vec![],
                platform_tags: vec![],
                exports: vec![],
                artifacts: vec![],
                hook_hashes: vec![],
                patch_hash: None,
            });
        }

        let lockfile = Lockfile {
            sources: vec![source],
            overrides: vec![],
            features: vec![],
            metadata: vec![],
            workspace_root: None,
            workspace_pkgs: vec![],
            hoist_boundaries: vec![],
            packages,
            artifacts: vec![],
            patches: vec![],
        };

        let cache_key = compute_cache_key(root_deps, &objective);

        Ok(ServerResolution {
            lockfile,
            cache_key,
            resolve_time_ms: start.elapsed().as_millis() as u64,
        })
    }

    fn solve(
        &self,
        root_deps: &[(String, String)],
        objective: &ResolutionObjective,
    ) -> Result<HashMap<String, VersionEntry>, ResolverError> {
        let mut resolved: HashMap<String, VersionEntry> = HashMap::new();
        let mut queue: Vec<(String, String)> = root_deps.to_vec();

        while let Some((name, constraint_str)) = queue.pop() {
            if resolved.contains_key(&name) {
                continue;
            }

            let constraint = VersionConstraint::parse(&constraint_str)
                .map_err(|e| ResolverError::ConstraintParse(e))?;

            let versions = self.index.get_versions(&name);
            if versions.is_empty() {
                return Err(ResolverError::PackageNotFound(name));
            }

            let satisfying: Vec<&VersionEntry> = versions
                .iter()
                .filter(|v| constraint.satisfies(&v.version))
                .collect();

            let selected = select_version(&satisfying, objective)
                .ok_or_else(|| ResolverError::NoSatisfyingVersion {
                    name: name.clone(),
                    constraint: constraint_str,
                })?;

            for dep in &selected.deps {
                if !resolved.contains_key(&dep.name) {
                    queue.push((dep.name.clone(), dep.version_constraint.clone()));
                }
            }

            resolved.insert(name, selected.clone());
        }

        Ok(resolved)
    }
}

fn select_version<'a>(
    versions: &[&'a VersionEntry],
    objective: &ResolutionObjective,
) -> Option<&'a VersionEntry> {
    if versions.is_empty() {
        return None;
    }

    match objective {
        ResolutionObjective::MinimalVersions => versions.iter().min_by(|a, b| {
            semver::Version::parse(&a.version)
                .unwrap_or_default()
                .cmp(&semver::Version::parse(&b.version).unwrap_or_default())
        }),
        ResolutionObjective::LatestVersions => versions.iter().max_by(|a, b| {
            semver::Version::parse(&a.version)
                .unwrap_or_default()
                .cmp(&semver::Version::parse(&b.version).unwrap_or_default())
        }),
        ResolutionObjective::MaximalTrust => versions.iter().max_by(|a, b| {
            a.trust_level.cmp(&b.trust_level)
        }),
    }
    .copied()
}

fn parse_version(version: &str) -> (u64, u64, u64) {
    let parts: Vec<&str> = version.split('.').collect();
    (
        parts.first().and_then(|s| s.parse().ok()).unwrap_or(0),
        parts.get(1).and_then(|s| s.parse().ok()).unwrap_or(0),
        parts.get(2).and_then(|s| s.parse().ok()).unwrap_or(0),
    )
}

fn hex_to_bytes(hex: &str) -> Vec<u8> {
    (0..hex.len())
        .step_by(2)
        .filter_map(|i| u8::from_str_radix(&hex[i..i + 2], 16).ok())
        .collect()
}

fn compute_cache_key(
    deps: &[(String, String)],
    objective: &ResolutionObjective,
) -> String {
    use sha2::{Digest, Sha256};
    let mut hasher = Sha256::new();
    for (name, constraint) in deps {
        hasher.update(name.as_bytes());
        hasher.update(constraint.as_bytes());
    }
    hasher.update(format!("{:?}", objective).as_bytes());
    hex::encode(hasher.finalize())
}

/// Errors that can occur during resolution.
#[derive(Debug, thiserror::Error)]
pub enum ResolverError {
    #[error("package not found: {0}")]
    PackageNotFound(String),
    #[error("no version of '{name}' satisfies constraint {constraint}")]
    NoSatisfyingVersion { name: String, constraint: String },
    #[error("constraint parse error: {0}")]
    ConstraintParse(String),
    #[error("cache error: {0}")]
    Cache(String),
}
```

```rust
// src/resolve_handlers.rs
//! REST handlers for dependency resolution.

use axum::{
    extract::State,
    http::StatusCode,
    response::{IntoResponse, Json},
};
use glyim_registry_resolver::{ResolutionObjective, ServerResolver};
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::RwLock;

use crate::AppState;

#[derive(Deserialize)]
pub struct ResolveRequest {
    pub dependencies: std::collections::HashMap<String, String>,
    pub resolve_as_of: Option<String>,
    pub objective: Option<String>,
}

#[derive(Serialize)]
pub struct ResolveResponse {
    pub cache_key: String,
    pub resolve_time_ms: u64,
    pub package_count: usize,
    pub lockfile_bytes: String, // base64-encoded
}

/// POST /api/v1/resolve
pub async fn resolve(
    State(state): State<Arc<AppState>>,
    Json(req): Json<ResolveRequest>,
) -> impl IntoResponse {
    let index = state.index.read().await;
    let resolver = ServerResolver::new(index.clone());

    let objective = match req.objective.as_deref() {
        Some("minimal") => ResolutionObjective::MinimalVersions,
        Some("trust") => ResolutionObjective::MaximalTrust,
        _ => ResolutionObjective::LatestVersions,
    };

    let deps: Vec<(String, String)> = req.dependencies.into_iter().collect();

    match resolver.resolve(
        &deps,
        "http://127.0.0.1:9090",
        objective,
        req.resolve_as_of.as_deref(),
    ) {
        Ok(result) => {
            // Serialize the lockfile
            let mut lockfile = result.lockfile;
            match hlock::serialize(&mut lockfile) {
                Ok(serialized) => {
                    let response = ResolveResponse {
                        cache_key: result.cache_key,
                        resolve_time_ms: result.resolve_time_ms,
                        package_count: lockfile.packages.len(),
                        lockfile_bytes: base64_encode(&serialized),
                    };
                    Json(response).into_response()
                }
                Err(e) => {
                    (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response()
                }
            }
        }
        Err(e) => {
            (StatusCode::UNPROCESSABLE_ENTITY, e.to_string()).into_response()
        }
    }
}

fn base64_encode(data: &[u8]) -> String {
    use base64::engine::general_purpose::STANDARD;
    use base64::Engine;
    STANDARD.encode(data)
}
```

```rust
pub struct TrustSignals {
    pub age_days: u32,
    pub download_count: u64,
    pub publisher_count: u32,
    pub audit_passed: bool,
    pub reproducible_build: bool,
    pub author_signed: bool,

}

impl TrustScorer {

    pub fn score(signals: &TrustSignals) -> TrustLevel {
        let mut points: u32 = 0;

        // Age-based trust
        if signals.age_days >= 365 {
            points += 20;
        } else if signals.age_days >= 30 {
            points += 10;
        } else if signals.age_days >= 7 {
            points += 5;
        }

        // Download count
        if signals.download_count >= 10000 {
            points += 15;
        } else if signals.download_count >= 100 {
            points += 5;
        }

        // Publisher count
        if signals.publisher_count >= 3 {
            points += 10;
        } else if signals.publisher_count >= 2 {
            points += 5;
        }

        // Automated audit
        if signals.audit_passed {
            points += 15;
        }

        // Reproducible build
        if signals.reproducible_build {
            points += 15;
        }

        // Author signature
        if signals.author_signed {
            points += 5;
        }

        // CI attestation
        if signals.ci_attested {
            points += 10;
        }

        // Community attestations
        if signals.community_attestations >= 3 {
            points += 10;
        } else if signals.community_attestations >= 1 {
            points += 5;
        }

        // No known vulnerabilities
        if !signals.has_known_vulnerabilities {
            points += 10;
        }

        // Map points to trust level
        match points {
            0..=29 => TrustLevel::New,
            30..=59 => TrustLevel::Established,
            60..=89 => TrustLevel::Verified,
```

```rust
// src/manifest.rs
//! GranularityManifest — maps exports to CAS blob hashes.

use serde::{Deserialize, Serialize};
use std::collections::HashMap;

/// The granularity manifest stored alongside a package in the CAS.
/// Maps each export name to its minimal CAS blob hash.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
pub struct GranularityManifest {
    /// Package name
    pub package: String,
    /// Package version
    pub version: String,
    /// Export name → CAS blob hash
    pub export_blobs: HashMap<String, String>,
    /// Full artifact blob hash (fallback when no granularity)
    pub full_blob_hash: String,
}

impl GranularityManifest {
    /// Convert to hlock Export entries.
    pub fn to_hlock_exports(&self) -> Vec<hlock::Export> {
        self.export_blobs
            .iter()
            .map(|(identifier, hash_hex)| hlock::Export {
                identifier: identifier.clone(),
                hash_algo: hlock::HashAlgorithm::Blake3,
                digest: hex_to_bytes(hash_hex),
            })
            .collect()
    }

    /// Get the blob hash for a specific export.
    pub fn get_export_blob(&self, export_name: &str) -> Option<&str> {
        self.export_blobs.get(export_name).map(|s| s.as_str())
    }
}

fn hex_to_bytes(hex: &str) -> Vec<u8> {
    (0..hex.len())
        .step_by(2)
        .filter_map(|i| u8::from_str_radix(&hex[i..i + 2], 16).ok())
        .collect()
}
```

```rust
// src/tree_shake.rs
//! Tree-shaking: compute the minimal set of blobs for requested exports.

use crate::manifest::GranularityManifest;

/// Result of tree-shaking: the minimal set of blob hashes needed.
#[derive(Debug, Clone)]
pub struct TreeShakeResult {
    /// The blob hashes to fetch
    pub blob_hashes: Vec<String>,
    /// Total size in bytes (if known)
    pub total_size_bytes: Option<u64>,
}

/// Compute the minimal set of blobs needed for the requested exports.
///
/// If an export is not found in the manifest, falls back to the full blob.
pub fn tree_shake(
    manifest: &GranularityManifest,
    requested_exports: &[String],
) -> TreeShakeResult {
    let mut blob_hashes = Vec::new();
    let mut seen = std::collections::HashSet::new();

    for export_name in requested_exports {
        if let Some(blob_hash) = manifest.get_export_blob(export_name) {
            if seen.insert(blob_hash.to_string()) {
                blob_hashes.push(blob_hash.to_string());
            }
        } else {
            // Fallback: include full blob
            if seen.insert(manifest.full_blob_hash.clone()) {
                blob_hashes.push(manifest.full_blob_hash.clone());
            }
        }
    }

    // If no exports were matched, return the full blob
    if blob_hashes.is_empty() {
        blob_hashes.push(manifest.full_blob_hash.clone());
    }

    TreeShakeResult {
        blob_hashes,
        total_size_bytes: None,
    }
}
```

```rust
//! Fuzzy capability matching.

/// Compute a relevance score (0.0–1.0) for a capability query against
/// a list of declared capabilities.
pub fn match_capability(query: &str, capabilities: &[String]) -> f64 {
    let query_lower = query.to_lowercase();

    for cap in capabilities {
        let cap_lower = cap.to_lowercase();

        // Exact match
        if cap_lower == query_lower {
            return 1.0;
        }

        // Contains match
        if cap_lower.contains(&query_lower) || query_lower.contains(&cap_lower) {
            return 0.7 * (query.len().min(cap.len()) as f64 / query.len().max(cap.len()) as f64);
        }

        // Prefix match
        if cap_lower.starts_with(&query_lower) || query_lower.starts_with(&cap_lower) {
            return 0.5 * (query.len().min(cap.len()) as f64 / query.len().max(cap.len()) as f64);
        }
    }

    0.0
}
```

```markdown
# ADR-0002: hlock as Lockfile Format

- **Status**: Accepted
- **Date**: 2025-07-11
- **Decision Drivers**: CON-02 (must use hlock), ASR-07 (approved crypto only)

## Context
The resolver needs to output lockfiles with cryptographic integrity (BLAKE3), PQ signatures (ML-DSA-65), per-export hashes, and graph operations.

## Decision
Use hlock as the exclusive lockfile format without modifying its source.

## Alternatives
- Custom TOML lockfile: no built-in crypto or graph operations
- Extend hlock: risk of forking

## Consequences
+ All crypto and graph ops provided by hlock
+ Binary format is compact and tamper-evident
- Dependency on external crate
- Must map VersionEntry fields to hlock Package fields
```
