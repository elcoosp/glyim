use crate::Span;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum Severity { Error, Warning, Note }

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
pub enum LabelStyle {
    #[default]
    Primary,
    Secondary,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct DiagnosticLabel {
    pub span: Span,
    pub message: String,
    pub style: LabelStyle,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ExpansionFrame {
    pub macro_name: String,
    pub location: Option<Span>,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Diagnostic {
    pub severity: Severity,
    pub message: String,
    pub span: Option<Span>,
    pub labels: Vec<DiagnosticLabel>,
    pub notes: Vec<String>,
    pub expansion_trace: Vec<ExpansionFrame>,
}

impl Diagnostic {
    pub fn error(message: impl Into<String>) -> Self {
        Self { severity: Severity::Error, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn warning(message: impl Into<String>) -> Self {
        Self { severity: Severity::Warning, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn note(message: impl Into<String>) -> Self {
        Self { severity: Severity::Note, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn with_span(mut self, span: Span) -> Self { self.span = Some(span); self }
    pub fn with_span_opt(mut self, span: Option<Span>) -> Self { self.span = span; self }
    pub fn is_error(&self) -> bool { self.severity == Severity::Error }

    /// Add a primary label at the given span.
    pub fn with_label(mut self, span: Span, message: impl Into<String>) -> Self {
        self.labels.push(DiagnosticLabel { span, message: message.into(), style: LabelStyle::Primary });
        self
    }

    /// Add a label with explicit style (primary or secondary).
    pub fn with_label_styled(mut self, span: Span, style: LabelStyle, message: impl Into<String>) -> Self {
        self.labels.push(DiagnosticLabel { span, message: message.into(), style });
        self
    }

    /// Add a note below the diagnostic.
    pub fn with_note(mut self, message: impl Into<String>) -> Self {
        self.notes.push(message.into());
        self
    }

    /// Add a macro expansion frame to the trace.
    pub fn with_expansion_frame(mut self, macro_name: impl Into<String>, location: Option<Span>) -> Self {
        self.expansion_trace.push(ExpansionFrame { macro_name: macro_name.into(), location });
        self
    }
}

use crate::{Diagnostic, ExpansionFrame, LabelStyle, Severity};
use ariadne::{Label, Report, ReportKind, Source};

pub fn render_diagnostics(source: &str, file_path: &str, diagnostics: &[Diagnostic]) -> String {
    let mut out = Vec::new();
    let cache = Source::from(source);
    for diag in diagnostics {
        let rendered = render_one(source, file_path, diag, &cache);
        out.extend(rendered);
    }
    String::from_utf8(out).unwrap_or_default()
}

pub fn render_single(source: &str, file_path: &str, diag: &Diagnostic) -> String {
    let cache = Source::from(source);
    let bytes = render_one(source, file_path, diag, &cache);
    String::from_utf8(bytes).unwrap_or_default()
}

fn render_one(source: &str, file_path: &str, diag: &Diagnostic, cache: &Source) -> Vec<u8> {
    let mut out = Vec::new();
    let kind = match diag.severity {
        Severity::Error => ReportKind::Error,
        Severity::Warning => ReportKind::Warning,
        Severity::Note => ReportKind::Advice,
    };

    // Determine the primary span for the report header
    let primary_span = if !diag.labels.is_empty() {
        diag.labels.iter().find(|l| l.style == LabelStyle::Primary)
            .or_else(|| diag.labels.first())
            .map(|l| l.span.start..l.span.end)
    } else {
        diag.span.map(|s| s.start..s.end)
    };
    let header_span = primary_span.unwrap_or(0..0);

    let mut report = Report::build(kind, (file_path, header_span)).with_message(&diag.message);

    // Add labels
    if !diag.labels.is_empty() {
        for label in &diag.labels {
            let ariadne_style = match label.style {
                LabelStyle::Primary => ariadne::LabelStyle::Primary,
                LabelStyle::Secondary => ariadne::LabelStyle::Secondary,
            };
            if label.span.start != label.span.end {
                report = report.with_label(
                    Label::new((file_path, label.span.start..label.span.end))
                        .with_message(&label.message)
                        .with_style(ariadne_style),
                );
            }
        }
    } else if let Some(span) = diag.span {
        // Backward compat: single span, no labels
        if span.start != span.end {
            report = report.with_label(
                Label::new((file_path, span.start..span.end)).with_message(&diag.message),
            );
        }
    }

    // Add notes
    for note_text in &diag.notes {
        report = report.with_note(note_text);
    }

    // Add expansion trace as notes
    if !diag.expansion_trace.is_empty() {
        let mut trace_lines = Vec::new();
        for frame in &diag.expansion_trace {
            let loc = frame.location
                .map(|s| format!(":{}", s.start))
                .unwrap_or_default();
            trace_lines.push(format!("  {}{}", frame.macro_name, loc));
        }
        report = report.with_note(format!("expansion trace:\n{}", trace_lines.join("\n")));
    }

    report
        .finish()
        .write((file_path, cache.clone()), &mut out)
        .unwrap();
    out
}

use crate::{HirBinOp, HirExpr, HirStmt, HirUnOp};

/// Fluent builder for constructing HIR nodes.
/// This is the target API that `quote!` expansion will generate calls to.
pub struct HirBuilder;

impl HirBuilder {
    // â”€â”€ Literals â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

    pub fn int_lit(n: i64) -> HirExpr {
        HirExpr::IntLit(n)
    }

    pub fn str_lit(s: impl Into<String>) -> HirExpr {
        HirExpr::StrLit(s.into())
    }

    pub fn ident(sym: Symbol) -> HirExpr {
        HirExpr::Ident(sym)
    }

    pub fn ident_str(interner: &mut Interner, name: &str) -> HirExpr {
        HirExpr::Ident(interner.intern(name))
    }

    // â”€â”€ Operators â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

    pub fn binary(op: HirBinOp, lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        HirExpr::Binary { op, lhs: Box::new(lhs), rhs: Box::new(rhs) }
    }

    pub fn add(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Add, lhs, rhs)
    }

    pub fn sub(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Sub, lhs, rhs)
    }

    pub fn mul(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Mul, lhs, rhs)
    }

    pub fn neg(operand: HirExpr) -> HirExpr {
        HirExpr::Unary { op: HirUnOp::Neg, operand: Box::new(operand) }
    }

    pub fn not(operand: HirExpr) -> HirExpr {
        HirExpr::Unary { op: HirUnOp::Not, operand: Box::new(operand) }
    }

    // â”€â”€ Control flow â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

    pub fn block(stmts: Vec<HirStmt>) -> HirExpr {
        HirExpr::Block(stmts)
    }

    pub fn block_expr(exprs: Vec<HirExpr>) -> HirExpr {
        let stmts: Vec<HirStmt> = exprs.into_iter().map(HirStmt::Expr).collect();

fn parse_quote_expr(&mut self) -> Option<ExprNode> {
    // We're positioned at Ident("quote")
    let name_tok = self.tokens.bump()?; // consume "quote"
    let name_start = name_tok.start;

    if !self.tokens.eat(SyntaxKind::Bang) {
        // Not a macro invocation â€” this shouldn't be reached since we check in parse_expr_atom
        // But handle gracefully: treat "quote" as an ident
        let sym = self.interner.intern(name_tok.text);
        return Some(ExprNode {
            kind: ExprKind::Ident(sym),
            span: Span::new(name_start, name_tok.end),
        });
    }

    if !self.tokens.eat(SyntaxKind::OpenBrace) {
        self.errors.push(ParseError::expected(SyntaxKind::OpenBrace, self.tokens.peek().map_or(SyntaxKind::Eof, |t| t.kind), self.tokens.peek().map_or(0, |t| t.start), self.tokens.peek().map_or(0, |t| t.end)));
        return None;
    }

    // Read the body as raw text, tracking brace depth
    let mut body = String::new();
    let mut brace_depth = 0u32;

    loop {
        match self.tokens.peek_raw() {
            None => {
                self.errors.push(ParseError::Message {
                    msg: "unexpected end of input in quote body".into(),
                    span: (name_start, name_start + 5),
                });
                break;
            }
            Some(tok) if tok.kind == SyntaxKind::OpenBrace => {
                brace_depth += 1;
                body.push_str(tok.text);
                self.tokens.bump_raw();
            }
            Some(tok) if tok.kind == SyntaxKind::CloseBrace => {
                if brace_depth > 0 {
                    brace_depth -= 1;
                    body.push_str(tok.text);
                    self.tokens.bump_raw();
                } else {
                    // Closing brace of quote body
                    self.tokens.bump_raw();
                    break;
                }
            }
            Some(tok) => {
                body.push_str(tok.text);
                self.tokens.bump_raw();
            }
        }
    }

    Some(ExprNode {
        kind: ExprKind::Quote { body },
        span: Span::new(name_start, self.tokens.peek().map_or(name_start + 5, |t| t.start)),
    })
}

use glyim_diag::{Diagnostic, Span};

/// Error produced by a macro during expansion.
/// Wraps a Diagnostic with macro-specific provenance.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct MacroError {
    /// The underlying diagnostic with spans, notes, and expansion trace.
    pub diagnostic: Diagnostic,
    /// The name of the macro that produced this error.
    pub macro_name: String,
}

impl MacroError {
    /// Create a macro error from a diagnostic and macro name.
    pub fn new(diagnostic: Diagnostic, macro_name: impl Into<String>) -> Self {
        Self { diagnostic, macro_name: macro_name.into() }
    }

    /// Create a simple macro error with a message and primary span.
    pub fn at_span(macro_name: impl Into<String>, span: Span, message: impl Into<String>) -> Self {
        Self {
            diagnostic: Diagnostic::error(message).with_span(span),
            macro_name: macro_name.into(),
        }
    }

    /// Create a macro error with multiple spans.
    pub fn with_labels(
        macro_name: impl Into<String>,
        primary_span: Span,
        primary_message: impl Into<String>,
        related: Vec<(Span, String)>,
    ) -> Self {
        let mut diag = Diagnostic::error(primary_message).with_label(primary_span, "");
        for (span, msg) in related {
            diag = diag.with_label(span, msg);
        }
        Self { diagnostic: diag, macro_name: macro_name.into() }
    }

    /// Convert to the underlying Diagnostic (for rendering).
    pub fn into_diagnostic(self) -> Diagnostic {
        self.diagnostic
    }

    /// Borrow the underlying Diagnostic.
    pub fn as_diagnostic(&self) -> &Diagnostic {
        &self.diagnostic
    }
}

    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

/// Tracks what a macro expansion accessed, for incremental invalidation.
#[derive(Debug, Clone, PartialEq, Eq, Default)]
pub struct MacroExpansionDependencies {
    pub types_accessed: HashSet<String>,
    pub fields_accessed: HashSet<(String, String)>,
    pub files_read: HashSet<String>,
    pub env_vars_read: HashSet<String>,
}

impl MacroExpansionDependencies {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn add_type_accessed(&mut self, type_name: String) {
        self.types_accessed.insert(type_name);
    }

    pub fn add_field_accessed(&mut self, type_name: String, field_name: String) {
        self.fields_accessed.insert((type_name, field_name));
    }

    pub fn add_file_read(&mut self, path: String) {
        self.files_read.insert(path);
    }

    pub fn add_env_var_read(&mut self, key: String) {
        self.env_vars_read.insert(key);
    }

    pub fn is_empty(&self) -> bool {
        self.types_accessed.is_empty()
            && self.fields_accessed.is_empty()
            && self.files_read.is_empty()
            && self.env_vars_read.is_empty()
    }
}

/// A single step in macro expansion, recorded by the stepper.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct MacroExpansionStep {
    pub invocation_id: MacroInvocationId,
    pub macro_name: String,
    /// Source text before this expansion.
    pub before: String,
    /// Source text after this expansion.
    pub after: String,
}

/// Cache for macro expansion results.
/// Implementations may be in-memory, on-disk, or use the distributed CAS.
pub trait MacroExpansionCache<Expr>: Send + Sync {
    /// Look up a cached expansion result.
    fn get(&self, invocation_id: &MacroInvocationId) -> Option<MacroOutput<Expr>>;

    /// Store an expansion result.
    fn put(&mut self, invocation_id: &MacroInvocationId, output: MacroOutput<Expr>);

    /// Invalidate a cached expansion.
    fn invalidate(&mut self, invocation_id: &MacroInvocationId);
}

/// Records expansion steps for the macro debugger.
pub trait MacroExpansionRecorder: Send + Sync {
    /// Record a single expansion step.
    fn record_step(&self, step: MacroExpansionStep);

    /// Get all recorded steps.
    fn steps(&self) -> Vec<MacroExpansionStep>;

    /// Reset the recorder (clear all steps).
    fn reset(&mut self);
}

use glyim_diag::Span;
use crate::attr::AttributeValue;
use crate::error::MacroError;

/// Parse a raw attribute value string into a typed `AttributeValue`.
///
/// Rules:
/// - `"hello"` â†’ String("hello")  (strips quotes)
/// - `'hello'` â†’ String("hello")  (strips single quotes)
/// - `42` â†’ Int(42)
/// - `-7` â†’ Int(-7)
/// - `true` â†’ Bool(true)
/// - `false` â†’ Bool(false)
/// - `std::fmt::Display` â†’ Path(["std", "fmt", "Display"])
/// - Anything else â†’ String (as-is, no stripping)
pub fn parse_attr_value(raw: &str, span: Span) -> Result<AttributeValue, AttrParseError> {
    let trimmed = raw.trim();

    // Quoted string
    if (trimmed.starts_with('"') && trimmed.ends_with('"') && trimmed.len() >= 2)
        || (trimmed.starts_with('\'') && trimmed.ends_with('\'') && trimmed.len() >= 2)
    {
        return Ok(AttributeValue::String(trimmed[1..trimmed.len()-1].to_string()));
    }

    // Boolean
    if trimmed == "true" { return Ok(AttributeValue::Bool(true)); }
    if trimmed == "false" { return Ok(AttributeValue::Bool(false)); }

    // Integer (possibly negative)
    if let Ok(n) = trimmed.parse::<i64>() {
        return Ok(AttributeValue::Int(n));
    }

    // Path (contains ::)
    if trimmed.contains("::") {
        let parts: Vec<String> = trimmed.split("::").map(|s| s.to_string()).collect();
        if parts.iter().all(|p| !p.is_empty()) {
            return Ok(AttributeValue::Path(parts));
        }
    }

    // Fallback: treat as bare string
    Ok(AttributeValue::String(trimmed.to_string()))
}

use crate::attr::{AttributeArg as CoreAttrArg, AttributeArgs};

/// Parse a list of raw attribute arguments into typed `AttributeArgs`.
///
/// Each input is a `(key, optional_raw_value, span)` triple. If `raw_value`
/// is `None`, the argument is a flag (treated as `key = true`).
/// If `raw_value` starts with `(` and ends with `)`, it's parsed as nested
/// attribute arguments.
///
/// # Errors
/// Returns `AttrParseError` if nested argument parsing fails (unbalanced
/// parentheses).
pub fn parse_attr_args(
    pairs: &[(&str, Option<&str>, Span)],
) -> Result<AttributeArgs, AttrParseError> {
    let mut args = Vec::with_capacity(pairs.len());
    for (key, raw_value, span) in pairs {
        if let Some(raw) = raw_value {
            let trimmed = raw.trim();
            if trimmed.starts_with('(') && trimmed.ends_with(')') && trimmed.len() >= 2 {
                // Nested arguments: (key = "val", flag, ...)
                let inner = &trimmed[1..trimmed.len()-1];
                let nested = parse_nested_args(inner, *span)?;
                args.push(CoreAttrArg::KeyValue {
                    key: key.to_string(),
                    value: AttributeValue::Nested(nested),
                    span: *span,
                });
            } else {
                let value = parse_attr_value(trimmed, *span)?;
                args.push(CoreAttrArg::KeyValue {
                    key: key.to_string(),
                    value,
                    span: *span,
                });
            }
        } else {
            args.push(CoreAttrArg::Flag {
                key: key.to_string(),
                span: *span,
            });
        }
    }
    Ok(AttributeArgs::new(args))
}

/// Parse the inside of a `(...)` nested attribute argument block.
///
/// Handles: `key = "val"`, `key = 42`, `flag`, `key = "a=b"`, etc.
/// Splits on commas, respecting quoted strings.
fn parse_nested_args(content: &str, outer_span: Span) -> Result<AttributeArgs, AttrParseError> {
    let mut pairs: Vec<(&str, Option<&str>, Span)> = vec![];
    let mut current = String::new();
    let mut in_quotes = false;
    let mut quote_char: char = '\0';
    let mut depth: u32 = 0;
    let mut start_offset = 0;

    for (i, ch) in content.char_indices() {
        if in_quotes {
            current.push(ch);
            if ch == quote_char {
                in_quotes = false;
            }
            continue;
        }
        match ch {
            '"' | '\'' => {
                in_quotes = true;
                quote_char = ch;
                current.push(ch);
            }
            '(' => {
                depth += 1;
                current.push(ch);
            }
            ')' => {
                if depth > 0 {
                    depth -= 1;
                    current.push(ch);
                } else {
                    return Err(AttrParseError::at_span(
                        outer_span,
                        "unbalanced parenthesis in nested attribute arguments",
                    ));
                }
            }
            ',' if depth == 0 => {
                flush_nested_token(&mut pairs, &mut current, outer_span, start_offset, i);
                start_offset = i + ch.len_utf8();
            }
            _ => {
                current.push(ch);
            }
        }
    }
    // Flush last token
    if !current.trim().is_empty() {
        flush_nested_token(&mut pairs, &mut current, outer_span, start_offset, content.len());
    }

    if depth != 0 {
        return Err(AttrParseError::at_span(
            outer_span,
            "unbalanced parenthesis in nested attribute arguments",
        ));
    }

    parse_attr_args(&pairs)
}

fn flush_nested_token(
    pairs: &mut Vec<(&str, Option<&str>, Span)>,
    current: &mut String,
    outer_span: Span,
    start_offset: usize,
    end_offset: usize,
) {
    let trimmed = current.trim();
    if trimmed.is_empty() {
        current.clear();
        return;
    }
    // Try to split on first '=' (not inside quotes)
    if let Some(eq_pos) = find_unquoted_eq(trimmed) {
        let key = trimmed[..eq_pos].trim();
        let val = trimmed[eq_pos + 1..].trim();
        // Compute an approximate inner span
        let inner_start = outer_span.start.saturating_add(start_offset).saturating_add(trimmed.len().saturating_sub(current.trim().len()));
        let inner_end = outer_span.start.saturating_add(end_offset);
        let inner_span = Span::new(inner_start, inner_end);
        pairs.push((key, Some(val), inner_span));
    } else {
        // Flag
        let inner_start = outer_span.start.saturating_add(start_offset);
        let inner_end = outer_span.start.saturating_add(end_offset);
        let inner_span = Span::new(inner_start, inner_end);
        pairs.push((trimmed, None, inner_span));
    }
    current.clear();
}

/// Find the first '=' that's not inside quotes.
fn find_unquoted_eq(s: &str) -> Option<usize> {
    let mut in_quotes = false;
    let mut quote_char: char = '\0';
    for (i, ch) in s.char_indices() {
        if in_quotes {
            if ch == quote_char { in_quotes = false; }
            continue;
        }
        if ch == '"' || ch == '\'' {
            in_quotes = true;
            quote_char = ch;
        } else if ch == '=' {
            return Some(i);
        }
    }
    None
}

    /// The derive name (e.g., "Debug", "Serialize").
    pub name: Symbol,

    pub description: String,

use glyim_interner::{Interner, Symbol};
use crate::registry::DeriveRegistry;

/// Result of a composability check.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ComposabilityResult {
    pub errors: Vec<ComposabilityError>,
    pub warnings: Vec<String>,
}

impl ComposabilityResult {
    pub fn ok() -> Self {
        Self { errors: vec![], warnings: vec![] }
    }

    pub fn is_ok(&self) -> bool {
        self.errors.is_empty()
    }

    pub fn error(msg: impl Into<String>) -> Self {
        Self { errors: vec![ComposabilityError::Other(msg.into())], warnings: vec![] }
    }
}

/// A composability error detected before macro expansion.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum ComposabilityError {
    /// A required derive is missing from the attribute list.
    MissingRequirement {
        derive: Symbol,
        requires: Symbol,
    },
    /// Circular requirement: A requires B requires A.
    CyclicRequirement {
        cycle: Vec<Symbol>,
    },
    /// Other error.
    Other(String),
}

impl std::fmt::Display for ComposabilityError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::MissingRequirement { derive, requires } => {
                write!(f, "derive `{}` requires `{}` to be derived first", requires, derive)
            }
            Self::CyclicRequirement { cycle } => {
                let names: Vec<_> = cycle.iter().map(|s| format!("{:?}", s)).collect();
                write!(f, "cyclic requirement: {}", names.join(" â†’ "))
            }
            Self::Other(msg) => write!(f, "{}", msg),
        }
    }
}

impl std::error::Error for ComposabilityError {}

/// Checks derive composability at the metadata level.
///
/// This does NOT expand macros. It checks:
/// 1. All required traits are present in the derive list
/// 2. No circular requirements exist in the derive graph
pub struct ComposabilityChecker;

impl ComposabilityChecker {
    /// Check a list of derives applied to a single type.
    ///
    /// `derives` is the ordered list of derive names (e.g., `["Debug", "Clone", "Eq"]`).
    /// `registry` provides the metadata for each derive.
    pub fn check(
        derives: &[Symbol],
        registry: &DeriveRegistry,
    ) -> ComposabilityResult {
        let mut result = ComposabilityResult::ok();

        // Build a set of available derives for quick lookup
        let available: std::collections::HashSet<Symbol> = derives.iter().copied().collect();

        // Check each derive's requirements
        for &derive_sym in derives {
            let entry = match registry.get(derive_sym) {
                Some(e) => e,
                None => continue, // Unknown derive â€” not our concern at metadata level
            };
            for &req in &entry.required_traits {
                if !available.contains(&req) {
                    result.errors.push(ComposabilityError::MissingRequirement {
                        derive: derive_sym,
                        requires: req,
                    });
                }
            }
        }

        // Check for cycles in the requirement graph
        if let Some(cycle) = Self::find_cycle(derives, registry) {
            result.errors.push(ComposabilityError::CyclicRequirement { cycle });
        }

        result
    }

    /// Find a cycle in the derive requirement graph using DFS.
    fn find_cycle(
        derives: &[Symbol],
        registry: &DeriveRegistry,
    ) -> Option<Vec<Symbol>> {
        let available: std::collections::HashSet<Symbol> = derives.iter().copied().collect();

        for &start in derives {
            let mut visiting = std::collections::HashSet::new();
            let mut path = vec![];
            if Self::dfs(start, &available, registry, &mut visiting, &mut path) {
                // Close the cycle
                let cycle_start = path.iter().position(|&s| s == start)?;
                return Some(path[cycle_start..].to_vec());
            }
        }
        None
    }

    fn dfs(
        current: Symbol,
        available: &std::collections::HashSet<Symbol>,
        registry: &DeriveRegistry,
        visiting: &mut std::collections::HashSet<Symbol>,
        path: &mut Vec<Symbol>,
    ) -> bool {
        if visiting.contains(&current) {
            path.push(current);
            return true; // cycle found
        }
        visiting.insert(current);
        path.push(current);

        if let Some(entry) = registry.get(current) {
            for &req in &entry.required_traits {
                if available.contains(&req) {
                    if Self::dfs(req, available, registry, visiting, path) {
                        return true;
                    }
                }
            }
        }

        path.pop();
        visiting.remove(&current);
        false
    }
}

use glyim_interner::{Interner, Symbol};
use std::collections::{HashMap, HashSet, VecDeque};

/// Error from topological sort.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum TopoSortError {
    /// The graph has a cycle. Contains the nodes in the cycle.
    Cycle(Vec<Symbol>),
}

impl std::fmt::Display for TopoSortError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result) {
        match self {
            Self::Cycle(nodes) => {
                let names: Vec<_> = nodes.iter().map(|s| format!("{:?}", s)).collect();
                write!(f, "dependency cycle: {}", names.join(" â†’ "))
            }
        }
    }
}

impl std::error::Error for TopoSortError {}

/// Result of topological sort.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct TopoSortResult {
    /// Nodes in valid topological order (dependencies before dependents).
    pub ordered: Vec<Symbol>,
    /// Warnings about sub-optimal ordering (e.g., a node could appear earlier).
    pub warnings: Vec<String>,
}

/// A directed graph of derive dependencies.
///
/// Nodes are derive names (Symbols). An edge A â†’ B means "A requires B".
pub struct DependencyGraph {
    /// adjacency list: node â†’ set of nodes it depends on
    deps: HashMap<Symbol, HashSet<Symbol>>,
    /// all nodes (including those with no dependencies)
    all_nodes: HashSet<Symbol>,
}

impl DependencyGraph {
    /// Create an empty graph.
    pub fn new() -> Self {
        Self { deps: HashMap::new(), all_nodes: HashSet::new() }
    }

    /// Add a node.
    pub fn add_node(&mut self, node: Symbol) {
        self.all_nodes.insert(node);
        self deps.entry(node).or_default();
    }

    /// Add a dependency edge: `node` depends on `depends_on`.
    /// Both nodes are added if not present.
    pub fn add_dependency(&mut self, node: Symbol, depends_on: Symbol) {
        self.all_nodes.insert(node);
        self.all_nodes.insert(depends_on);
        self.deps.entry(node).or_default().insert(depends_on);
    }

    /// Get the direct dependencies of a node.
    pub fn dependencies(&self, node: Symbol) -> Option<&HashSet<Symbol>> {
        self.deps.get(&node)
    }

    /// Get all nodes.
    pub fn nodes(&self) -> &HashSet<Symbol> {
        &self.all_nodes
    }

    /// Number of nodes.
    pub fn len(&self) -> usize {
        self.all_nodes.len()
    }

    /// Check if empty.
    pub fn is_empty(&self) -> bool {
        self.all_nodes.is_empty()
    }

    /// Topological sort using Kahn's algorithm.
    ///
    /// Returns nodes in an order where all dependencies appear before
    /// the nodes that depend on them.
    pub fn topological_sort(&self) -> Result<TopoSortResult, TopoSortError> {
        // Build in-degree map
        let mut in_degree: HashMap<Symbol, usize> = HashMap::new();
        for &node in &self.all_nodes {
            in_degree.insert(node, 0);
        }
        for (_, deps) in &self.deps {
            for &dep in deps {
                *in_degree.entry(dep).or_insert(0) += 0; // ensure dep exists
                *in_degree.entry(dep).or_insert(0) += 0;
            }
        }
        // Count in-degrees properly: for each edge nodeâ†’dep, increment dep's in-degree
        let mut in_degree: HashMap<Symbol, usize> = HashMap::new();
        for &node in &self.all_nodes {
            in_degree.insert(node, 0);
        }
        for (node, deps) in &self.deps {
            for &dep in deps {
                *in_degree.entry(dep).or_insert(0) += 1;
            }
        }

        // Start with nodes that have no incoming edges
        let mut queue: VecDeque<Symbol> = VecDeque::new();
        for (&node, &deg) in &in_degree {
            if deg == 0 {
                queue.push_back(node);
            }
        }

        let mut ordered = Vec::with_capacity(self.all_nodes.len());
        let mut warnings = Vec::new();
        let mut processed: HashSet<Symbol> = HashSet::new();

        while let Some(node) = queue.pop_front() {
            if processed.contains(&node) {
                continue;
            }
            processed.insert(node);
            ordered.push(node);

            // Decrement in-degree for dependents
            for (other, other_deps) in &self.deps {
                if other_deps.contains(&node) {
                    let deg = in_degree.get_mut(other).unwrap();
                    *deg -= 1;
                    if *deg == 0 {
                        queue.push_back(*other);
                    }
                }
            }
        }

        if ordered.len() != self.all_nodes.len() {
            // Find the cycle nodes
            let cycle_nodes: Vec<_> = self.all_nodes.iter()
                .filter(|n| !processed.contains(n))
                .copied()
                .collect();
            return Err(TopoSortError::Cycle(cycle_nodes));
        }

        Ok(TopoSortResult { ordered, warnings })
    }

    /// Check if the graph contains any cycles.
    pub fn has_cycle(&self) -> bool {
        self.topological_sort().is_err()
    }

    /// Build a graph from a registry and a list of derive symbols.
    pub fn from_registry_and_derives(
        registry: &crate::registry::DeriveRegistry,
        derives: &[Symbol],
    ) -> Self {
        let mut graph = Self::new();
        let available: HashSet<Symbol> = derives.iter().copied().collect();

        for &derive_sym in derives {
            graph.add_node(derive_sym);
            if let Some(entry) = registry.get(derive_sym) {
                for &req in &entry.required_traits {
                    if available.contains(&req) {
                        graph.add_dependency(derive_sym, req);
                    }
                }
            }
        }
        graph
    }
}

impl Default for DependencyGraph {
    fn default() -> Self { Self::new() }
}

use glyim_interner::Symbol;

/// The kind of conflict between impl blocks.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum ConflictKind {
    /// Two macros generate methods with the same name in the SAME trait impl.
    SameTraitMethod {
        trait_name: Symbol,
        method_name: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    },
    /// Two macros generate an impl for the same trait (this is actually OK per spec,
    /// but we track it for informational purposes).
    DuplicateTraitImpl {
        trait_name: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    },
    /// A method name collision across different trait impls (not an error per spec,
    /// but may indicate confusion).
    CrossTraitMethodShadowing {
        method_name: Symbol,
        first_trait: Symbol,
        second_trait: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    },
}

impl std::fmt::Display for ConflictKind {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::SameTraitMethod { trait_name, method_name, first_macro, second_macro } => {
                write!(f, "derives `{}` and `{}` both generate method `{}` in impl {}",
                    first_macro, second_macro, method_name, trait_name)
            }
            Self::DuplicateTraitImpl { trait_name, first_macro, second_macro } => {
                write!(f, "derives `{}` and `{}` both generate impl for {}",
                    first_macro, second_macro, trait_name)
            }
            Self::CrossTraitMethodShadowing { method_name, first_trait, second_trait, .. } => {
                write!(f, "method `{}` appears in both impl {} and impl {}",
                    method_name, first_trait, second_trait)
            }
        }
    }
}

/// A single conflict between two macro expansions.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ImplConflict {
    pub kind: ConflictKind,
    pub is_error: bool,
}

impl ImplConflict {
    pub fn same_trait_method(
        trait_name: Symbol,
        method_name: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    ) -> Self {
        Self {
            kind: ConflictKind::SameTraitMethod { trait_name, method_name, first_macro, second_macro },
            is_error: true,
        }
    }

    pub fn duplicate_trait_impl(
        trait_name: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    ) -> Self {
        Self {
            kind: ConflictKind::DuplicateTraitImpl { trait_name, first_macro, second_macro },
            is_error: false,
        }
    }

    pub fn cross_trait_shadowing(
        method_name: Symbol,
        first_trait: Symbol,
        second_trait: Symbol,
        first_macro: Symbol,
        second_macro: Symbol,
    ) -> Self {
        Self {
            kind: ConflictKind::CrossTraitMethodShadowing {
                method_name, first_trait, second_trait, first_macro, second_macro,
            },
            is_error: false,
        }
    }
}

/// A report of all conflicts found across macro expansions.
#[derive(Debug, Clone, PartialEq, Eq, Default)]
pub struct ConflictReport {
    pub conflicts: Vec<ImplConflict>,
}

impl ConflictReport {
    pub fn new() -> Self { Self::default() }

    pub fn add(&mut self, conflict: ImplConflict) {
        self.conflicts.push(conflict);
    }

    pub fn is_ok(&self) -> bool {
        self.conflicts.iter().all(|c| !c.is_error)
    }

    pub fn has_errors(&self) -> bool {
        self.conflicts.iter().any(|c| c.is_error)
    }

    pub fn errors(&self) -> impl Iterator<Item = &ImplConflict> {
        self.conflicts.iter().filter(|c| c.is_error)
    }

    pub fn warnings(&self) -> impl Iterator<Item = &ImplConflict> {
        self.conflicts.iter().filter(|c| !c.is_error)
    }

    pub fn len(&self) -> usize { self.conflicts.len() }
    pub fn is_empty(&self) -> bool { self.conflicts.is_empty() }
}

/// Checks for method name conflicts between trait impls.
pub struct MethodConflictChecker;

impl MethodConflictChecker {
    /// Check two method sets for same-trait conflicts.
    ///
    /// `first_trait` and `second_trait` are the trait names.
    /// `first_methods` and `second_methods` are the method name sets.
    /// `first_macro` and `second_macro` are the macro names that generated them.
    pub fn check_pair(
        first_trait: Symbol,
        first_methods: &HashSet<Symbol>,
        first_macro: Symbol,
        second_trait: Symbol,
        second_methods: &HashSet<Symbol>,
        second_macro: Symbol,
    ) -> ConflictReport {
        let mut report = ConflictReport::new();

        if first_trait == second_trait {
            // Same trait â€” any overlapping method is an ERROR
            for &method in first_methods {
                if second_methods.contains(&method) {
                    report.add(ImplConflict::same_trait_method(
                        first_trait, method, first_macro, second_macro,
                    ));
                }
            }
            if !first_methods.is_empty() || !second_methods.is_empty() {
                report.add(ImplConflict::duplicate_trait_impl(
                    first_trait, first_macro, second_macro,
                ));
            }
        } else {
            // Different traits â€” overlapping method names are warnings (not errors)
            for &method in first_methods {
                if second_methods.contains(&method) {
                    report.add(ImplConflict::cross_trait_shadowing(
                        method, first_trait, second_trait, first_macro, second_macro,
                    ));
                }
            }
        }

        report
    }

    /// Check multiple impls against each other (all pairs).
    pub fn check_all(
        impls: &[(Symbol, HashSet<Symbol>, Symbol)], // (trait_name, methods, macro_name)
    ) -> ConflictReport {
        let mut report = ConflictReport::new();
        for i in 0..impls.len() {
            for j in (i+1)..impls.len() {
                let (t1, m1, macro1) = &impls[i];
                let (t2, m2, macro2) = &impls[j];
                let pair_report = Self::check_pair(*t1, m1, *macro1, *t2, m2, *macro2);
                for conflict in pair_report.conflicts {
                    report.add(conflict);
                }
            }
        }
        report
    }
}

use glyim_interner::Symbol;
use crate::dep_graph::DependencyGraph;

/// A single step in the expansion plan.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ExpansionStep {
    /// The derive macro to expand.
    pub derive: Symbol,
    /// 0-based index indicating expansion order.
    pub order: usize,
    /// Whether this derive has any unsatisfied requirements (missing from the derive list).
    pub has_missing_requirements: bool,
}

/// An ordered plan for macro expansion.
///
/// Computed by topologically sorting derive dependencies.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ExpansionPlan {
    pub steps: Vec<ExpansionStep>,
}

impl ExpansionPlan {
    /// Create an expansion plan from a dependency graph and a set of derives.
    ///
    /// Returns `None` if the dependency graph has cycles.
    pub fn from_graph(
        graph: &DependencyGraph,
        derives: &[Symbol],
        missing: &std::collections::HashSet<Symbol>,
    ) -> Option<Self> {
        let sort_result = graph.topological_sort().ok()?;

        // Build steps in topological order, filtering to only the requested derives
        let ordered_set: std::collections::HashSet<Symbol> = sort_result.ordered.iter().copied().collect();
        let steps: Vec<ExpansionStep> = derives
            .iter()
            .filter(|d| ordered_set.contains(d))
            .enumerate()
            .map(|(order, &derive)| ExpansionStep {
                derive,
                order,
                has_missing_requirements: missing.contains(&derive),
            })
            .collect();

        Some(Self { steps })
    }

    /// Create a trivial plan (no dependencies, all derives expand in listed order).
    pub fn trivial(derives: &[Symbol]) -> Self {
        let steps = derives
            .iter()
            .enumerate()
            .map(|(order, &derive)| ExpansionStep {
                derive,
                order,
                has_missing_requirements: false,
            })
            .collect();
        Self { steps }
    }

    /// Number of expansion steps.
    pub fn len(&self) -> usize {
        self.steps.len()
    }

    /// Check if empty.
    pub fn is_empty(&self) -> bool {
        self.steps.is_empty()
    }

    /// Iterate over steps in expansion order.
    pub fn iter(&self) -> impl Iterator<Item = &ExpansionStep> {
        self.steps.iter()
    }

    /// Check if any step has missing requirements.
    pub fn has_missing_requirements(&self) -> bool {

    pub fn is_empty(&self) -> bool { self.steps.borrow().is_empty() }
    pub fn clear(&self) { self.steps.borrow_mut().clear(); }
}

impl MacroExpansionRecorder for InMemoryExpansionRecorder {
    fn record_step(&self, step: MacroExpansionStep) {
        self.steps.borrow_mut().push(step);
    }

    fn steps(&self) -> Vec<MacroExpansionStep> {
        self.steps.borrow().clone()
    }

    fn reset(&mut self) {
        self.steps.get_mut().clear();
    }
}

impl Default for InMemoryExpansionRecorder {
    fn default() -> Self { Self::new() }
}

/// A cache key computed from macro identity, input, and dependencies.
///
/// In production (v0.5.0+), this uses SHA-256. For now, we use a
/// deterministic hash via `std::hash::Hasher` with version tagging.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub struct CacheKey(pub u64);

impl CacheKey {
    /// Compute a cache key from the components.
    ///
    /// `macro_version_tag` identifies the macro definition version.
    /// `input_hash` identifies the macro input (AST of the annotated item).
    /// `deps` identifies what the macro accessed (for invalidation).
    pub fn compute(
        macro_version_tag: u64,
        input_hash: u64,
        deps: &MacroExpansionDependencies,
        compiler_version_tag: u64,
    ) -> Self {
        use std::hash::{Hash, Hasher};
        let mut h = std::collections::hash_map::DefaultHasher::new();
        h.write_u64(compiler_version_tag);
        h.write_u64(macro_version_tag);
        h.write_u64(input_hash);
        // Hash dependency fields in sorted order for determinism
        let mut types: Vec<_> = deps.types_accessed.iter().collect();
        types.sort();
        for t in &types {
            h.write(t.as_bytes());
        }
        let mut fields: Vec<_> = deps.fields_accessed.iter().collect();
        fields.sort();
        for (type_name, field_name) in &fields {
            h.write(type_name.as_bytes());
            h.write_u8(b'.');
            h.write(field_name.as_bytes());
        }
        let mut files: Vec<_> = deps.files_read.iter().collect();
        files.sort();
        for f in &files {
            h.write(f.as_bytes());
        }
        let mut envs: Vec<_> = deps.env_vars_read.iter().collect();
        envs.sort();
        for e in &envs {
            h.write(e.as_bytes());
        }
        Self(h.finish())
    }

    /// Create a zero key (for testing).
    pub fn zero() -> Self { Self(0) }
}

/// Result of an invalidation pass.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct InvalidationResult {
    /// IDs of cache entries to evict.
    pub evict: HashSet<MacroInvocationId>,
    /// IDs of cache entries that survived (dependencies unchanged).
    pub kept: HashSet<MacroInvocationId>,
}

/// Determines which cache entries to invalidate given changed dependencies.
pub struct CacheInvalidator;

impl CacheInvalidator {
    /// Given a set of cache entries (id â†’ their dependencies) and a set of
    /// changed dependency keys, determine which entries to evict.
    ///
    /// `changed_types`: type names that changed (struct fields modified, etc.)
    /// `changed_fields`: `(type_name, field_name)` pairs that changed.
    /// `changed_files`: file paths whose content changed.
    /// `changed_env_vars`: env var names whose values changed.
    pub fn invalidate_for_dep_changes(
        entries: &HashMap<MacroInvocationId, MacroExpansionDependencies>,
        changed_types: &HashSet<&str>,
        changed_fields: &HashSet<(&str, &str)>,
        changed_files: &HashSet<&str>,
        changed_env_vars: &HashSet<&str>,
    ) -> InvalidationResult {
        let mut evict = HashSet::new();
        let mut kept = HashSet::new();

        for (&id, deps) in entries {
            let mut should_evict = false;

            if deps.types_accessed.iter().any(|t| changed_types.contains(t.as_str())) {
                should_evict = true;
            }
            if deps.fields_accessed.iter().any(|(t, f)| changed_fields.contains(&(t.as_str(), f.as_str()))) {
                should_evict = true;
            }
            if deps.files_read.iter().any(|f| changed_files.contains(f.as_str())) {
                should_evict = true;
            }
            if deps.env_vars_read.iter().any(|e| changed_env_vars.contains(e.as_str())) {
                should_evict = true;
            }

            if should_evict {
                evict.insert(id);
            } else {
                kept.insert(id);
            }
        }

        InvalidationResult { evict, kept }
    }

    /// Invalidate all entries (full rebuild).
    pub fn invalidate_all(entries: &HashMap<MacroInvocationId, MacroExpansionDependencies>) -> InvalidationResult {
        InvalidationResult {
            evict: entries.keys().copied().collect(),
            kept: HashSet::new(),
        }
    }
}

/// Statistics from an incremental expansion pass.
#[derive(Debug, Clone, PartialEq, Eq, Default)]
pub struct CacheStats {
    pub cache_hits: usize,
    pub cache_misses: usize,
    pub evictions: usize,
}

impl CacheStats {
    pub fn total_requests(&self) -> usize {
        self.cache_hits + self.cache_misses
    }

    pub fn hit_rate(&self) -> f64 {
        let total = self.total_requests();
        if total == 0 { 0.0 } else { self.cache_hits as f64 / total as f64 }
    }

    pub fn is_empty(&self) -> bool {
        self.total_requests() == 0
    }
}

use crate::cache::{MacroExpansionCache, MacroExpansionDependencies, MacroExpansionStep};
use crate::cache_impls::{InMemoryExpansionCache, InMemoryExpansionRecorderMut};
use crate::id::MacroInvocationId;

/// Orchestrates incremental macro expansion with caching and recording.
///
/// Generic over the expression type to keep `glyim-macro-core` independent of `glyim-hir`.
///
/// Usage (once macro expansion exists):
/// ```ignore
/// let result = engine.expand_or_cache(
///     id,
///     macro_name,
///     &deps,
///     || expand_macro(ctx, input),
/// );
/// ```
pub struct IncrementalEngine<Expr> {
    cache: InMemoryExpansionCache<Expr>,
    recorder: InMemoryRecorderMut,
    stats: CacheStats,
}

impl<Expr: Clone + Send + Sync + 'static> IncrementalEngine<Expr> {
    /// Create a new engine with empty cache and recorder.
    pub fn new() -> Self {
        Self {
            cache: InMemoryExpansionCache::new(),
            recorder: InMemoryRecorderMut::new(),
            stats: CacheStats::default(),
        }
    }

    /// Try to get a cached result, or expand and cache it.
    ///
    /// `expand_fn` is called only on cache miss. If `expand_fn` returns `Err`,
    /// the error is propagated and the result is NOT cached.
    pub fn expand_or_cache(
        &mut self,
        id: MacroInvocationId,
        macro_name: &str,
        deps: &MacroExpansionDependencies,
        expand_fn: impl FnOnce() -> Result<Expr, String>,
    ) -> Result<Expr, String> {
        self.stats.cache_misses += 1;

        if let Some(cached) = self.cache.get(&id) {
            self.stats.cache_misses -= 1; // Correct: it was a miss check, not a miss
            self.stats.cache_hits += 1;
            return Ok(cached.into_expr());
        }

        let result = expand_fn()?;
        self.cache.put(&id, MacroOutput::from_expr(result.clone()));
        self.recorder.record_step(MacroExpansionStep {
            invocation_id: id,
            macro_name: macro_name.to_string(),
            before: String::new(), // TODO: populate from actual input
            after: String::new(),  // TODO: populate from actual output
        });
        Ok(result)
    }

    /// Invalidate cache entries by ID.
    pub fn invalidate(&mut self, ids: &[MacroInvocationId]) {
        for id in ids {
            self.cache.invalidate(id);
        }
    }

    /// Clear all cache entries and stats.
    pub fn clear(&mut self) {
        self.cache.clear();
        self.recorder.reset();
        self.stats = CacheStats::default();
    }

    /// Number of cached entries.
    pub fn cache_len(&self) -> usize {
        self.cache.len()
    }

    /// Get expansion statistics.
    pub fn stats(&self) -> &CacheStats {
        &self.stats
    }

    /// Number of recorded expansion steps.
    pub fn recorder_len(&self) -> usize {
        self.recorder.len()
    }

    /// Take all recorded steps and reset the recorder.
    pub fn take_steps(&mut self) -> Vec<MacroExpansionStep> {
        let steps = self.recorder.steps();
        self.recorder.reset();
        steps
    }

    /// Get a reference to the underlying cache (for advanced use).
    pub fn cache(&self) -> &InMemoryExpansionCache<Expr> {
        &self.cache
    }

    /// Get a mutable reference to the underlying cache (for advanced use).
    pub fn cache_mut(&mut self) -> &mut InMemoryExpansionCache<Expr> {
        &mut self.cache
    }

    /// Get cache entry raw (bypasses cache-check logic).
    pub fn get_cache_entry(&self, id: &MacroInvocationId) -> Option<&crate::output::MacroOutput<Expr>> {
        self.cache.get_raw(id)
    }
}

impl<Expr: Clone + Send + Sync + 'static> Default for IncrementalEngine<Expr> {
    fn default() -> Self { Self::new() }
}
