> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Scope check:** The v0.6.0 spec covers 7 phases across ~8 subsystems (quote/splice, macro stepper, rich errors, built-in derives, composability protocol, incremental expansion, attribute arguments). These MUST be separate plans. The codebase is at v0.1.0 level — no type checker, no structs, no enums, no generics, no macro expansion pipeline (`glyim-macro-core/src/lib.rs` is `//! TODO`). Every v0.6.0 phase depends on v0.3.0/v0.4.0/v0.5.0 landing first. This plan covers the only subset achievable at v0.1.0: multi-span diagnostic infrastructure, HirBuilder fluent API, and `quote! { ... }` syntax parsing. These three are independent of missing features and form the foundation that ALL subsequent v0.6.0 phases depend on.

**Blocked phases and what unblocks them:**
| Phase | Blocker |
|-------|---------|

**Goal:** Add multi-span diagnostic support to `glyim-diag`, a fluent `HirBuilder` API to `glyim-hir`, and `quote! { ... }` syntax recognition to `glyim-parse` — all purely additive with zero behavior change to existing compilation pipeline.

**Architecture:** `Diagnostic` gains `labels`, `notes`, and `expansion_trace` fields with full backward compatibility. `HirBuilder` provides ergonomic constructors for all existing `HirExpr`/`HirStmt` variants (the target that `quote!` expansion will generate calls to). use crate::Span;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum Severity { Error, Warning, Note }

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
pub enum LabelStyle {
    #[default]
    Primary,
    Secondary,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct DiagnosticLabel {
    pub span: Span,
    pub message: String,
    pub style: LabelStyle,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ExpansionFrame {
    pub macro_name: String,
    pub location: Option<Span>,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Diagnostic {
    pub severity: Severity,
    pub message: String,
    pub span: Option<Span>,
    pub labels: Vec<DiagnosticLabel>,
    pub notes: Vec<String>,
    pub expansion_trace: Vec<ExpansionFrame>,
}

impl Diagnostic {
    pub fn error(message: impl Into<String>) -> Self {
        Self { severity: Severity::Error, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn warning(message: impl Into<String>) -> Self {
        Self { severity: Severity::Warning, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn note(message: impl Into<String>) -> Self {
        Self { severity: Severity::Note, message: message.into(), span: None, labels: vec![], notes: vec![], expansion_trace: vec![] }
    }
    pub fn with_span(mut self, span: Span) -> Self { self.span = Some(span); self }
    pub fn with_span_opt(mut self, span: Option<Span>) -> Self { self.span = span; self }
    pub fn is_error(&self) -> bool { self.severity == Severity::Error }

    /// Add a primary label at the given span.
    pub fn with_label(mut self, span: Span, message: impl Into<String>) -> Self {
        self.labels.push(DiagnosticLabel { span, message: message.into(), style: LabelStyle::Primary });
        self
    }

    /// Add a label with explicit style (primary or secondary).
    pub fn with_label_styled(mut self, span: Span, style: LabelStyle, message: impl Into<String>) -> Self {
        self.labels.push(DiagnosticLabel { span, message: message.into(), style });
        self
    }

    /// Add a note below the diagnostic.
    pub fn with_note(mut self, message: impl Into<String>) -> Self {
        self.notes.push(message.into());
        self
    }

    /// Add a macro expansion frame to the trace.
    pub fn with_expansion_frame(mut self, macro_name: impl Into<String>, location: Option<Span>) -> Self {
        self.expansion_trace.push(ExpansionFrame { macro_name: macro_name.into(), location });
        self
    }
}

use crate::{Diagnostic, ExpansionFrame, LabelStyle, Severity};
use ariadne::{Label, Report, ReportKind, Source};

pub fn render_diagnostics(source: &str, file_path: &str, diagnostics: &[Diagnostic]) -> String {
    let mut out = Vec::new();
    let cache = Source::from(source);
    for diag in diagnostics {
        let rendered = render_one(source, file_path, diag, &cache);
        out.extend(rendered);
    }
    String::from_utf8(out).unwrap_or_default()
}

pub fn render_single(source: &str, file_path: &str, diag: &Diagnostic) -> String {
    let cache = Source::from(source);
    let bytes = render_one(source, file_path, diag, &cache);
    String::from_utf8(bytes).unwrap_or_default()
}

fn render_one(source: &str, file_path: &str, diag: &Diagnostic, cache: &Source) -> Vec<u8> {
    let mut out = Vec::new();
    let kind = match diag.severity {
        Severity::Error => ReportKind::Error,
        Severity::Warning => ReportKind::Warning,
        Severity::Note => ReportKind::Advice,
    };

    // Determine the primary span for the report header
    let primary_span = if !diag.labels.is_empty() {
        diag.labels.iter().find(|l| l.style == LabelStyle::Primary)
            .or_else(|| diag.labels.first())
            .map(|l| l.span.start..l.span.end)
    } else {
        diag.span.map(|s| s.start..s.end)
    };
    let header_span = primary_span.unwrap_or(0..0);

    let mut report = Report::build(kind, (file_path, header_span)).with_message(&diag.message);

    // Add labels
    if !diag.labels.is_empty() {
        for label in &diag.labels {
            let ariadne_style = match label.style {
                LabelStyle::Primary => ariadne::LabelStyle::Primary,
                LabelStyle::Secondary => ariadne::LabelStyle::Secondary,
            };
            if label.span.start != label.span.end {
                report = report.with_label(
                    Label::new((file_path, label.span.start..label.span.end))
                        .with_message(&label.message)
                        .with_style(ariadne_style),
                );
            }
        }
    } else if let Some(span) = diag.span {
        // Backward compat: single span, no labels
        if span.start != span.end {
            report = report.with_label(
                Label::new((file_path, span.start..span.end)).with_message(&diag.message),
            );
        }
    }

    // Add notes
    for note_text in &diag.notes {
        report = report.with_note(note_text);
    }

    // Add expansion trace as notes
    if !diag.expansion_trace.is_empty() {
        let mut trace_lines = Vec::new();
        for frame in &diag.expansion_trace {
            let loc = frame.location
                .map(|s| format!(":{}", s.start))
                .unwrap_or_default();
            trace_lines.push(format!("  {}{}", frame.macro_name, loc));
        }
        report = report.with_note(format!("expansion trace:\n{}", trace_lines.join("\n")));
    }

    report
        .finish()
        .write((file_path, cache.clone()), &mut out)
        .unwrap();
    out
}

// crates/glyim-hir/src/hir_builder.rs
use glyim_interner::{Interner, Symbol};
use crate::{HirBinOp, HirExpr, HirStmt, HirUnOp};

/// Fluent builder for constructing HIR nodes.
/// This is the target API that `quote!` expansion will generate calls to.
pub struct HirBuilder;

impl HirBuilder {
    // ── Literals ────────────────────────────────────────

    pub fn int_lit(n: i64) -> HirExpr {
        HirExpr::IntLit(n)
    }

    pub fn str_lit(s: impl Into<String>) -> HirExpr {
        HirExpr::StrLit(s.into())
    }

    pub fn ident(sym: Symbol) -> HirExpr {
        HirExpr::Ident(sym)
    }

    pub fn ident_str(interner: &mut Interner, name: &str) -> HirExpr {
        HirExpr::Ident(interner.intern(name))
    }

    // ── Operators ───────────────────────────────────────

    pub fn binary(op: HirBinOp, lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        HirExpr::Binary { op, lhs: Box::new(lhs), rhs: Box::new(rhs) }
    }

    pub fn add(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Add, lhs, rhs)
    }

    pub fn sub(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Sub, lhs, rhs)
    }

    pub fn mul(lhs: HirExpr, rhs: HirExpr) -> HirExpr {
        Self::binary(HirBinOp::Mul, lhs, rhs)
    }

    pub fn neg(operand: HirExpr) -> HirExpr {
        HirExpr::Unary { op: HirUnOp::Neg, operand: Box::new(operand) }
    }

    pub fn not(operand: HirExpr) -> HirExpr {
        HirExpr::Unary { op: HirUnOp::Not, operand: Box::new(operand) }
    }

    // ── Control flow ────────────────────────────────────

    pub fn block(stmts: Vec<HirStmt>) -> HirExpr {
        HirExpr::Block(stmts)
    }

    pub fn block_expr(exprs: Vec<HirExpr>) -> HirExpr {
        let stmts: Vec<HirStmt> = exprs.into_iter().map(HirStmt::Expr).collect();
        HirExpr::Block(stmts)
    }

    pub fn if_expr(condition: HirExpr, then_branch: HirExpr, else_branch: Option<HirExpr>) -> HirExpr {
        HirExpr::If {
            condition: Box::new(condition),
            then_branch: Box::new(then_branch),
            else_branch: else_branch.map(Box::new),
        }
    }

    // ── Builtins ────────────────────────────────────────

    pub fn println(arg: HirExpr) -> HirExpr {
        HirExpr::Println(Box::new(arg))
    }

    pub fn assert(condition: HirExpr, message: Option<HirExpr>) -> HirExpr {
        HirExpr::Assert { condition: Box::new(condition), message: message.map(Box::new) }
    }

    // ── Statements ──────────────────────────────────────

    pub fn let_stmt(name: Symbol, mutable: bool, value: HirExpr) -> HirStmt {
        HirStmt::Let { name, mutable, value }
    }

    pub fn let_immutable(name: Symbol, value: HirExpr) -> HirStmt {
        Self::let_stmt(name, false, value)
    }

    pub fn let_mutable(name: Symbol, value: HirExpr) -> HirStmt {
        Self::let_stmt(name, true, value)
    }

fn parse_quote_expr(&mut self) -> Option<ExprNode> {
    // We're positioned at Ident("quote")
    let name_tok = self.tokens.bump()?; // consume "quote"
    let name_start = name_tok.start;

    if !self.tokens.eat(SyntaxKind::Bang) {
        // Not a macro invocation — this shouldn't be reached since we check in parse_expr_atom
        // But handle gracefully: treat "quote" as an ident
        let sym = self.interner.intern(name_tok.text);
        return Some(ExprNode {
            kind: ExprKind::Ident(sym),
            span: Span::new(name_start, name_tok.end),
        });
    }

    if !self.tokens.eat(SyntaxKind::OpenBrace) {
        self.errors.push(ParseError::expected(SyntaxKind::OpenBrace, self.tokens.peek().map_or(SyntaxKind::Eof, |t| t.kind), self.tokens.peek().map_or(0, |t| t.start), self.tokens.peek().map_or(0, |t| t.end)));
        return None;
    }

    // Read the body as raw text, tracking brace depth
    let mut body = String::new();
    let mut brace_depth = 0u32;

    loop {
        match self.tokens.peek_raw() {
            None => {
                self.errors.push(ParseError::Message {
                    msg: "unexpected end of input in quote body".into(),
                    span: (name_start, name_start + 5),
                });
                break;
            }
            Some(tok) if tok.kind == SyntaxKind::OpenBrace => {
                brace_depth += 1;
                body.push_str(tok.text);
                self.tokens.bump_raw();
            }
            Some(tok) if tok.kind == SyntaxKind::CloseBrace => {
                if brace_depth > 0 {
                    brace_depth -= 1;
                    body.push_str(tok.text);
                    self.tokens.bump_raw();
                } else {
                    // Closing brace of quote body
                    self.tokens.bump_raw();
                    break;
                }
            }
            Some(tok) => {
                body.push_str(tok.text);
                self.tokens.bump_raw();
            }
        }
    }

    Some(ExprNode {
        kind: ExprKind::Quote { body },
        span: Span::new(name_start, self.tokens.peek().map_or(name_start + 5, |t| t.start)),
    })
}

use glyim_diag::{Diagnostic, Span};

/// Error produced by a macro during expansion.
/// Wraps a Diagnostic with macro-specific provenance.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct MacroError {
    /// The underlying diagnostic with spans, notes, and expansion trace.
    pub diagnostic: Diagnostic,
    /// The name of the macro that produced this error.
    pub macro_name: String,
}

impl MacroError {
    /// Create a macro error from a diagnostic and macro name.
    pub fn new(diagnostic: Diagnostic, macro_name: impl Into<String>) -> Self {
        Self { diagnostic, macro_name: macro_name.into() }
    }

    /// Create a simple macro error with a message and primary span.
    pub fn at_span(macro_name: impl Into<String>, span: Span, message: impl Into<String>) -> Self {
        Self {
            diagnostic: Diagnostic::error(message).with_span(span),
            macro_name: macro_name.into(),
        }
    }

    /// Create a macro error with multiple spans.
    pub fn with_labels(
        macro_name: impl Into<String>,
        primary_span: Span,
        primary_message: impl Into<String>,
        related: Vec<(Span, String)>,
    ) -> Self {
        let mut diag = Diagnostic::error(primary_message).with_label(primary_span, "");
        for (span, msg) in related {
            diag = diag.with_label(span, msg);
        }
        Self { diagnostic: diag, macro_name: macro_name.into() }
    }

    /// Convert to the underlying Diagnostic (for rendering).
    pub fn into_diagnostic(self) -> Diagnostic {
        self.diagnostic
    }

    /// Borrow the underlying Diagnostic.
    pub fn as_diagnostic(&self) -> &Diagnostic {
        &self.diagnostic
    }
}

impl std::fmt::Display for MacroError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "macro `{}` error: {}", self.macro_name, self.diagnostic.message)
    }
}

impl std::error::Error for MacroError {}

/// The result of a macro expansion.

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct MacroOutput<Expr> {
    /// The expanded expression (or block of expressions).
    pub expr: Expr,
    /// Dependencies tracked during this expansion, for incremental invalidation.
    pub dependencies: MacroExpansionDependencies,
}

impl<Expr> MacroOutput<Expr> {
    /// Create output from just an expression (no tracked dependencies).
    pub fn from_expr(expr: Expr) -> Self {
        Self { expr, dependencies: MacroExpansionDependencies::new() }

        Self { expr, dependencies }

        self.expr

    pub fn as_expr(&self) -> &Expr {
        &self.expr

use std::collections::HashSet;

/// Tracks what a macro expansion accessed, for incremental invalidation.
#[derive(Debug, Clone, PartialEq, Eq, Default)]
pub struct MacroExpansionDependencies {
    pub types_accessed: HashSet<String>,
    pub fields_accessed: HashSet<(String, String)>,
    pub files_read: HashSet<String>,
    pub env_vars_read: HashSet<String>,
}

impl MacroExpansionDependencies {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn add_type_accessed(&mut self, type_name: String) {
        self.types_accessed.insert(type_name);
    }

    pub fn add_field_accessed(&mut self, type_name: String, field_name: String) {
        self.fields_accessed.insert((type_name, field_name));
    }

    pub fn add_file_read(&mut self, path: String) {
        self.files_read.insert(path);
    }

    pub fn add_env_var_read(&mut self, key: String) {
        self.env_vars_read.insert(key);
    }

    pub fn is_empty(&self) -> bool {
        self.types_accessed.is_empty()
            && self.fields_accessed.is_empty()
            && self.files_read.is_empty()
            && self.env_vars_read.is_empty()
    }
}

/// Tracks what a macro expansion accessed, for incremental invalidation.
#[derive(Debug, Clone, PartialEq, Eq, Default)]
pub struct MacroExpansionDependencies {
    pub types_accessed: HashSet<String>,

    pub files_read: HashSet<String>,
    pub env_vars_read: HashSet<String>,
}

        Self::default()
    }

        self.types_accessed.insert(type_name);
    }

    pub fn add_field_accessed(&mut self, type_name: String, field_name: String) {
        self.fields_accessed.insert((type_name, field_name));
    }

    pub fn add_file_read(&mut self, path: String) {
        self.files_read.insert(path);
    }

    pub fn add_env_var_read(&mut self, key: String) {
        self.env_vars_read.insert(key);
    }

        self.types_accessed.is_empty()
            && self.fields_accessed.is_empty()
            && self.files_read.is_empty()
            && self.env_vars_read.is_empty()
    }
}

/// Parse a raw attribute value string into a typed `AttributeValue`.
///
/// Rules:
/// - `"hello"` → String("hello")  (strips quotes)
/// - `'hello'` → String("hello")  (strips single quotes)
/// - `42` → Int(42)
/// - `-7` → Int(-7)
/// - `true` → Bool(true)
/// - `false` → Bool(false)
/// - `std::fmt::Display` → Path(["std", "fmt", "Display"])
/// - Anything else → String (as-is, no stripping)
pub fn parse_attr_value(raw: &str, span: Span) -> Result<AttributeValue, AttrParseError> {
    let trimmed = raw.trim();

    // Quoted string
    if (trimmed.starts_with('"') && trimmed.ends_with('"') && trimmed.len() >= 2)
        || (trimmed.starts_with('\'') && trimmed.ends_with('\'') && trimmed.len() >= 2)
    {
        return Ok(AttributeValue::String(trimmed[1..trimmed.len()-1].to_string()));
    }

    // Boolean
    if trimmed == "true" { return Ok(AttributeValue::Bool(true)); }
    if trimmed == "false" { return Ok(AttributeValue::Bool(false)); }

    // Integer (possibly negative)
    if let Ok(n) = trimmed.parse::<i64>() {
        return Ok(AttributeValue::Int(n));
    }

    // Path (contains ::)
    if trimmed.contains("::") {
        let parts: Vec<String> = trimmed.split("::").map(|s| s.to_string()).collect();
        if parts.iter().all(|p| !p.is_empty()) {
            return Ok(AttributeValue::Path(parts));
        }
    }

    // Fallback: treat as bare string
    Ok(AttributeValue::String(trimmed.to_string()))
}

/// Error from attribute argument parsing.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct AttrParseError {
    /// The underlying MacroError for rendering.
    pub error: MacroError,
}

impl AttrParseError {
    pub fn at_span(span: Span, message: impl Into<String>) -> Self {
        Self { error: MacroError::at_span("attr_parse", span, message) }
    }

    pub fn into_macro_error(self) -> MacroError {
        self.error
    }

    pub fn as_macro_error(&self) -> &MacroError {
        &self.error
    }
}

impl std::fmt::Display for AttrParseError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "{}", self.error)
    }
}

use crate::attr::{AttributeArg as CoreAttrArg, AttributeArgs};

/// Parse a list of raw attribute arguments into typed `AttributeArgs`.
///
/// Each input is a `(key, optional_raw_value, span)` triple. If `raw_value`
/// is `None`, the argument is a flag (treated as `key = true`).
/// If `raw_value` starts with `(` and ends with `)`, it's parsed as nested
/// attribute arguments.
///
/// # Errors
/// Returns `AttrParseError` if nested argument parsing fails (unbalanced
/// parentheses).
pub fn parse_attr_args(
    pairs: &[(&str, Option<&str>, Span)],
) -> Result<AttributeArgs, AttrParseError> {
    let mut args = Vec::with_capacity(pairs.len());
    for (key, raw_value, span) in pairs {
        if let Some(raw) = raw_value {
            let trimmed = raw.trim();
            if trimmed.starts_with('(') && trimmed.ends_with(')') && trimmed.len() >= 2 {
                // Nested arguments: (key = "val", flag, ...)
                let inner = &trimmed[1..trimmed.len()-1];
                let nested = parse_nested_args(inner, *span)?;
                args.push(CoreAttrArg::KeyValue {
                    key: key.to_string(),
                    value: AttributeValue::Nested(nested),
                    span: *span,
                });
            } else {
                let value = parse_attr_value(trimmed, *span)?;
                args.push(CoreAttrArg::KeyValue {
                    key: key.to_string(),
                    value,
                    span: *span,
                });
            }
        } else {
            args.push(CoreAttrArg::Flag {
                key: key.to_string(),
                span: *span,
            });
        }
    }
    Ok(AttributeArgs::new(args))
}

/// Parse the inside of a `(...)` nested attribute argument block.
///
/// Handles: `key = "val"`, `key = 42`, `flag`, `key = "a=b"`, etc.
/// Splits on commas, respecting quoted strings.
fn parse_nested_args(content: &str, outer_span: Span) -> Result<AttributeArgs, AttrParseError> {
    let mut pairs: Vec<(&str, Option<&str>, Span)> = vec![];
    let mut current = String::new();
    let mut in_quotes = false;
    let mut quote_char: char = '\0';
    let mut depth: u32 = 0;
    let mut start_offset = 0;

    for (i, ch) in content.char_indices() {
        if in_quotes {
            current.push(ch);
            if ch == quote_char {
                in_quotes = false;
            }
            continue;
        }
        match ch {
            '"' | '\'' => {
                in_quotes = true;
                quote_char = ch;
                current.push(ch);
            }
            '(' => {
                depth += 1;
                current.push(ch);
            }
            ')' => {
                if depth > 0 {
                    depth -= 1;
                    current.push(ch);
                } else {
                    return Err(AttrParseError::at_span(
                        outer_span,
                        "unbalanced parenthesis in nested attribute arguments",
                    ));
                }
            }
            ',' if depth == 0 => {
                flush_nested_token(&mut pairs, &mut current, outer_span, start_offset, i);
                start_offset = i + ch.len_utf8();
            }
            _ => {
                current.push(ch);
            }
        }
    }
    // Flush last token
    if !current.trim().is_empty() {
        flush_nested_token(&mut pairs, &mut current, outer_span, start_offset, content.len());
    }

    if depth != 0 {
        return Err(AttrParseError::at_span(
            outer_span,
            "unbalanced parenthesis in nested attribute arguments",
        ));
    }

    parse_attr_args(&pairs)
}

fn flush_nested_token(
    pairs: &mut Vec<(&str, Option<&str>, Span)>,
    current: &mut String,
    outer_span: Span,
    start_offset: usize,
    end_offset: usize,
) {
    let trimmed = current.trim();
    if trimmed.is_empty() {
        current.clear();
        return;
    }
    // Try to split on first '=' (not inside quotes)
    if let Some(eq_pos) = find_unquoted_eq(trimmed) {
        let key = trimmed[..eq_pos].trim();
        let val = trimmed[eq_pos + 1..].trim();
        // Compute an approximate inner span
        let inner_start = outer_span.start.saturating_add(start_offset).saturating_add(trimmed.len().saturating_sub(current.trim().len()));
        let inner_end = outer_span.start.saturating_add(end_offset);
        let inner_span = Span::new(inner_start, inner_end);
        pairs.push((key, Some(val), inner_span));
    } else {
        // Flag
        let inner_start = outer_span.start.saturating_add(start_offset);
        let inner_end = outer_span.start.saturating_add(end_offset);
        let inner_span = Span::new(inner_start, inner_end);
        pairs.push((trimmed, None, inner_span));
    }
    current.clear();
}

/// Find the first '=' that's not inside quotes.
fn find_unquoted_eq(s: &str) -> Option<usize> {
    let mut in_quotes = false;
    let mut quote_char: char = '\0';
    for (i, ch) in s.char_indices() {
        if in_quotes {
            if ch == quote_char { in_quotes = false; }
            continue;
        }
        if ch == '"' || ch == '\'' {
            in_quotes = true;
            quote_char = ch;
        } else if ch == '=' {
            return Some(i);
        }
    }
    None
}
